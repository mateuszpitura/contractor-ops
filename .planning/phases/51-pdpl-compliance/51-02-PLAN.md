```typescript
import { z } from "zod";
import { router } from "../init.js";
import { tenantProcedure } from "../middleware/tenant.js";
import { requirePermission } from "../middleware/rbac.js";
import { sensitiveActionProcedure } from "../middleware/sensitive.js";
import {
  grantConsent,
  revokeConsent,
  getCurrentConsent,
  getConsentHistory,
  hasRequiredConsents,
  bulkGrantConsent,
} from "../services/consent-record.js";
import { getPrivacyNotice } from "../services/privacy-notice.js";
import {
  grantConsentSchema,
  bulkGrantConsentSchema,
  consentQuerySchema,
  consentPurposeEnum,
} from "@contractor-ops/validators";
import { cached, CacheKeys, CacheTTL } from "../services/cache.js";

export const consentRouter = router({
  // Get privacy notice for current org's jurisdiction
  getPrivacyNotice: tenantProcedure
    .query(async ({ ctx }) => {
      const org = await prisma.organization.findUniqueOrThrow({
        where: { id: ctx.organizationId },
        select: { countryCode: true },
      });
      if (!org.countryCode) return null;
      return getPrivacyNotice(ctx.organizationId, org.countryCode);
    }),

  // Get current consent state for authenticated user
  getCurrentConsent: tenantProcedure
    .query(async ({ ctx }) => {
      return getCurrentConsent(ctx.organizationId, ctx.user.id);
    }),

  // Get consent history for authenticated user (audit)
  getConsentHistory: tenantProcedure
    .input(consentQuerySchema)
    .query(async ({ ctx, input }) => {
      return getConsentHistory(ctx.organizationId, ctx.user.id, input.purpose);
    }),

  // Check if user has all required consents
  hasRequiredConsents: tenantProcedure
    .query(async ({ ctx }) => {
      return hasRequiredConsents(ctx.organizationId, ctx.user.id);
    }),

  // Grant or update consent for a single purpose
  grant: sensitiveActionProcedure
    .input(grantConsentSchema)
    .mutation(async ({ ctx, input }) => {
      if (input.granted) {
        return grantConsent(
          ctx.organizationId, ctx.user.id, input.purpose,
          ctx.ipAddress, ctx.userAgent
        );
      } else {
        return revokeConsent(
          ctx.organizationId, ctx.user.id, input.purpose,
          ctx.ipAddress, ctx.userAgent
        );
      }
    }),

  // Bulk grant consents (onboarding flow)
  bulkGrant: sensitiveActionProcedure
    .input(bulkGrantConsentSchema)
    .mutation(async ({ ctx, input }) => {
      return bulkGrantConsent(
        ctx.organizationId, ctx.user.id, input.consents,
        ctx.ipAddress, ctx.userAgent
      );
    }),

  // Admin: get consent state for any user in org
  adminGetUserConsent: tenantProcedure
    .use(requirePermission({ settings: ["read"] }))
    .input(z.object({ userId: z.string() }))
    .query(async ({ ctx, input }) => {
      return getCurrentConsent(ctx.organizationId, input.userId);
    }),

  // Admin: get consent history for any user in org
  adminGetUserConsentHistory: tenantProcedure
    .use(requirePermission({ settings: ["read"] }))
    .input(z.object({ userId: z.string(), purpose: consentPurposeEnum.optional() }))
    .query(async ({ ctx, input }) => {
      return getConsentHistory(ctx.organizationId, input.userId, input.purpose);
    }),
});
```

Note: Import `prisma` from `@contractor-ops/db` following the existing pattern in other routers. Extract `ctx.ipAddress` and `ctx.userAgent` from the request context — check how sensitiveActionProcedure exposes these (likely via `ctx.headers`). If not directly available, extract from `ctx.headers` using `x-forwarded-for` and `user-agent` headers.

Register in `packages/api/src/root.ts`: add `consent: consentRouter` to the appRouter object, importing from `"./routers/consent.js"`.
