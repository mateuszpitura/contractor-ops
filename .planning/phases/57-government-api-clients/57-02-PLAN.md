```typescript
// packages/gov-api/src/clients/hmrc-vat-client.ts
import type { SecretStore } from '@contractor-ops/secrets';
import { GovApiClient, type GovApiConfig, type GovApiEnvironment } from '../index.js';
import { hmrcOauthTokenSchema, hmrcVatLookupResponseSchema, type HmrcVatLookupResponse } from '../schemas/hmrc-vat.schema.js';

export type HmrcVatLookupResult =
  | { status: 'valid'; raw: HmrcVatLookupResponse; confirmationRef: string | null }
  | { status: 'invalid'; raw: null };

export class HmrcApiError extends Error {
  constructor(message: string, public readonly httpStatus: number, public readonly upstreamCode?: string) {
    super(message);
  }
}

export interface HmrcVatClientDeps {
  config: GovApiConfig;
  environment: GovApiEnvironment;   // 'sandbox' | 'production'
  secretStore: SecretStore;
  platformVrn: string;              // HMRC_PLATFORM_VRN — SECURITY: never user-supplied
  pkgVersion: string;
}

export class HmrcVatClient extends GovApiClient {
  constructor(deps: HmrcVatClientDeps);
  getApiName(): 'hmrc-vat';
  checkVatNumber(
    targetVrn: string,
    opts: { organizationId: string; useVerifiedLookup?: boolean },
  ): Promise<HmrcVatLookupResult>;
}
```

```typescript
// packages/gov-api/src/clients/vies-client.ts
import { GovApiClient, type GovApiConfig, type GovApiEnvironment } from '../index.js';
import { viesLookupResponseSchema, type ViesLookupResponse } from '../schemas/vies.schema.js';

export type ViesLookupResult =
  | { status: 'valid'; raw: ViesLookupResponse; confirmationRef: string | null }
  | { status: 'invalid'; raw: ViesLookupResponse }
  | { status: 'unavailable'; userError: string; raw: ViesLookupResponse };

export class ViesApiError extends Error {
  constructor(message: string, public readonly httpStatus: number) { super(message); }
}

export interface ViesClientDeps {
  config: GovApiConfig;
  environment: GovApiEnvironment;
  requesterMemberStateCode?: string;  // optional platform identity for qualified lookups
  requesterNumber?: string;
}

export class ViesClient extends GovApiClient {
  constructor(deps: ViesClientDeps);
  getApiName(): 'vies';
  checkVatNumber(
    countryCode: string,
    vatNumber: string,
    opts: { organizationId: string; qualified?: boolean },
  ): Promise<ViesLookupResult>;
}
```
