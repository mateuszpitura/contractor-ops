`src/registry.ts`:
```typescript
import type { EInvoiceProfile } from "./types/profile.js";

const profiles = new Map<string, EInvoiceProfile>();

export function registerProfile(profile: EInvoiceProfile): void {
  if (profiles.has(profile.profileId)) {
    throw new Error(`E-invoicing profile already registered: ${profile.profileId}`);
  }
  profiles.set(profile.profileId, profile);
}

export function getProfile(profileId: string): EInvoiceProfile {
  const profile = profiles.get(profileId);
  if (!profile) {
    throw new Error(`Unknown e-invoicing profile: ${profileId}. Available: ${Array.from(profiles.keys()).join(", ")}`);
  }
  return profile;
}

export function listProfiles(): EInvoiceProfile[] {
  return Array.from(profiles.values());
}

export function clearProfiles(): void {
  profiles.clear();
}
```

`src/engine/engine.ts` — EInvoiceEngine class:
- Constructor takes no args (uses registry)
- `generate(profileId: string, invoice: EInvoice): Promise<string>` — looks up profile, calls generate, validates result if profile supports it

- `validate(profileId: string, xml: string): Promise<ValidationResult>` — looks up profile, calls validate
