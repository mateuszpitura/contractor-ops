  1. **Create `apps/web/src/hooks/use-rtl-chart-config.ts`:**
     ```ts
     import { useLocale } from "next-intl";

     interface RtlChartConfig {
       isRtl: boolean;
       xAxisProps: { reversed?: boolean };
       yAxisProps: { orientation?: "left" | "right" };
       legendProps: { align?: "left" | "center" | "right" };
       tooltipProps: { position?: { x: number; y: number } } | Record<string, never>;
     }

     /**
      * Returns RTL-aware configuration props for Recharts components.
      * Detects locale automatically and returns appropriate axis/legend config.
      *
      * Usage:
      *   const { xAxisProps, yAxisProps, legendProps } = useRtlChartConfig();
      *   <XAxis {...xAxisProps} />
      *   <YAxis {...yAxisProps} />
      *   <Legend {...legendProps} />
      */
     export function useRtlChartConfig(): RtlChartConfig {
       const locale = useLocale();
       const isRtl = locale === "ar";

       if (!isRtl) {
         return {
           isRtl: false,
           xAxisProps: {},
           yAxisProps: {},
           legendProps: {},
           tooltipProps: {},
         };
       }

       return {
         isRtl: true,
         xAxisProps: { reversed: true },
         yAxisProps: { orientation: "right" },
         legendProps: { align: "right" },
         tooltipProps: {},
       };
     }
     ```

  2. **Apply to `apps/web/src/components/dashboard/spend-chart.tsx`:**
     - Import: `import { useRtlChartConfig } from "@/hooks/use-rtl-chart-config";`
     - Call hook: `const { xAxisProps, yAxisProps, legendProps } = useRtlChartConfig();`
     - Spread RTL props onto Recharts components:
       - `<XAxis {...xAxisProps} dataKey="..." ... />`
       - `<YAxis {...yAxisProps} ... />`
       - `<Legend {...legendProps} ... />` (if Legend exists)
     - Existing props take precedence — spread RTL props first, then specific props override.

  3. **Apply to `apps/web/src/components/reports/report-chart.tsx`:**

  4. **Apply to `apps/web/src/app/[locale]/(dashboard)/v2/page.tsx`:**
