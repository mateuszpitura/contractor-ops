Add a new `generateZugferdPdf` mutation to `packages/api/src/routers/einvoice.ts`:

```ts
generateZugferdPdf: organizationProcedure
  .input(z.object({ invoiceId: z.string().cuid() }))
  .mutation(async ({ ctx, input }) => {
    const log = logger.child({ module: "einvoice-router", procedure: "generateZugferdPdf", orgId: ctx.activeOrgId, invoiceId: input.invoiceId });

    // 1. Load invoice + lifecycle in one Prisma query, assert org ownership.
    const invoice = await ctx.db.invoice.findFirst({ where: { id: input.invoiceId, organizationId: ctx.activeOrgId }, include: { lines: true, contractor: true, lifecycle: true } });
    if (!invoice) throw new TRPCError({ code: "NOT_FOUND" });

    // 2. Build the EInvoice envelope (same helper used by Phase 61 `finalize`).
    const envelope = buildEInvoiceEnvelope(invoice); // existing helper from phase 61

    // 3. Generate PDF bytes.
    let pdfBytes: Uint8Array;
    try {
      pdfBytes = await generateZugferdPdf({ invoice: envelope });
    } catch (e) {
      log.error({ err: e }, "ZUGFeRD generation failed");
      throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: "ZUGFERD_WRAPPING_FAILED" });
    }

    // 4. Idempotency: compute sha256; if lifecycle.zugferdPdfSha256 already matches, short-circuit and return the existing signed URL.
    const sha = createHash("sha256").update(pdfBytes).digest("hex");
    const key = `einvoice-pdf/${ctx.activeOrgId}/${input.invoiceId}/${sha.slice(0,16)}.pdf`;

    const existing = invoice.lifecycle;
    if (existing?.zugferdPdfSha256 === sha && existing?.zugferdPdfKey) {
      const signedUrl = await signExistingDownload(existing.zugferdPdfKey, 300);
      return { pdfKey: existing.zugferdPdfKey, signedUrl, generatedAt: existing.zugferdGeneratedAt! };
    }

    // 5. Upload to R2.
    const { signedUrl } = await putObjectAndSignDownload({ key, body: Buffer.from(pdfBytes), contentType: "application/pdf", ttlSeconds: 300 });

    // 6. Upsert EInvoiceLifecycle with zugferd fields + append ZUGFERD_GENERATED event (inside $transaction).
    const now = new Date();
    await ctx.db.$transaction(async tx => {
      const lifecycle = await tx.eInvoiceLifecycle.upsert({
        where: { invoiceId: input.invoiceId },
        update: { zugferdPdfKey: key, zugferdPdfSha256: sha, zugferdGeneratedAt: now },
        create: { organizationId: ctx.activeOrgId, invoiceId: input.invoiceId, profileId: "zugferd-de", zugferdPdfKey: key, zugferdPdfSha256: sha, zugferdGeneratedAt: now },
      });
      await tx.eInvoiceLifecycleEvent.create({
        data: { organizationId: ctx.activeOrgId, lifecycleId: lifecycle.id, eventType: "ZUGFERD_GENERATED", actorUserId: ctx.userId, detailsJson: { sha256: sha } },
      });
    });

    return { pdfKey: key, signedUrl, generatedAt: now };
  }),
```
