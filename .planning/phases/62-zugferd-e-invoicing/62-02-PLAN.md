```ts
export type ParserError =
```

```ts
import { PDFDocument, PDFName, PDFDict, PDFString, PDFHexString, PDFArray, PDFStream } from "pdf-lib";
import { logger } from "@contractor-ops/logger";
import { parseXrechnungCii, type ParsedXrechnung, type ParserWarning } from "../xrechnung-de/parser.js";
import { ZUGFERD_ATTACHMENT_FILENAME } from "./constants.js";

export type ParsedZugferd = ParsedXrechnung & {
  rawPdfBuffer: Uint8Array;
  extractedXml: string;
};

export async function parseZugferdPdf(bytes: Uint8Array): Promise<ParsedZugferd> {
  const log = logger.child({ module: "zugferd-de/parser" });
  let doc: PDFDocument;
  try {
    doc = await PDFDocument.load(bytes, { ignoreEncryption: true, throwOnInvalidObject: false });
  } catch (e) {
    log.warn({ err: e }, "Failed to load PDF");
    throw { code: "ZUGFERD_PDF_UNREADABLE", message: (e as Error).message };
  }

  // Locate embedded file via /Names /EmbeddedFiles tree
  const xmlBytes = findEmbeddedFacturXXml(doc);
  if (!xmlBytes) {
    throw { code: "ZUGFERD_NO_XML_ATTACHMENT", message: `PDF contains no ${ZUGFERD_ATTACHMENT_FILENAME} attachment with AFRelationship=Alternative` };
  }

  const extractedXml = new TextDecoder("utf-8").decode(xmlBytes);
  const parsed = parseXrechnungCii(extractedXml); // may throw ZUGFERD_LEVEL_UNSUPPORTED
  return { ...parsed, rawPdfBuffer: bytes, extractedXml };
}

function findEmbeddedFacturXXml(doc: PDFDocument): Uint8Array | null {
  // Priority 1: catalog /Names /EmbeddedFiles → array of [name, fileSpec, ...]
  //   match filename (case-insensitive) === 'factur-x.xml'
  // Priority 2: scan /AF entries (associated files array), match any entry with
```
