```typescript
generateDrvDefenseBundle: tenantProcedure
  .use(requirePermission('contractor:write'))
  .input(z.object({ classificationAssessmentId: z.string().min(1) }))
  .mutation(async ({ input, ctx }) => {
    const assessment = await ctx.db.classificationAssessment.findUniqueOrThrow({
      where: { id: input.classificationAssessmentId },
      include: { contractorAssignment: { include: { contractor: true, organization: true } } },
    }).catch(() => { throw new TRPCError({ code: 'NOT_FOUND' }); });

    if (assessment.status !== 'completed' || assessment.questionsSnapshot === null) {
      throw new TRPCError({ code: 'PRECONDITION_FAILED', message: 'Assessment must be completed with a questions snapshot.' });
    }
    const outcome = assessment.outcome as { kind?: string } | null;
    if (outcome?.kind !== 'SCHEINSELBSTANDIGKEIT') {
      throw new TRPCError({ code: 'PRECONDITION_FAILED', message: 'generateDrvDefenseBundle only applies to Schein (DE) assessments.' });
    }

    // Attestation must exist AND be signed.
    const attestation = await ctx.db.ir35OtherClientAttestation.findUnique({
      where: { contractorAssignmentId: assessment.contractorAssignmentId },
    });
    if (!attestation || !attestation.signedAt) {
      throw new TRPCError({ code: 'PRECONDITION_FAILED', message: 'Other-client attestation must be signed before generating the DRV bundle.' });
    }

    // History: all completed DE assessments for this engagement, newest first.
    const priorAssessments = await ctx.db.classificationAssessment.findMany({
      where: { contractorAssignmentId: assessment.contractorAssignmentId, countryCode: 'DE', status: 'completed' },
      orderBy: { completedAt: 'desc' },
    });

    // Same-tenant cross-reference.
    const crossReference = await ctx.db.contractorAssignment.findMany({
      where: {
        contractorId: assessment.contractorAssignment.contractor.id,
        organizationId: ctx.organizationId,
        id: { not: assessment.contractorAssignmentId },
      },
      select: { id: true, displayName: true, projectScope: true, startDate: true, endDate: true },
      orderBy: { startDate: 'desc' },
    });

    const renderedAt = assessment.completedAt ?? new Date(0);
    const buf = await renderToBuffer(
      <DRVDefenseBundleDocument
        assessment={assessment as never}
        priorAssessments={priorAssessments as never}
        engagement={{
          id: assessment.contractorAssignment.id,
          displayName: assessment.contractorAssignment.displayName ?? 'Engagement',
          projectScope: assessment.contractorAssignment.projectScope ?? '',
          dayRate: assessment.contractorAssignment.dayRate ?? 0,
          dayRateCurrency: assessment.contractorAssignment.dayRateCurrency ?? 'EUR',
          startDate: assessment.contractorAssignment.startDate,
          endDate: assessment.contractorAssignment.endDate,
        }}
        contractor={{
          id: assessment.contractorAssignment.contractor.id,
          displayName: assessment.contractorAssignment.contractor.displayName,
        }}
        organization={{
          id: assessment.contractorAssignment.organization.id,
          name: assessment.contractorAssignment.organization.name,
          countryCode: assessment.contractorAssignment.organization.countryCode,
        }}
        attestation={{ statementText: attestation.statementText, signedName: attestation.signedName, signedAt: attestation.signedAt }}
        crossReference={crossReference.map(r => ({
          assignmentId: r.id,
          clientDisplayName: r.displayName ?? '',
          startDate: r.startDate,
          endDate: r.endDate,
          projectScope: r.projectScope ?? '',
        }))}
        renderedAt={renderedAt}
      />,
    );

    const sha256Hash = createHash('sha256').update(buf).digest('hex');
    const key = buildClassificationDocumentKey({
      organizationId: ctx.organizationId,
      classificationAssessmentId: assessment.id,
      kind: 'DRV_DEFENSE_BUNDLE',
      ruleSetVersion: assessment.ruleSetVersion,
      sha256: sha256Hash,
    });
    const rendererVersion = `@react-pdf/renderer@${REACT_PDF_VERSION}+${DRV_RENDERER_SLUG}@${DRV_TEMPLATE_VERSION}`;

    const { signedUrl, expiresInSeconds } = await putObjectAndSignDownload({
      key, body: buf, contentType: 'application/pdf', ttlSeconds: PDF_TTL_SECONDS,
      downloadFilename: `DRV-Defense-${assessment.contractorAssignment.contractor.displayName}.pdf`.replace(/[/\\]/g, '-'),
    });

    let row;
    try {
      row = await ctx.db.classificationDocument.create({
        data: {
          organizationId: ctx.organizationId,
          classificationAssessmentId: assessment.id,
          kind: 'DRV_DEFENSE_BUNDLE',
          pdfKey: key,
          sha256Hash,
          byteSize: buf.byteLength,
          rendererVersion,
          ruleSetVersion: assessment.ruleSetVersion,
          generatedByUserId: ctx.session.user.id,
        },
      });
    } catch (err) {
      await deleteObject(key).catch(() => undefined);
      throw err;
    }

    return { url: signedUrl, expiresInSeconds, documentId: row.id, byteSize: buf.byteLength, sha256Hash };
  }),
```
