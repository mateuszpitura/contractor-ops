```typescript
import { TRPCError } from '@trpc/server';
import { z } from 'zod';

import { createTRPCRouter } from '../init.js';
import { requirePermission } from '../middleware/rbac.js';
import { tenantProcedure } from '../middleware/tenant.js';

const chainRoleSchema = z.enum(['CLIENT', 'AGENCY', 'PSC', 'WORKER']);

const participantDtoSelect = {
  id: true,
  organizationId: true,
  contractorAssignmentId: true,
  role: true,
  orderIndex: true,
  displayName: true,
  contactEmail: true,
  linkedOrganizationId: true,
  linkedContractorId: true,
  sdsDeliveredAt: true,
  sdsDeliveredNote: true,
  sdsAcknowledgedAt: true,
  sdsAcknowledgedNote: true,
  createdAt: true,
  updatedAt: true,
} as const;

const upsertInputSchema = z.object({
  id: z.string().optional(),
  contractorAssignmentId: z.string().min(1),
  role: chainRoleSchema,
  orderIndex: z.number().int().min(0).max(99),
  displayName: z.string().min(1).max(200),
  contactEmail: z.string().email().max(320).optional(),
  linkedContractorId: z.string().optional(),
});

const reorderInputSchema = z.object({
  contractorAssignmentId: z.string().min(1),
  orderedIds: z.array(z.string().min(1)).min(1).max(20),
});

const markDeliveredInputSchema = z.object({
  id: z.string().min(1),
  note: z.string().max(500).optional(),
});

export const ir35ChainRouter = createTRPCRouter({
  listByEngagement: tenantProcedure
    .use(requirePermission('contractor:read'))
    .input(z.object({ contractorAssignmentId: z.string().min(1) }))
    .query(async ({ input, ctx }) => {
      const engagement = await ctx.db.contractorAssignment.findUniqueOrThrow({
        where: { id: input.contractorAssignmentId },
        include: { contractor: true, organization: true },
      }).catch(() => {
        throw new TRPCError({ code: 'NOT_FOUND', message: 'Engagement not found' });
      });

      const existing = await ctx.db.ir35ChainParticipant.findMany({
        where: { contractorAssignmentId: input.contractorAssignmentId },
        orderBy: { orderIndex: 'asc' },
        select: participantDtoSelect,
      });
      if (existing.length > 0) return existing;

      // Auto-seed CLIENT + WORKER for GB engagements only.
      if (engagement.organization.countryCode !== 'GB' && engagement.countryCode !== 'GB') return [];

      return ctx.db.$transaction(async (tx) => {
        const race = await tx.ir35ChainParticipant.findMany({
          where: { contractorAssignmentId: input.contractorAssignmentId },
          orderBy: { orderIndex: 'asc' },
          select: participantDtoSelect,
        });
        if (race.length > 0) return race;
        await tx.ir35ChainParticipant.createMany({
          data: [
            {
              organizationId: ctx.organizationId,
              contractorAssignmentId: input.contractorAssignmentId,
              role: 'CLIENT',
              orderIndex: 0,
              displayName: engagement.organization.name,
              linkedOrganizationId: ctx.organizationId,
            },
            {
              organizationId: ctx.organizationId,
              contractorAssignmentId: input.contractorAssignmentId,
              role: 'WORKER',
              orderIndex: 1,
              displayName: engagement.contractor.displayName,
              linkedContractorId: engagement.contractor.id,
            },
          ],
        });
        return tx.ir35ChainParticipant.findMany({
          where: { contractorAssignmentId: input.contractorAssignmentId },
          orderBy: { orderIndex: 'asc' },
          select: participantDtoSelect,
        });
      });
    }),

  upsertParticipant: tenantProcedure
    .use(requirePermission('contractor:write'))
    .input(upsertInputSchema)
    .mutation(async ({ input, ctx }) => {
      // Same-tenant check on linkedContractorId.
      if (input.linkedContractorId) {
        await ctx.db.contractor.findUniqueOrThrow({
          where: { id: input.linkedContractorId },
        }).catch(() => {
          throw new TRPCError({ code: 'NOT_FOUND', message: 'Linked contractor not found in this organization.' });
        });
      }

      const linkedOrganizationId = input.role === 'CLIENT' ? ctx.organizationId : null;
      const linkedContractorId = input.role === 'CLIENT' ? null : (input.linkedContractorId ?? null);

      const data = {
        contractorAssignmentId: input.contractorAssignmentId,
        role: input.role,
        orderIndex: input.orderIndex,
        displayName: input.displayName,
        contactEmail: input.contactEmail ?? null,
        linkedOrganizationId,
        linkedContractorId,
      };

      if (input.id) {
        return ctx.db.ir35ChainParticipant.update({
          where: { id: input.id },
          data,
          select: participantDtoSelect,
        });
      }
      return ctx.db.ir35ChainParticipant.create({
        data: { ...data, organizationId: ctx.organizationId },
        select: participantDtoSelect,
      });
    }),

  reorderParticipants: tenantProcedure
    .use(requirePermission('contractor:write'))
    .input(reorderInputSchema)
    .mutation(async ({ input, ctx }) => {
      const current = await ctx.db.ir35ChainParticipant.findMany({
        where: { contractorAssignmentId: input.contractorAssignmentId },
        select: { id: true },
      });
      const currentIdSet = new Set(current.map((p) => p.id));
      if (current.length !== input.orderedIds.length ||
          !input.orderedIds.every((id) => currentIdSet.has(id))) {
        throw new TRPCError({ code: 'BAD_REQUEST', message: 'orderedIds must match the participants of this engagement exactly.' });
      }
      await ctx.db.$transaction(
        input.orderedIds.map((id, i) =>
          ctx.db.ir35ChainParticipant.update({ where: { id }, data: { orderIndex: i } }),
        ),
      );
      return { success: true as const };
    }),

  markDelivered: tenantProcedure
    .use(requirePermission('contractor:write'))
    .input(markDeliveredInputSchema)
    .mutation(async ({ input, ctx }) => {
      return ctx.db.ir35ChainParticipant.update({
        where: { id: input.id },
        data: { sdsDeliveredAt: new Date(), sdsDeliveredNote: input.note ?? null },
        select: participantDtoSelect,
      });
    }),

  markAcknowledged: tenantProcedure
    .use(requirePermission('contractor:write'))
    .input(markDeliveredInputSchema)
    .mutation(async ({ input, ctx }) => {
      return ctx.db.ir35ChainParticipant.update({
        where: { id: input.id },
        data: { sdsAcknowledgedAt: new Date(), sdsAcknowledgedNote: input.note ?? null },
        select: participantDtoSelect,
      });
    }),

  removeParticipant: tenantProcedure
    .use(requirePermission('contractor:write'))
    .input(z.object({ id: z.string().min(1) }))
    .mutation(async ({ input, ctx }) => {
      const row = await ctx.db.ir35ChainParticipant.findUniqueOrThrow({ where: { id: input.id } })
        .catch(() => { throw new TRPCError({ code: 'NOT_FOUND' }); });
      if (row.role === 'CLIENT' || row.role === 'WORKER') {
        throw new TRPCError({ code: 'BAD_REQUEST', message: 'CLIENT and WORKER rows are auto-populated and cannot be removed.' });
      }
      await ctx.db.ir35ChainParticipant.delete({ where: { id: input.id } });
      return { success: true as const };
    }),
});
```

```typescript
import { TRPCError } from '@trpc/server';
import { z } from 'zod';

import { createTRPCRouter } from '../init.js';
import { requirePermission } from '../middleware/rbac.js';
import { tenantProcedure } from '../middleware/tenant.js';

const attestationDtoSelect = {
  id: true,
  organizationId: true,
  contractorAssignmentId: true,
  statementText: true,
  signedName: true,
  signedAt: true,
  createdAt: true,
  updatedAt: true,
} as const;

export const ir35AttestationRouter = createTRPCRouter({
  getForEngagement: tenantProcedure
    .use(requirePermission('contractor:read'))
    .input(z.object({ contractorAssignmentId: z.string().min(1) }))
    .query(async ({ input, ctx }) => {
      return ctx.db.ir35OtherClientAttestation.findUnique({
        where: { contractorAssignmentId: input.contractorAssignmentId },
        select: attestationDtoSelect,
      });
    }),

  upsert: tenantProcedure
    .use(requirePermission('contractor:write'))
    .input(z.object({
      contractorAssignmentId: z.string().min(1),
      statementText: z.string().min(1).max(4000),
      signedName: z.string().min(1).max(120),
    }))
    .mutation(async ({ input, ctx }) => {
      return ctx.db.ir35OtherClientAttestation.upsert({
        where: { contractorAssignmentId: input.contractorAssignmentId },
        update: { statementText: input.statementText, signedName: input.signedName, signedAt: new Date() },
        create: {
          organizationId: ctx.organizationId,
          contractorAssignmentId: input.contractorAssignmentId,
          statementText: input.statementText,
          signedName: input.signedName,
          signedAt: new Date(),
        },
        select: attestationDtoSelect,
      });
    }),

  getPlatformCrossReference: tenantProcedure
    .use(requirePermission('contractor:read'))
    .input(z.object({ contractorAssignmentId: z.string().min(1) }))
    .query(async ({ input, ctx }) => {
      const engagement = await ctx.db.contractorAssignment.findUniqueOrThrow({
        where: { id: input.contractorAssignmentId },
        select: { id: true, contractorId: true },
      }).catch(() => { throw new TRPCError({ code: 'NOT_FOUND' }); });

      const siblings = await ctx.db.contractorAssignment.findMany({
        where: {
          contractorId: engagement.contractorId,
          // organizationId redundantly enforced (Prisma extension already scopes)
          organizationId: ctx.organizationId,
          id: { not: input.contractorAssignmentId },
```

```tsx
'use client';
import { useTranslations } from 'next-intl';
import { useState } from 'react';
import { trpc } from '@/lib/trpc';
import { Button } from '@/components/ui/button';
import { Alert } from '@/components/ui/alert';
import { ChainParticipantRow } from './chain-participant-row.js';
import { AddParticipantDialog } from './add-participant-dialog.js';
import { MarkDeliveredDialog } from './mark-delivered-dialog.js';

interface Props { engagementId: string }

export function Ir35ChainPanel({ engagementId }: Props) {
  const t = useTranslations('Classification.chain');
  const utils = trpc.useContext();
  const { data: participants, error } = trpc.ir35Chain.listByEngagement.useQuery({ contractorAssignmentId: engagementId });
  const reorder = trpc.ir35Chain.reorderParticipants.useMutation({
    onSuccess: () => utils.ir35Chain.listByEngagement.invalidate({ contractorAssignmentId: engagementId }),
  });

  const [addOpen, setAddOpen] = useState(false);
  const [markDialog, setMarkDialog] = useState<{ id: string; kind: 'delivered' | 'acknowledged' } | null>(null);

  if (error) return <Alert variant="destructive">{error.message}</Alert>;
  if (!participants) return <p>{t('loading')}</p>;

  function moveRow(i: number, dir: 1 | -1) {
    if (!participants) return;
    const j = i + dir;
    if (j < 0 || j >= participants.length) return;
    const next = [...participants];
    [next[i], next[j]] = [next[j], next[i]];
    reorder.mutate({ contractorAssignmentId: engagementId, orderedIds: next.map(p => p.id) });
  }

  return (
    <section aria-labelledby="ir35-chain-heading" className="rounded-lg border bg-card p-6">
      <header className="mb-4 flex items-center justify-between">
        <div>
          <h2 id="ir35-chain-heading" className="text-lg font-semibold">{t('title')}</h2>
          <p className="mt-1 text-sm text-muted-foreground">{t('subtitle')}</p>
        </div>
        <Button onClick={() => setAddOpen(true)}>{t('addParticipant')}</Button>
      </header>

      <table className="w-full text-sm">
        <thead>
          <tr className="border-b text-left text-xs uppercase text-muted-foreground">
            <th scope="col" className="p-2">{t('columns.role')}</th>
            <th scope="col" className="p-2">{t('columns.name')}</th>
            <th scope="col" className="p-2">{t('columns.email')}</th>
            <th scope="col" className="p-2">{t('columns.delivered')}</th>
            <th scope="col" className="p-2">{t('columns.acknowledged')}</th>
            <th scope="col" className="p-2"><span className="sr-only">{t('columns.actions')}</span></th>
          </tr>
        </thead>
        <tbody>
          {participants.map((p, i) => (
            <ChainParticipantRow
              key={p.id}
              participant={p}
              isFirst={i === 0}
              isLast={i === participants.length - 1}
              onMoveUp={() => moveRow(i, -1)}
              onMoveDown={() => moveRow(i, 1)}
              onMarkDelivered={() => setMarkDialog({ id: p.id, kind: 'delivered' })}
              onMarkAcknowledged={() => setMarkDialog({ id: p.id, kind: 'acknowledged' })}
            />
          ))}
        </tbody>
      </table>

      <AddParticipantDialog open={addOpen} onOpenChange={setAddOpen} engagementId={engagementId} nextOrderIndex={participants.length} />
      {markDialog && (
        <MarkDeliveredDialog
          open
          onOpenChange={(o) => o ? undefined : setMarkDialog(null)}
          participantId={markDialog.id}
          kind={markDialog.kind}
          engagementId={engagementId}
        />
      )}
    </section>
  );
}
```

```tsx
'use client';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';
import { useTranslations } from 'next-intl';
import { trpc } from '@/lib/trpc';
import { Button } from '@/components/ui/button';
import { Textarea } from '@/components/ui/textarea';
import { Input } from '@/components/ui/input';
import { Alert } from '@/components/ui/alert';

const schema = z.object({
  statementText: z.string().min(1).max(4000),
  signedName: z.string().min(1).max(120),
});

type FormValues = z.infer<typeof schema>;

interface Props { engagementId: string }

export function OtherClientAttestationForm({ engagementId }: Props) {
  const t = useTranslations('Classification.attestation');
  const utils = trpc.useContext();
  const { data: existing } = trpc.ir35Attestation.getForEngagement.useQuery({ contractorAssignmentId: engagementId });
  const upsert = trpc.ir35Attestation.upsert.useMutation({
    onSuccess: () => utils.ir35Attestation.getForEngagement.invalidate({ contractorAssignmentId: engagementId }),
  });

  const form = useForm<FormValues>({
    resolver: zodResolver(schema),
    defaultValues: { statementText: existing?.statementText ?? '', signedName: existing?.signedName ?? '' },
    values: existing ? { statementText: existing.statementText, signedName: existing.signedName } : undefined,
  });
  const statementText = form.watch('statementText');

  return (
    <form
      onSubmit={form.handleSubmit((v) => upsert.mutate({ contractorAssignmentId: engagementId, ...v }))}
      className="space-y-4"
      aria-labelledby="other-client-attestation-heading"
    >
      <div>
        <h3 id="other-client-attestation-heading" className="text-base font-semibold">{t('title')}</h3>
        <p className="text-sm text-muted-foreground">{t('subtitle')}</p>
      </div>

      <div>
        <label htmlFor="statement-text" className="text-sm font-medium">{t('statementLabel')}</label>
        <Textarea id="statement-text" rows={6} maxLength={4000} {...form.register('statementText')} aria-describedby="statement-counter statement-error" />
        <div id="statement-counter" className="mt-1 text-right text-xs text-muted-foreground">
          {statementText?.length ?? 0} / 4000
        </div>
        {form.formState.errors.statementText && (
          <p id="statement-error" className="text-xs text-destructive">{form.formState.errors.statementText.message}</p>
        )}
      </div>

      <div>
        <label htmlFor="signed-name" className="text-sm font-medium">{t('signedNameLabel')}</label>
        <Input id="signed-name" maxLength={120} {...form.register('signedName')} aria-describedby="signed-name-error" />
        {form.formState.errors.signedName && (
          <p id="signed-name-error" className="text-xs text-destructive">{form.formState.errors.signedName.message}</p>
        )}
      </div>

      {upsert.error && <Alert variant="destructive">{upsert.error.message}</Alert>}

      {existing?.signedAt && (
        <p className="text-xs text-muted-foreground">
          {t('lastSignedOn', { date: existing.signedAt.toISOString().slice(0, 10) })}
        </p>
      )}

      <Button type="submit" disabled={upsert.isPending || !form.formState.isValid}>
        {t('submit')}
      </Button>
    </form>
  );
}
```
