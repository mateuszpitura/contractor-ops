export function IR35SDSDocument({ assessment, engagement, contractor, organization, renderedAt }: IR35SDSDocumentProps) {
  if (assessment.outcome?.kind !== 'IR35') {
    throw new Error('IR35SDSDocument: assessment.outcome must be of kind IR35');

            <Text style={styles.engagementKey}>Worker</Text>
            <Text style={styles.engagementValue}>{contractor.displayName}</Text>
          </View>
          <View style={styles.engagementKV}>
            <Text style={styles.engagementKey}>Engagement</Text>
            <Text style={styles.engagementValue}>{engagement.displayName}</Text>
          </View>
          <View style={styles.engagementKV}>
            <Text style={styles.engagementKey}>Day rate</Text>
            <Text style={styles.engagementValue}>{engagement.dayRate} {engagement.dayRateCurrency}</Text>
          </View>
          <View style={styles.engagementKV}>
            <Text style={styles.engagementKey}>Start date</Text>
            <Text style={styles.engagementValue}>{engagement.startDate.toISOString().slice(0,10)}</Text>
          </View>

            <Text style={styles.engagementKey}>End date</Text>
            <Text style={styles.engagementValue}>{engagement.endDate?.toISOString().slice(0,10) ?? 'Ongoing'}</Text>

                      {q.caseLawCitation && <Text style={styles.evidenceCitation}>{q.caseLawCitation}</Text>}
                    </View>
                  );
                })}
              </View>
            </View>
          );
        })}

        <Text style={styles.footer} fixed>Status Determination Statement · {organization.name} · Rendered {renderedAt.toISOString().slice(0,10)}</Text>
        <Text style={styles.pageNumber} fixed render={({ pageNumber, totalPages }) => `${pageNumber} / ${totalPages}`} />
      </Page>

      {/* Final page — dispute + disclaimer */}
      <Page size="A4" style={styles.page} wrap>
        <View style={styles.disputeBlock}>
          <Text style={styles.disputeTitle}>Challenging this determination</Text>

        <Text style={styles.pageNumber} fixed render={({ pageNumber, totalPages }) => `${pageNumber} / ${totalPages}`} />
      </Page>
    </Document>
  );
}

function formatAnswer(answer: { value?: unknown; rawScore?: number; isNotApplicable?: boolean } | undefined): string {
  if (!answer) return 'No answer';
  if (answer.isNotApplicable) return 'Not applicable';
  if (typeof answer.value === 'string') return answer.value;
  if (typeof answer.value === 'number') return String(answer.value);
  if (typeof answer.rawScore === 'number') return `Score: ${answer.rawScore}`;
  return JSON.stringify(answer.value ?? answer.rawScore ?? '');
}

import { createHash } from 'node:crypto';

import { TRPCError } from '@trpc/server';
import { renderToBuffer } from '@react-pdf/renderer';
import { z } from 'zod';

import { createTRPCRouter } from '../init.js';
import { tenantProcedure } from '../middleware/tenant.js';
import { requirePermission } from '../middleware/rbac.js';
import { RENDERER_SLUG, TEMPLATE_VERSION, IR35SDSDocument } from '../pdf-templates/ir35-sds.js';
import { buildClassificationDocumentKey } from '../services/classification-document-keys.js';
import { deleteObject, putObjectAndSignDownload, signExistingDownload } from '../services/r2.js';

// Bump on @react-pdf/renderer upgrade. Embedded into ClassificationDocument.rendererVersion for audit forensics.
const REACT_PDF_VERSION = '3.4.5' as const;
const PDF_TTL_SECONDS = 300;

const generateSdsInputSchema = z.object({
  classificationAssessmentId: z.string().min(1),
});

const getDownloadUrlInputSchema = z.object({
  classificationDocumentId: z.string().min(1),
});

const listByEngagementInputSchema = z.object({
  contractorAssignmentId: z.string().min(1),
});

export const classificationDocumentRouter = createTRPCRouter({
  generateSds: tenantProcedure
    .use(requirePermission('contractor:write'))
    .input(generateSdsInputSchema)
    .mutation(async ({ input, ctx }) => {
      // 1. Load assessment (org-scoped via Prisma extension).
      const assessment = await ctx.db.classificationAssessment
        .findUniqueOrThrow({
          where: { id: input.classificationAssessmentId },
          include: {
            contractorAssignment: {
              include: { contractor: true, organization: true },
            },
          },
        })
        .catch(() => {
          throw new TRPCError({ code: 'NOT_FOUND', message: 'Assessment not found' });
        });

      // 2. Preconditions.
      if (assessment.status !== 'completed' || assessment.questionsSnapshot === null) {
        throw new TRPCError({
          code: 'PRECONDITION_FAILED',
          message: 'Assessment must be completed with a questions snapshot before generating an SDS.',
        });
      }
      const outcome = assessment.outcome as { kind?: string } | null;
      if (outcome?.kind !== 'IR35') {
        throw new TRPCError({
          code: 'PRECONDITION_FAILED',
          message: 'generateSds only applies to IR35 assessments (countryCode=GB).',
        });
      }

      // 3. Render PDF bytes.
      const renderedAt = assessment.completedAt ?? new Date(0); // stable; byte-equal across re-renders
      const buf = await renderToBuffer(
        <IR35SDSDocument
          assessment={assessment as never}
          engagement={{
            id: assessment.contractorAssignment.id,
            displayName: assessment.contractorAssignment.displayName ?? 'Engagement',
            projectScope: assessment.contractorAssignment.projectScope ?? '',
            dayRate: assessment.contractorAssignment.dayRate ?? 0,
            dayRateCurrency: assessment.contractorAssignment.dayRateCurrency ?? 'GBP',
            startDate: assessment.contractorAssignment.startDate,
            endDate: assessment.contractorAssignment.endDate,
          }}
          contractor={{
            id: assessment.contractorAssignment.contractor.id,
            displayName: assessment.contractorAssignment.contractor.displayName,
          }}
          organization={{
            id: assessment.contractorAssignment.organization.id,
            name: assessment.contractorAssignment.organization.name,
            countryCode: assessment.contractorAssignment.organization.countryCode,
          }}
          renderedAt={renderedAt}
        />,
      );

      // 4. Hash + key.
      const sha256Hash = createHash('sha256').update(buf).digest('hex');
      const key = buildClassificationDocumentKey({
        organizationId: ctx.organizationId,
        classificationAssessmentId: assessment.id,
        kind: 'SDS',
        ruleSetVersion: assessment.ruleSetVersion,
        sha256: sha256Hash,
      });
      const rendererVersion = `@react-pdf/renderer@${REACT_PDF_VERSION}+${RENDERER_SLUG}@${TEMPLATE_VERSION}`;

      // 5. Upload.
      const { signedUrl, expiresInSeconds } = await putObjectAndSignDownload({
        key,
        body: buf,
        contentType: 'application/pdf',
        ttlSeconds: PDF_TTL_SECONDS,
        downloadFilename: `SDS-${assessment.contractorAssignment.contractor.displayName}-${assessment.contractorAssignment.displayName ?? 'engagement'}.pdf`.replace(/[/\\]/g, '-'),
      });

      // 6. Insert row — rollback R2 on failure.
      let row;
      try {
        row = await ctx.db.classificationDocument.create({
          data: {
            organizationId: ctx.organizationId,
            classificationAssessmentId: assessment.id,
            kind: 'SDS',
            pdfKey: key,
            sha256Hash,
            byteSize: buf.byteLength,
            rendererVersion,
            ruleSetVersion: assessment.ruleSetVersion,
            generatedByUserId: ctx.session.user.id,
          },
        });
      } catch (err) {
        await deleteObject(key).catch(() => undefined);
        throw err;
      }

      return {
        url: signedUrl,
        expiresInSeconds,
        documentId: row.id,
        byteSize: buf.byteLength,
        sha256Hash,
      };
    }),

  getDownloadUrl: tenantProcedure
    .use(requirePermission('contractor:read'))
    .input(getDownloadUrlInputSchema)
    .query(async ({ input, ctx }) => {
      const doc = await ctx.db.classificationDocument
        .findUniqueOrThrow({
          where: { id: input.classificationDocumentId },
          include: {
            classificationAssessment: { include: { contractorAssignment: { include: { contractor: true } } } },
          },
        })
        .catch(() => {
          throw new TRPCError({ code: 'NOT_FOUND', message: 'Document not found' });
        });

      const downloadFilename =
        `${doc.kind === 'SDS' ? 'SDS' : 'DRV-Defense'}-${doc.classificationAssessment.contractorAssignment.contractor.displayName}.pdf`.replace(/[/\\]/g, '-');
      const { signedUrl, expiresInSeconds } = await signExistingDownload(doc.pdfKey, PDF_TTL_SECONDS, downloadFilename);

      return {
        url: signedUrl,
        expiresInSeconds,
        kind: doc.kind,
        generatedAt: doc.generatedAt,
        byteSize: doc.byteSize,
        sha256Hash: doc.sha256Hash,
      };
    }),

  listByEngagement: tenantProcedure
    .use(requirePermission('contractor:read'))
    .input(listByEngagementInputSchema)
    .query(async ({ input, ctx }) => {
      const docs = await ctx.db.classificationDocument.findMany({
        where: { contractorAssignmentId: undefined, classificationAssessment: { contractorAssignmentId: input.contractorAssignmentId } },
        orderBy: { generatedAt: 'desc' },
        select: {
          id: true,
          kind: true,
          generatedAt: true,
          byteSize: true,
          rendererVersion: true,
          ruleSetVersion: true,
        },
      });
      return docs;
    }),
});
