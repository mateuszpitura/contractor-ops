```typescript
import { S3Client, PutObjectCommand, GetObjectCommand, HeadObjectCommand, DeleteObjectCommand } from "@aws-sdk/client-s3";
import { getSignedUrl } from "@aws-sdk/s3-request-presigner";

let r2Client: S3Client | null = null;

export function createR2Client(): S3Client {
  if (!r2Client) {
    r2Client = new S3Client({
      region: "auto",
      endpoint: `https://${process.env.R2_ACCOUNT_ID}.r2.cloudflarestorage.com`,
      credentials: {
        accessKeyId: process.env.R2_ACCESS_KEY_ID!,
        secretAccessKey: process.env.R2_SECRET_ACCESS_KEY!,
      },
    });
  }
  return r2Client;
}

export function generateStorageKey(orgId: string, docId: string, filename: string): string {
  const ext = filename.split(".").pop() ?? "";
  return `orgs/${orgId}/documents/${docId}${ext ? `.${ext}` : ""}`;
}

export async function createPresignedUploadUrl(key: string, contentType: string, expiresIn = 300): Promise<string> {
  const client = createR2Client();
  const command = new PutObjectCommand({
    Bucket: process.env.R2_BUCKET_NAME!,
    Key: key,
    ContentType: contentType,
  });
  return getSignedUrl(client, command, { expiresIn });
}

export async function createPresignedDownloadUrl(key: string, expiresIn = 900): Promise<string> {
  const client = createR2Client();
  const command = new GetObjectCommand({
    Bucket: process.env.R2_BUCKET_NAME!,
    Key: key,
  });
  return getSignedUrl(client, command, { expiresIn });
}

export async function headObject(key: string) {
  const client = createR2Client();
  const command = new HeadObjectCommand({
    Bucket: process.env.R2_BUCKET_NAME!,
    Key: key,
  });
  return client.send(command);
}

export async function deleteObject(key: string) {
  const client = createR2Client();
  const command = new DeleteObjectCommand({
    Bucket: process.env.R2_BUCKET_NAME!,
    Key: key,
  });
  return client.send(command);
}
```

```typescript
import { fileTypeFromBuffer } from "file-type";

export const ALLOWED_MIMES = new Set([
  "application/pdf",
  "application/vnd.openxmlformats-officedocument.wordprocessingml.document", // DOCX
  "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",       // XLSX
  "image/png",
  "image/jpeg",
]);

export const ALLOWED_EXTENSIONS = new Set(["pdf", "docx", "xlsx", "png", "jpg", "jpeg"]);

export const MAX_FILE_SIZE_BYTES = 25 * 1024 * 1024; // 25 MB

export async function validateMimeType(buffer: Buffer): Promise<{ valid: boolean; detectedMime: string | undefined }> {
  const result = await fileTypeFromBuffer(buffer);
  const detectedMime = result?.mime;
  return {
    valid: detectedMime !== undefined && ALLOWED_MIMES.has(detectedMime),
    detectedMime,
  };
}

export function isAllowedMimeType(mimeType: string): boolean {
  return ALLOWED_MIMES.has(mimeType);
}
```

**3. Create `packages/api/src/services/virus-scanner.ts`:**

```typescript
import NodeClam from "clamscan";

let clamInstance: Awaited<ReturnType<InstanceType<typeof NodeClam>["init"]>> | null = null;

async function getClamInstance() {
  if (!clamInstance) {
    const clam = new NodeClam();
    clamInstance = await clam.init({
      clamdscan: {
        host: process.env.CLAMAV_HOST ?? "127.0.0.1",
        port: Number(process.env.CLAMAV_PORT ?? 3310),
        timeout: 60000,
      },
      preference: "clamdscan",
    });
  }
  return clamInstance;
}

export async function scanBuffer(buffer: Buffer): Promise<{ isClean: boolean; virusName?: string }> {
  try {
    const clam = await getClamInstance();
    const { Readable } = await import("node:stream");
    const stream = Readable.from(buffer);
    const { isInfected, viruses } = await clam.scanStream(stream);
    return {
      isClean: !isInfected,
      virusName: viruses?.[0],
    };
  } catch (error) {
    console.error("[virus-scanner] Scan failed:", error);
    throw error;
  }
}

export async function isClamAvailable(): Promise<boolean> {
  try {
    await getClamInstance();
    return true;
  } catch {
    return false;
  }
}
```

Create `packages/api/src/routers/document.ts` following contractor.ts patterns:

**Procedures** (all use `tenantProcedure` + `requirePermission({ documents: ["read"|"create"|"update"|"delete"] })`):

1. **requestUpload** — mutation, input: documentRequestUploadSchema.
   - Validate mimeType is in ALLOWED_MIMES using `isAllowedMimeType()` from mime-validator. Throw TRPCError BAD_REQUEST if not.
   - Create Document record in PENDING state: status=ACTIVE, virusScanStatus=PENDING, checksumSha256="" (placeholder until confirmUpload), source=USER_UPLOAD, uploadedByUserId from ctx.session.user.id.
   - Generate storageKey using `generateStorageKey(orgId, doc.id, filename)`.
   - Update document with storageKey.
   - Generate presigned upload URL via `createPresignedUploadUrl(storageKey, mimeType, 300)`.
   - If entityType and entityId provided, create DocumentLink record.
   - Return `{ documentId: doc.id, uploadUrl, storageKey }`.

2. **confirmUpload** — mutation, input: documentConfirmUploadSchema.
   - Verify object exists in R2 via `headObject(doc.storageKey)`. Throw NOT_FOUND if missing.
   - Update document fileSizeBytes from R2 head response ContentLength.
   - Trigger async virus scan: fire-and-forget `scanAndUpdate(doc.id, doc.storageKey)` function that:
     a. Fetches first 4100 bytes from R2 via Range request for MIME validation
     b. Validates MIME type via `validateMimeType(buffer)`. If invalid, set virusScanStatus=FAILED.
     c. If ClamAV available (`isClamAvailable()`), scan and update virusScanStatus accordingly. If unavailable, set FAILED (never skip).
   - Return plain(updated document).

3. **getDownloadUrl** — query, input: z.object({ documentId: z.string() }).
   - Verify document exists and belongs to org. Check virusScanStatus is not INFECTED (throw FORBIDDEN if infected).
   - Generate presigned download URL via `createPresignedDownloadUrl(doc.storageKey, 900)` (15-minute expiry).
   - Return `{ url, expiresIn: 900 }`.

4. **list** — query, input: documentListSchema.
   - If entityType and entityId provided, query via DocumentLink join.
   - Filter by documentType[], status[].
   - Include uploadedBy user (select: id, name).
   - Pagination with skip/take.
   - Return `{ items: plain(documents), totalCount, page, pageSize }`.

5. **uploadNewVersion** — mutation, input: documentVersionUploadSchema.
   - Wrap in `prisma.$transaction()`:
     a. Find existing document, verify it exists and status is ACTIVE.
     b. Mark existing document as SUPERSEDED.
     c. Create new Document record (copy documentType, visibility, source from existing).
     d. Copy all DocumentLink records from existing document to new document.
     e. Generate storageKey and presigned upload URL for new document.
   - Return `{ documentId: newDoc.id, uploadUrl, storageKey }`.

6. **getVersionHistory** — query, input: z.object({ documentId: z.string() }).
   - Find all DocumentLink records for the same entityType+entityId as the given document.
   - Load all linked documents (including SUPERSEDED) ordered by createdAt desc.
   - Return plain(documents) as version chain.

7. **delete** — mutation, input: z.object({ documentId: z.string() }).
   - Soft-delete document (set deletedAt). Delete R2 object via `deleteObject(storageKey)`.
   - Remove associated DocumentLink records.
   - Return { success: true }.

8. **linkToEntity** — mutation, input: documentLinkSchema.
   - Create DocumentLink record connecting document to entity.
   - Return plain(link).
