```typescript
export const googleDirectoryUserSchema = z.object({

    familyName: z.string(),

  orgUnitPath: z.string().optional().nullable(),

  isAdmin: z.boolean().optional(),
});

export type GoogleDirectoryUserParsed = z.infer<typeof googleDirectoryUserSchema>;

export const googleGroupSchema = z.object({
  id: z.string(),
  email: z.string().email(),
  name: z.string(),
  description: z.string().optional().nullable(),
  directMembersCount: z.string().optional(),
});

export type GoogleGroupParsed = z.infer<typeof googleGroupSchema>;

export const groupRoleMappingSchema = z.object({
  groupEmail: z.string().email(),
  groupName: z.string(),
  role: z.enum(["admin", "manager", "viewer"]),
});

export const directoryImportInputSchema = z.object({
  users: z.array(z.object({
    email: z.string().email(),
    name: z.string(),
    googleUserId: z.string(),

  defaultRole: z.enum(["admin", "manager", "viewer"]),

});

export type DirectoryImportInput = z.infer<typeof directoryImportInputSchema>;

export const directoryImportResultSchema = z.object({
  succeeded: z.array(z.object({ email: z.string(), role: z.string() })),
  failed: z.array(z.object({ email: z.string(), error: z.string() })),
});

export type DirectoryImportResult = z.infer<typeof directoryImportResultSchema>;
```

**1. Create `packages/api/src/routers/google-workspace.ts`**

Import pattern (follow linear.ts / ksef.ts):
```typescript
import { z } from "zod";
import { TRPCError } from "@trpc/server";

import { auth } from "@contractor-ops/auth";
import {
  getAdapter,

  directoryImportInputSchema,

import { router } from "../init.js";
import { tenantProcedure } from "../middleware/tenant.js";
import { requirePermission } from "../middleware/rbac.js";
import type { GoogleWorkspaceAdapter } from "@contractor-ops/integrations/adapters/google-workspace-adapter";
```

Call `registerAllAdapters()` at module top level.

Helper function `getGoogleWorkspaceConnection(organizationId: string)`:
- Query `prisma.integrationConnection.findFirst({ where: { organizationId, provider: "GOOGLE_WORKSPACE", status: "CONNECTED" } })`
- If not found, throw `TRPCError({ code: "NOT_FOUND", message: "Google Workspace not connected" })`
- Decrypt credentials via `decryptCredentials(connection.credentialsRef, "google-workspace")`
- Get adapter via `getAdapter("google-workspace") as GoogleWorkspaceAdapter`
- Check if token expired, if so call `adapter.refreshToken(credentials)`, encrypt new creds, update connection
- Return `{ connection, credentials, adapter }`

Procedures:

**`listDirectory`** — `tenantProcedure.use(requirePermission({ member: ["read"] })).query()`:
- Get connection + adapter via helper
- Call `adapter.listAllDirectoryUsers(credentials.accessToken)`
- Get existing org members via `auth.api.getFullOrganization({ headers: ctx.headers, query: { organizationId: ctx.organizationId } })`
- Extract member emails into a Set
- Map each Google user to parsed shape: `{ id, primaryEmail, name: { givenName, familyName, fullName }, thumbnailPhotoUrl, orgUnitPath, department: user.organizations?.find(o => o.primary)?.department ?? user.organizations?.[0]?.department ?? null, isAdmin, alreadyExists: existingEmails.has(user.primaryEmail) }`
- Return `{ users, stats: { total, alreadyImported, new } }`

**`listUserGroups`** — `tenantProcedure.use(requirePermission({ member: ["read"] })).input(z.object({ userEmails: z.array(z.string().email()).min(1).max(500) })).mutation()`:
- Get connection + adapter via helper
- For each email in `userEmails`, call `adapter.listUserGroups(credentials.accessToken, email)`
- Deduplicate groups by `id` across all users
- Build map: `groupId -> { group, memberEmails: string[] }`
- Return `{ groups: Array<{ id, email, name, description, memberEmails }> }`

**`bulkImport`** — `tenantProcedure.use(requirePermission({ member: ["create"] })).input(directoryImportInputSchema).mutation()`:
- Per D-09/D-10: For each user, determine role: (1) check `userRoleOverrides[email]`, (2) check `groupRoleMappings` — find first group the user belongs to (requires groupMembership data passed from client), (3) fall back to `defaultRole`
- Actually, the client needs to send the resolved role per user. Simplify: add `resolvedUsers` to the input that has `email` + `role` already computed client-side.
- Alternative approach: the `directoryImportInputSchema` already has `users` array with per-user data, `defaultRole`, `groupRoleMappings`, and `userRoleOverrides`. The server resolves final roles.
- For role resolution, the server needs to know which groups each user belongs to. Add `userGroupMemberships: z.record(z.string().email(), z.array(z.string().email())).default({})` to the import input schema (maps user email to list of group emails they belong to). Update the validator in Task 1 accordingly.
- Loop through `input.users` sequentially, for each:
  - Determine final role: `userRoleOverrides[email] ?? firstMatchingGroupRole(userGroupMemberships[email], groupRoleMappings) ?? defaultRole`
  - Call `auth.api.createInvitation({ headers: ctx.headers, body: { email, role, organizationId: ctx.organizationId } })` (per D-11)
  - Collect successes and failures (partial failure handling per research)
- Store import metadata in `IntegrationConnection.configJson`: `{ lastImportAt: new Date(), lastImportCount: succeeded.length }`
- Return `{ succeeded: Array<{ email, role }>, failed: Array<{ email, error }> }`

**`triggerSync`** — `tenantProcedure.use(requirePermission({ member: ["read"] })).mutation()`:
- Per D-15: Admin can trigger manual sync
- Get connection via helper (just need connectionId + organizationId)
- Import QStash client: `import { Client } from "@upstash/qstash"`
- Publish to `${process.env.NEXT_PUBLIC_APP_URL}/api/google-workspace/_sync` with body `{ organizationId: ctx.organizationId, connectionId: connection.id }`
- Return `{ triggered: true }`

**`syncStatus`** — `tenantProcedure.use(requirePermission({ member: ["read"] })).query()`:
- Get connection (or return null if not connected)
- Query last sync log: `prisma.integrationSyncLog.findFirst({ where: { integrationConnectionId: connection.id }, orderBy: { startedAt: "desc" } })`
- Return `{ connected: true, connectionId, lastSyncAt, lastSyncStatus, configJson: connection.configJson }` or `{ connected: false }`
