  id, title, type, status, startDate, endDate, currency,

  id, name, type, mimeType, sizeBytes, addedAt, downloadUrl
}>

portal.listPayments() -> Array<{
  id, invoiceNumber, amountGrosze, currency, paidAt
}>

portal.getSession() -> {
  contractor: { id, displayName, email },

}
