**route.ts:**

```typescript
import { NextRequest, NextResponse } from "next/server";
// Follow the existing webhook pattern from the project

export async function POST(request: NextRequest) {
  // 1. Read raw body for signature verification
  const rawBody = await request.text();
  const signatureHeader = request.headers.get("x-hub-signature") ?? "";

  // 2. Parse payload to extract issue key for connection lookup
  let payload: unknown;
  try {
    payload = JSON.parse(rawBody);
  } catch {
    return NextResponse.json({ error: "Invalid JSON" }, { status: 400 });
  }

  // 3. Determine organization + connection from webhook
  //    Jira dynamic webhooks are registered per-connection.
  //    The webhook URL can include a query param for connection ID:
  //    /api/webhooks/jira?connectionId={id}
  const connectionId = request.nextUrl.searchParams.get("connectionId");
  if (!connectionId) {
    return NextResponse.json({ error: "Missing connectionId" }, { status: 400 });
  }

  // 4. Load connection to get organizationId and webhook secret
  const { prisma } = await import("@contractor-ops/db");
  const connection = await prisma.integrationConnection.findUnique({
    where: { id: connectionId },
    select: { id: true, organizationId: true, configJson: true },
  });

  if (!connection) {
    return NextResponse.json({ error: "Connection not found" }, { status: 404 });
  }

  // 5. Verify webhook signature if secret available
  const configJson = connection.configJson as Record<string, unknown> | null;
  const webhookSecret = configJson?.webhookSecret as string | undefined;
  if (webhookSecret && signatureHeader) {
    const { JiraAdapter } = await import("@contractor-ops/integrations/adapters/jira-adapter");
    const adapter = new JiraAdapter();
    const result = adapter.verifyWebhookSignature!(rawBody, {
      "x-hub-signature": signatureHeader,
      webhookSecret,
    });
    if (!result.valid) {
      return NextResponse.json({ error: "Invalid signature" }, { status: 401 });
    }
  }

  // 6. Record WebhookDelivery
  await prisma.webhookDelivery.create({
    data: {
      organizationId: connection.organizationId,
      integrationConnectionId: connection.id,
      provider: "JIRA",
      eventType: (payload as Record<string, unknown>).webhookEvent as string ?? "unknown",
      payloadJson: payload as object,
      signatureValid: true,
    },
  });

  // 7. Process webhook (direct — no QStash for simplicity, or via QStash if the project pattern requires it)
  try {
    const { processJiraWebhook } = await import("@contractor-ops/api/services/jira-webhook-handler");
    await processJiraWebhook(
      prisma,
      connection.organizationId,
      connection.id,
      payload,
    );

    await prisma.webhookDelivery.updateMany({
      where: {
        integrationConnectionId: connection.id,
        deliveryStatus: "RECEIVED",
      },
      data: { deliveryStatus: "PROCESSED", processedAt: new Date() },
    });
  } catch (error) {
    console.error("Jira webhook processing failed:", error);
    // Update latest delivery as failed
    return NextResponse.json({ error: "Processing failed" }, { status: 500 });
  }

  return NextResponse.json({ ok: true });
}
```
