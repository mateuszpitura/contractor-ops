// Jira webhook payload (inbound)
export const jiraWebhookPayloadSchema = z.object({
  webhookEvent: z.literal("jira:issue_updated"),
  timestamp: z.number(),
  issue: z.object({
    id: z.string(),
    key: z.string(),
    fields: z.object({
      summary: z.string(),
      status: z.object({
        name: z.string(),
        statusCategory: z.object({
          key: z.enum(["new", "indeterminate", "done"]),
          name: z.string(),
        }),
      }),
      project: z.object({
        id: z.string(),
        key: z.string(),
        name: z.string(),
      }),
    }),
  }),
  changelog: z.object({
    items: z.array(z.object({
      field: z.string(),
      fieldtype: z.string(),
      from: z.string().nullable(),
      fromString: z.string().nullable(),
      to: z.string().nullable(),
      toString: z.string().nullable(),
    })),
  }),
});

export type JiraWebhookPayload = z.infer<typeof jiraWebhookPayloadSchema>;

// Task template Jira config (stored in configJson)
export const jiraTaskConfigSchema = z.object({
  jiraEnabled: z.boolean().default(false),
  jiraProjectId: z.string().optional(),
  jiraProjectKey: z.string().optional(),
  jiraProjectName: z.string().optional(),
  jiraIssueTypeId: z.string().optional(),
  jiraIssueTypeName: z.string().optional(),
});

export type JiraTaskConfig = z.infer<typeof jiraTaskConfigSchema>;

  jiraTargetStatusName: z.string(),

**jira-issue-sync.ts:**

Exports:
- `createJiraIssue(prisma, organizationId, connectionId, taskRunId)`: Creates a Jira issue from WorkflowTaskRun data
  1. Load WorkflowTaskRun with workflowRun (for contractorId) and workflowTaskTemplateId
  2. Parse configJson with jiraTaskConfigSchema to get jiraProjectId, jiraIssueTypeId
  3. Decrypt credentials, build baseUrl (same pattern as worklog sync)
  4. POST /rest/api/3/issue with fields: project.id, issuetype.id, summary="{taskRun.title}", description in ADF format: `{ type: "doc", version: 1, content: [{ type: "paragraph", content: [{ type: "text", text: taskRun.description ?? taskRun.title }] }] }`
  5. Update WorkflowTaskRun: externalRefType="JIRA_ISSUE", externalRefId=created.key
  6. Create ExternalLink: entityType=WORKFLOW_TASK_RUN, entityId=taskRunId, externalType="JIRA_ISSUE", externalId=created.key, externalUrl=`{siteUrl}/browse/{created.key}`, metadataJson with JiraIssueMetadata (key, summary, status from created issue, statusCategory, url, lastSyncOrigin="APP", lastSyncAt=now)
  7. Create IntegrationSyncLog: direction=OUTBOUND, syncType="issue-create", status=SUCCESS
  8. Return { issueKey: created.key, issueId: created.id }

- `transitionJiraIssue(prisma, organizationId, connectionId, taskRunId, newWorkflowStatus)`: Outbound status sync
  1. Load ExternalLink for taskRunId where externalType="JIRA_ISSUE"
  2. Get project ID from ExternalLink.metadataJson or WorkflowTaskTemplate.configJson
  3. Call lookupJiraTransitionId for the newWorkflowStatus
  4. If no mapping, log to IntegrationSyncLog with syncType="issue-transition-unmapped" and return
  5. Set metadataJson.lastSyncOrigin="APP", metadataJson.lastSyncAt=now on ExternalLink
  6. POST /rest/api/3/issue/{issueKey}/transitions with { transition: { id: transitionId } }
  7. Update ExternalLink.metadataJson with new status info
  8. Create IntegrationSyncLog: direction=OUTBOUND, syncType="issue-transition"

- `detectScopeExpansionNeeded(storedScope: string): boolean`: Returns true if storedScope is missing write:jira-work or manage:jira-webhook

**jira-webhook-handler.ts:**

Exports:
- `processJiraWebhook(prisma, organizationId, connectionId, payload)`:
  1. Parse payload with jiraWebhookPayloadSchema
  2. Extract status change: `changelog.items.find(item => item.field === "status")`
  3. If no status change, return early (not a status update)
  4. Find ExternalLink where externalType="JIRA_ISSUE", externalId=payload.issue.key
  5. If no ExternalLink, log and return (issue not linked to any task)
  6. **Loop prevention (per D-08):** Check metadataJson.lastSyncOrigin. If "APP" and lastSyncAt within 30 seconds, skip (this is our own change bouncing back)
  7. **Deduplication:** Check IntegrationSyncLog for same issue key + same target status within last 5 seconds
  8. Get project ID from payload.issue.fields.project.id
  9. Call lookupWorkflowStatus(prisma, connectionId, projectId, statusChange.toString)
  10. If no mapping, log IntegrationSyncLog with syncType="webhook-status-unmapped" and return
  11. Update WorkflowTaskRun.status to the mapped value
  12. Update ExternalLink.metadataJson: new status, statusCategory, lastSyncOrigin="JIRA", lastSyncAt=now
  13. Create IntegrationSyncLog: direction=INBOUND, syncType="issue-status-change"

- `registerJiraWebhooks(prisma, connectionId, projectKeys: string[])`:
  1. Decrypt credentials, build baseUrl
  2. POST /rest/api/3/webhook with combined JQL `project IN (${projectKeys.join(", ")})`, events: ["jira:issue_updated"]
  3. Store webhook IDs in IntegrationConnection.configJson.webhookIds array
  4. Store webhookRegisteredAt timestamp in configJson

- `deregisterJiraWebhooks(prisma, connectionId)`:
  1. Read webhookIds from configJson
  2. DELETE /rest/api/3/webhook for each ID
  3. Clear webhookIds and webhookRegisteredAt from configJson

- `refreshJiraWebhooks(prisma, connectionId)`:
  1. Read webhookIds from configJson
  2. PUT /rest/api/3/webhook/refresh with webhookIds array
  3. Update webhookRegisteredAt to now

All services use the same error handling pattern: try/catch with IntegrationSyncLog failure recording and TRPCError wrapping.
