       - Implementation:
         ```ts
         import { addDays, startOfDay } from './date-utils.js'; // or existing helpers
         import { prismaRaw } from '@contractor-ops/db';
         import { resolveRbacRecipients } from '@contractor-ops/api/services/rbac-recipients';
         import { dispatch } from '@contractor-ops/api/services/notification-service';

         async function detectDrvClearanceExpiries(): Promise<number> {
           const now = new Date();
           const today = startOfDay(now);
           const bands = [
             { days: 90, type: 'classification.drv_expiry_90d' as const },
             { days: 30, type: 'classification.drv_expiry_30d' as const },
             { days:  7, type: 'classification.drv_expiry_7d'  as const },
           ];
           let notified = 0;
           for (const band of bands) {
             const target = addDays(today, band.days);
             // PHASE-60-CROSS-ORG-AGGREGATE: cron runs without tenant frame
             const clearances = await prismaRaw.statusfeststellungsverfahren.findMany({
               where: {
                 validTo: { gte: target, lt: addDays(target, 1) },
                 outcome: { in: ['SELBSTANDIG', 'ABHANGIG'] },
               },
               include: { contractorAssignment: { select: { id: true } } },
             });
             for (const c of clearances) {
               const prior = await prismaRaw.notification.findFirst({
                 where: { type: band.type, entityType: 'CONTRACTOR', entityId: c.id },
                 select: { id: true },
               });
               if (prior) continue;
               const recipientUserIds = await resolveRbacRecipients(c.organizationId, 'contractor:read');
               if (recipientUserIds.length === 0) continue;
               await dispatch({
                 organizationId: c.organizationId,
                 type: band.type,
                 recipientUserIds,
                 title: `DRV clearance expires in ${band.days} days`,
                 body: `Reference ${c.drvReference}, valid until ${c.validTo?.toISOString().slice(0,10)}. Begin the renewal filing — DRV processing typically takes 3-6 months.`,
                 entityType: 'CONTRACTOR',
                 entityId: c.id,
               });
               notified++;
             }
           }
           return notified;
         }
         ```
