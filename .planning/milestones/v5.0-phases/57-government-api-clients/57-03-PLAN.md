import type { Prisma, PrismaClient, TaxIdType, ValidationStatus } from '@contractor-ops/db';

export interface TaxIdValidationInput {

  requestedAt: Date;
  taxIdValidationId: string;
}

    // Mask tax IDs for safe logging. Mirrors sv-nummer-mask pattern from Phase 56.

    export function maskTaxId(value: string | null | undefined): string {
      if (!value) return '[empty]';
      const trimmed = value.trim();
      if (trimmed.length <= 4) return '*'.repeat(trimmed.length);
      const prefix = trimmed.slice(0, 2);    // country code
      const suffix = trimmed.slice(-4);      // last 4
      return `${prefix}${'*'.repeat(Math.max(1, trimmed.length - 6))}${suffix}`;
    }
    ```

    // packages/api/src/services/kleinunternehmer.service.ts
    //
    // Phase 57 D-11 — Kleinunternehmerregelung (§19 UStG).
    // When org.countryCode='DE' AND org.isKleinunternehmer=true:
    //   - All non-RC invoice lines use rate code 'KU' (0% exempt)
    //   - Invoice footer renders TAX_KLEINUNTERNEHMER_NOTICE (handled in Plan 57-04 UI)
    //   - The totals breakdown suppresses the VAT row entirely (RESEARCH Pitfall 7)

    export interface KleinunternehmerOverrideResult {
      vatRate: string;
      forced: boolean;
      reason?: string;
    }

    export function applyKleinunternehmerOverride(
      line: { vatRate: string | null; description?: string },
      org: { countryCode: string | null; isKleinunternehmer: boolean },
    ): KleinunternehmerOverrideResult {
      const original = line.vatRate ?? '';
      if (org.countryCode !== 'DE' || !org.isKleinunternehmer) {
        return { vatRate: original, forced: false };
      }
      // RC takes precedence — reverse charge is a legal attribution, KU is an exemption
      if (original === 'RC') {
        return {
          vatRate: 'RC',
          forced: false,
          reason: 'Reverse charge takes precedence over Kleinunternehmer',
        };
      }
      return {
        vatRate: 'KU',
        forced: true,
        reason: '§19 UStG Kleinunternehmerregelung',
      };
    }

    export function shouldSuppressVatBreakdown(
      org: { countryCode: string | null; isKleinunternehmer: boolean },
    ): boolean {
      return org.countryCode === 'DE' && org.isKleinunternehmer === true;
    }
    ```
