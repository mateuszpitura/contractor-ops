    ```typescript
    // packages/api/src/gov-api-clients.ts
    // Singleton factory: reads env at first call, caches clients for process lifetime.
    import { HmrcVatClient, ViesClient, type GovApiEnvironment } from '@contractor-ops/gov-api';
    import { getSecretStore } from '@contractor-ops/secrets';
    import { z } from 'zod';

    // envSchema uses superRefine to enforce HMRC_PLATFORM_VRN in production only.
    // Sandbox permits empty VRN (HMRC sandbox accepts single-arg /lookup:vrn calls — verified-lookup just degrades to unverified).
    // Production path REQUIRES a real platform VRN so every /lookup/:target/:requester URL is well-formed + attributable.
    const envSchema = z.object({
      HMRC_ENV: z.enum(['sandbox','production']).default('sandbox'),
      VIES_ENV: z.enum(['sandbox','production']).default('production'),
      HMRC_PLATFORM_VRN: z.string().optional(),
    }).superRefine((val, ctx) => {
      if (val.HMRC_ENV === 'production' && (!val.HMRC_PLATFORM_VRN || val.HMRC_PLATFORM_VRN.length === 0)) {
        ctx.addIssue({
          code: z.ZodIssueCode.custom,
          path: ['HMRC_PLATFORM_VRN'],
          message: 'HMRC_PLATFORM_VRN required in production (used as requesterVrn for verified /lookup/:target/:requester calls)',
        });
      }
    });

    let hmrc: HmrcVatClient | null = null;
    let vies: ViesClient | null = null;

    export function getHmrcVatClient(): HmrcVatClient {
      if (hmrc) return hmrc;
      const env = envSchema.parse(process.env);
      // Defense in depth: even if superRefine is ever bypassed (e.g. partial env mutation at runtime), throw here.
      if (env.HMRC_ENV === 'production' && (!env.HMRC_PLATFORM_VRN || env.HMRC_PLATFORM_VRN.length === 0)) {
        throw new Error('HMRC_PLATFORM_VRN required in production');
      }
      hmrc = new HmrcVatClient({
        config: {
          baseUrls: {
            sandbox: 'https://test-api.service.hmrc.gov.uk',
            production: 'https://api.service.hmrc.gov.uk',
          },
          timeoutMs: 30000,
          retry: { maxRetries: 3, baseDelayMs: 1000, maxDelayMs: 30000, retryableStatuses: [500,502,503,429] },
        },
        environment: env.HMRC_ENV as GovApiEnvironment,
        secretStore: getSecretStore(),
        platformVrn: env.HMRC_PLATFORM_VRN ?? '', // sandbox-only empty path — production path throws above
        pkgVersion: process.env.npm_package_version ?? '0.0.0',
      });
      return hmrc;
    }

    export function getViesClient(): ViesClient {
      if (vies) return vies;
      const env = envSchema.parse(process.env);
      vies = new ViesClient({
        config: {
          baseUrls: {
            sandbox: 'https://ec.europa.eu/taxation_customs/vies',
            production: 'https://ec.europa.eu/taxation_customs/vies',
          },
          timeoutMs: 30000,
          retry: { maxRetries: 2, baseDelayMs: 500, maxDelayMs: 5000, retryableStatuses: [500,502,503] },
        },
        environment: env.VIES_ENV as GovApiEnvironment,
        requesterMemberStateCode: env.HMRC_PLATFORM_VRN ? 'DE' : undefined,
        requesterNumber: env.HMRC_PLATFORM_VRN,  // reuse platform identifier; optional
      });
      return vies;
    }
    ```

         ```typescript
                 { organizationId: ctx.organizationId, contractorId: updated.id, taxIdType, taxIdValue: updated.vatId, actor: { userId: ctx.userId, intent: 'profile-save' } },
                 { prisma: ctx.db, hmrcClient: getHmrcVatClient(), viesClient: getViesClient() },
               );
             }
             // If countryCode is not GB/DE, intentionally no validation path exists — leave summary fields as-is
           }
         }
         return updated;
         ```

         Rationale: awaited inline (not fire-and-forget) because (a) orchestrator soft-fails on API outage — never blocks the save; (b) user expects fresh `latestVatValidationStatus` on next profile render — fire-and-forget would leave the pill stale until a background refresh; (c) keeps the whole profile save + validation inside a single tRPC roundtrip for predictable UX and easier testability.

         If the profile-save path submits multiple distinct fields (e.g. `vatRegistrationNumber` for UK + `ustIdNr` for DE as separate columns), compare BOTH columns for change, and dispatch on whichever changed that matches the contractor's `countryCode`.
