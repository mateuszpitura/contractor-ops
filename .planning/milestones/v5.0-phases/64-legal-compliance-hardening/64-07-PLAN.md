The component needs three states:
1. **No approval yet**: Show `SDS_APPROVAL_STATEMENT_EN` checkbox + `clientName` input + "Confirm Approval" button
2. **Approval in progress**: Disabled state while `approveSds` mutation is in-flight
3. **Approved**: Show "Generate SDS" button (original behavior, now enabled)

```typescript
// Key additions to generate-sds-button.tsx:

import {
  SDS_APPROVAL_STATEMENT_EN,
} from '@contractor-ops/validators';
import { api } from '@/trpc/react'; // or equivalent tRPC client import
import { useState } from 'react';
import { Checkbox } from '@/components/ui/checkbox';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Button } from '@/components/ui/button';

// Inside the component, add:
const [approved, setApproved] = useState(false);
const [clientName, setClientName] = useState('');
const [checked, setChecked] = useState(false);
const { t } = useTranslations('Legal.SdsApproval');

const approveSds = api.classification.approveSds.useMutation({
  onSuccess: () => setApproved(true),
});

// Render the approval gate before the Generate SDS button:
if (!approved) {
  return (
    <div className="space-y-4 rounded-lg border border-amber-200 bg-amber-50 p-4">
      <p className="text-sm font-medium">{t('gateTitle')}</p>
      
      {/* Client name input */}
      <div className="space-y-1.5">
        <Label htmlFor="clientName">{t('clientNameLabel')}</Label>
        <Input
          id="clientName"
          value={clientName}
          onChange={(e) => setClientName(e.target.value)}
          placeholder={t('clientNamePlaceholder')}
          maxLength={500}
        />
      </div>
      
      {/* Approval statement checkbox */}
      <div className="flex items-start gap-3">
        <Checkbox
          id="sds-approval"
          checked={checked}
          onCheckedChange={(c) => setChecked(c === true)}
        />
        <Label htmlFor="sds-approval" className="text-xs leading-relaxed text-muted-foreground">
          {SDS_APPROVAL_STATEMENT_EN}
        </Label>
      </div>
      
      <Button
        onClick={() => approveSds.mutate({ assessmentId, clientName })}
        disabled={!checked || !clientName.trim() || approveSds.isPending}
        size="sm"
      >
        {approveSds.isPending ? t('confirmingApproval') : t('confirmApproval')}
      </Button>
    </div>
  );
}
// Then show the original Generate SDS button
```

Note: Also check if the assessment already has an `SdsApproval` (query `trpc.classification.getSdsApproval` or check assessment data). If approval already exists, skip the gate.

```typescript
const handleFileSelect = async (e: React.ChangeEvent<HTMLInputElement>) => {
  const file = e.target.files?.[0];
  if (!file) return;
  if (file.size > 10 * 1024 * 1024) {
    // Show error: file too large
    return;
  }
  const reader = new FileReader();
  reader.onload = (ev) => {
    const base64 = (ev.target?.result as string).split(',')[1];
    uploadMutation.mutate({
      classificationAssessmentId: assessmentId,
      fileBase64: base64,
      fileName: file.name,
      mimeType: file.type as 'application/pdf' | 'image/jpeg' | 'image/png',
      fileSizeBytes: file.size,
    });
  };
  reader.readAsDataURL(file);
};
```
