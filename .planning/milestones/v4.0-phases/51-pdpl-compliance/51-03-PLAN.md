```typescript
// Static hosting region config — becomes dynamic in Phase 52
const DATA_HOSTING_REGION = process.env.DATA_HOSTING_REGION ?? "EU";

export function detectCrossBorderTransfer(orgCountryCode: string): {
  isCrossBorder: boolean;
  orgRegion: string;
  hostingRegion: string;
} {
  // Map country codes to regions
  const countryToRegion: Record<string, string> = {
    AE: "GCC", SA: "GCC", BH: "GCC", KW: "GCC", OM: "GCC", QA: "GCC",
    PL: "EU", DE: "EU", FR: "EU", // ... EU countries
  };
  const orgRegion = countryToRegion[orgCountryCode] ?? "OTHER";
  return {
    isCrossBorder: orgRegion !== DATA_HOSTING_REGION,
    orgRegion,
    hostingRegion: DATA_HOSTING_REGION,
  };
}
```

**2. Add download endpoints to consent router (`packages/api/src/routers/consent.ts`):**

```typescript
// Download DPA for current org
downloadDPA: tenantProcedure
  .use(requirePermission({ settings: ["read"] }))
  .mutation(async ({ ctx }) => {
    const result = await generateDPA(ctx.organizationId);
    if (!result) throw new TRPCError({ code: "NOT_FOUND", message: "DPA not available for this jurisdiction" });
    return {
      buffer: result.buffer.toString("base64"),
      filename: result.filename,
      contentType: "application/pdf",
    };
  }),

// Download SCC for current org (null if no cross-border transfer)
downloadSCC: tenantProcedure
  .use(requirePermission({ settings: ["read"] }))
  .mutation(async ({ ctx }) => {
    const result = await generateSCC(ctx.organizationId);
    if (!result) throw new TRPCError({ code: "NOT_FOUND", message: "No cross-border transfer detected — SCC not required" });
    return {
      buffer: result.buffer.toString("base64"),
      filename: result.filename,
      contentType: "application/pdf",
    };
  }),
```
