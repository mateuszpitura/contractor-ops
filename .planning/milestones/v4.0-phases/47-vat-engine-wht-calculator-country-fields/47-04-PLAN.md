```typescript
import { z } from "zod";

// ---------------------------------------------------------------------------
// UAE country fields (PROF-01)
// ---------------------------------------------------------------------------

export const uaeCountryFieldsSchema = z.object({
  freelancePermitNumber: z.string().max(50).optional(),
  tradeLicenseNumber: z.string().max(50).optional(),
  freeZone: z.boolean().optional(),
  tradeLicenseExpiry: z.string().date().optional(),
});

export type UaeCountryFields = z.infer<typeof uaeCountryFieldsSchema>;

// ---------------------------------------------------------------------------
// Saudi Arabia country fields (PROF-02)
// ---------------------------------------------------------------------------

export const saudiCountryFieldsSchema = z.object({
  freelanceSaLicense: z.string().max(50).optional(),
  commercialRegistration: z.string().max(20).optional(),
  commercialRegistrationExpiry: z.string().date().optional(),
});

export type SaudiCountryFields = z.infer<typeof saudiCountryFieldsSchema>;

// ---------------------------------------------------------------------------
// Country fields discriminated union (PROF-03)
// ---------------------------------------------------------------------------

/** Validate countryFields based on org country code. Returns the validated schema or null for unsupported countries. */
export function getCountryFieldsSchema(countryCode: string): z.ZodType | null {
  switch (countryCode) {
    case "AE":
      return uaeCountryFieldsSchema;
    case "SA":
      return saudiCountryFieldsSchema;
    default:
      return null;
  }
}

/** List of country codes that have country-specific fields */
export const COUNTRIES_WITH_FIELDS = ["AE", "SA"] as const;

// ---------------------------------------------------------------------------
// Per-country TIN format validation (TAX-04 / PROF-04)
// ---------------------------------------------------------------------------

/**
 * UAE Tax Registration Number (TRN): exactly 15 digits
 * Format: 100XXXXXXXXXXXX (starts with 100)
 */
export function validateUaeTin(tin: string): boolean {
  return /^\d{15}$/.test(tin);
}

/**
 * Saudi Arabia TIN: 13 digits, starts with "3", position 11 (0-indexed 10) is "3"
 * Format: 3XXXXXXXXX3XXX
 */
export function validateSaudiTin(tin: string): boolean {
  if (!/^\d{13}$/.test(tin)) return false;
  if (tin[0] !== "3") return false;
  if (tin[10] !== "3") return false;
  return true;
}

/**
 * Polish NIP: 10 digits with checksum validation
 * Weights: [6, 5, 7, 2, 3, 4, 5, 6, 7]
 */
export function validatePolishNip(nip: string): boolean {
  const clean = nip.replace(/[-\s]/g, "");
  if (!/^\d{10}$/.test(clean)) return false;
  const weights = [6, 5, 7, 2, 3, 4, 5, 6, 7];
  const sum = weights.reduce((acc, w, i) => acc + w * parseInt(clean[i]!, 10), 0);
  return sum % 11 === parseInt(clean[9]!, 10);
}

/**
 * Validate TIN format for a given country.
 * Returns { valid: boolean, message?: string }
 */
export function validateTin(countryCode: string, tin: string): { valid: boolean; message?: string } {
  switch (countryCode) {
    case "AE":
      return validateUaeTin(tin)
        ? { valid: true }
        : { valid: false, message: "UAE TRN must be exactly 15 digits" };
    case "SA":
      return validateSaudiTin(tin)
        ? { valid: true }
        : { valid: false, message: "Saudi TIN must be 13 digits, starting with 3, with 3 at position 11" };
    case "PL":
      return validatePolishNip(tin)
        ? { valid: true }
        : { valid: false, message: "Polish NIP must be 10 digits with valid checksum" };
    default:
      // Unknown country — accept any non-empty string
      return tin.length > 0
        ? { valid: true }
        : { valid: false, message: "Tax ID is required" };
  }
}
```

```typescript
/** Get country-specific field schema for the org's country */
getCountryFieldsConfig: tenantProcedure
  .query(async ({ ctx }) => {
    const org = await ctx.prisma.organization.findUniqueOrThrow({
      where: { id: ctx.organizationId },
      select: { countryCode: true },
    });
    if (!org.countryCode || !COUNTRIES_WITH_FIELDS.includes(org.countryCode as any)) {
      return { hasCountryFields: false, countryCode: org.countryCode };
    }
    return {
      hasCountryFields: true,
      countryCode: org.countryCode,
      fields: org.countryCode === "AE"
        ? ["freelancePermitNumber", "tradeLicenseNumber", "freeZone", "tradeLicenseExpiry"]
        : ["freelanceSaLicense", "commercialRegistration", "commercialRegistrationExpiry"],
    };
  }),

/** Get country fields for a specific contractor */
getCountryFields: tenantProcedure
  .input(z.object({ contractorId: z.string() }))
  .query(async ({ ctx, input }) => {
    const contractor = await ctx.prisma.contractor.findUnique({
      where: { id: input.contractorId, organizationId: ctx.organizationId },
      select: { countryFields: true },
    });
    if (!contractor) throw new TRPCError({ code: "NOT_FOUND" });
    return contractor.countryFields ?? {};
  }),

/** Update country fields for a contractor (validated per org country) */
updateCountryFields: tenantProcedure
  .input(updateCountryFieldsSchema)
  .mutation(async ({ ctx, input }) => {
    // Get org country to validate field schema
    const org = await ctx.prisma.organization.findUniqueOrThrow({
      where: { id: ctx.organizationId },
      select: { countryCode: true },
    });

    if (!org.countryCode) {
      throw new TRPCError({ code: "BAD_REQUEST", message: "Organization has no country set" });
    }

    // Validate fields against country-specific schema
    const schema = getCountryFieldsSchema(org.countryCode);
    if (!schema) {
      throw new TRPCError({
        code: "BAD_REQUEST",
        message: `No country-specific fields defined for ${org.countryCode}`,
      });
    }

    const parsed = schema.safeParse(input.fields);
    if (!parsed.success) {
      throw new TRPCError({
        code: "BAD_REQUEST",
        message: `Invalid country fields: ${parsed.error.message}`,
      });
    }

    return ctx.prisma.contractor.update({
      where: { id: input.contractorId, organizationId: ctx.organizationId },
      data: { countryFields: parsed.data },
    });
  }),

/** Validate a TIN for a given country */
validateTin: tenantProcedure
  .input(z.object({
    countryCode: z.string().length(2),
    tin: z.string().min(1),
  }))
  .query(({ input }) => {
    return validateTin(input.countryCode, input.tin);
  }),
```

```tsx
"use client";

import { useState } from "react";
import { api } from "@/lib/trpc";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Button } from "@/components/ui/button";
import { Switch } from "@/components/ui/switch";
import { DatePicker } from "@/components/ui/date-picker";
import { Badge } from "@/components/ui/badge";
import { AlertCircle, Check, Loader2 } from "lucide-react";
import { toast } from "sonner";

interface CountryComplianceSectionProps {
  contractorId: string;
}

export function CountryComplianceSection({ contractorId }: CountryComplianceSectionProps) {
  const configQuery = api.contractor.getCountryFieldsConfig.useQuery();
  const fieldsQuery = api.contractor.getCountryFields.useQuery({ contractorId });
  const updateMutation = api.contractor.updateCountryFields.useMutation({
    onSuccess: () => {
      toast.success("Compliance fields saved");
      fieldsQuery.refetch();
    },
    onError: (err) => {
      toast.error(err.message || "Certificate generation failed. Check that all payment details are complete and try again.");
    },
  });

  const [formData, setFormData] = useState<Record<string, unknown>>({});

  // Don't render if org has no country fields
  if (configQuery.isLoading || fieldsQuery.isLoading) {
    return (
      <Card>
        <CardContent className="flex items-center justify-center py-lg">
          <Loader2 className="h-4 w-4 animate-spin text-muted-foreground" />
        </CardContent>
      </Card>
    );
  }

  if (!configQuery.data?.hasCountryFields) {
    return null; // No country-specific fields for this org
  }

  const { countryCode } = configQuery.data;
  const existingFields = (fieldsQuery.data ?? {}) as Record<string, unknown>;
  const merged = { ...existingFields, ...formData };

  const countryLabel = countryCode === "AE" ? "UAE" : countryCode === "SA" ? "Saudi Arabia" : countryCode;

  function handleSave() {
    if (!countryCode) return;
    updateMutation.mutate({
      contractorId,
      countryCode,
      fields: merged,
    });
  }

  const missingCount = configQuery.data.fields
    ? configQuery.data.fields.filter((f: string) => !merged[f]).length
    : 0;

  return (
    <Card>
      <CardHeader className="flex flex-row items-center justify-between">
        <CardTitle className="text-base font-semibold">
          Country Compliance — {countryLabel}
        </CardTitle>
        {missingCount > 0 && (
          <Badge variant="outline" className="border-warning/20 bg-warning/5 text-warning">
            <AlertCircle className="mr-1 h-3 w-3" />
            {missingCount} incomplete
          </Badge>
        )}
      </CardHeader>
      <CardContent className="space-y-md">
        {countryCode === "AE" && (
          <UaeFields values={merged} onChange={(key, val) => setFormData((prev) => ({ ...prev, [key]: val }))} />
        )}
        {countryCode === "SA" && (
          <SaudiFields values={merged} onChange={(key, val) => setFormData((prev) => ({ ...prev, [key]: val }))} />
        )}
        <Button
          onClick={handleSave}
          disabled={updateMutation.isPending}
          className="mt-md"
        >
          {updateMutation.isPending ? (
            <Loader2 className="mr-2 h-4 w-4 animate-spin" />
          ) : (
            <Check className="mr-2 h-4 w-4" />
          )}
          Save Compliance Fields
        </Button>
      </CardContent>
    </Card>
  );
}

function UaeFields({
  values,
  onChange,
}: {
  values: Record<string, unknown>;
  onChange: (key: string, val: unknown) => void;
}) {
  return (
    <>
      <div className="space-y-sm">
        <Label htmlFor="freelancePermitNumber" className="text-sm font-medium">
          Freelance Permit Number
        </Label>
        <Input
          id="freelancePermitNumber"
          value={(values.freelancePermitNumber as string) ?? ""}
          onChange={(e) => onChange("freelancePermitNumber", e.target.value || undefined)}
          placeholder="Enter permit number"
        />
      </div>
      <div className="space-y-sm">
        <Label htmlFor="tradeLicenseNumber" className="text-sm font-medium">
          Trade License Number
        </Label>
        <Input
          id="tradeLicenseNumber"
          value={(values.tradeLicenseNumber as string) ?? ""}
          onChange={(e) => onChange("tradeLicenseNumber", e.target.value || undefined)}
          placeholder="Enter license number"
        />
      </div>
      <div className="flex items-center gap-sm">
        <Switch
          id="freeZone"
          checked={(values.freeZone as boolean) ?? false}
          onCheckedChange={(checked) => onChange("freeZone", checked)}
        />
        <Label htmlFor="freeZone" className="text-sm font-medium">
          Free Zone
        </Label>
      </div>
      <div className="space-y-sm">
        <Label htmlFor="tradeLicenseExpiry" className="text-sm font-medium">
          Trade License Expiry
        </Label>
        <Input
          id="tradeLicenseExpiry"
          type="date"
          value={(values.tradeLicenseExpiry as string) ?? ""}
          onChange={(e) => onChange("tradeLicenseExpiry", e.target.value || undefined)}
        />
      </div>
    </>
  );
}

function SaudiFields({
  values,
  onChange,
}: {
  values: Record<string, unknown>;
  onChange: (key: string, val: unknown) => void;
}) {
  return (
    <>
      <div className="space-y-sm">
        <Label htmlFor="freelanceSaLicense" className="text-sm font-medium">
          Freelance.sa License
        </Label>
        <Input
          id="freelanceSaLicense"
          value={(values.freelanceSaLicense as string) ?? ""}
          onChange={(e) => onChange("freelanceSaLicense", e.target.value || undefined)}
          placeholder="Enter Freelance.sa license number"
        />
      </div>
      <div className="space-y-sm">
        <Label htmlFor="commercialRegistration" className="text-sm font-medium">
          Commercial Registration
        </Label>
        <Input
          id="commercialRegistration"
          value={(values.commercialRegistration as string) ?? ""}
          onChange={(e) => onChange("commercialRegistration", e.target.value || undefined)}
          placeholder="Enter CR number"
        />
      </div>
      <div className="space-y-sm">
        <Label htmlFor="commercialRegistrationExpiry" className="text-sm font-medium">
          CR Expiry Date
        </Label>
        <Input
          id="commercialRegistrationExpiry"
          type="date"
          value={(values.commercialRegistrationExpiry as string) ?? ""}
          onChange={(e) => onChange("commercialRegistrationExpiry", e.target.value || undefined)}
        />
      </div>
    </>
  );
}
```
