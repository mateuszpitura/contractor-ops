4. **Add rate limiting helper:**
```typescript
private async checkRateLimit(identifier: string): Promise<void> {
  if (!this.rateLimiter) return;
  const result = await this.rateLimiter.checkLimit(identifier);
  if (!result.allowed) {
    throw new Error(
      `Storecove API rate limit exceeded for ${identifier}. Remaining: ${result.remaining}, resets in ${result.resetMs}ms`,
    );
  }
}
```
