**1. Create `apps/web/src/components/settings/dpd-provider-section.tsx`:**

Per D-06 and UI-SPEC section 1, create a DPD provider card with credential form dialog. Follow TeamsProviderSection pattern:

```typescript
"use client";

import { useState } from "react";
import { useQuery } from "@tanstack/react-query";
import { useTranslations } from "next-intl";
import { Truck } from "lucide-react";

import { trpc } from "@/trpc/init";
import { Button } from "@/components/ui/button";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from "@/components/ui/dialog";
import { ProviderConnectionCard } from "./provider-connection-card";
import { CarrierCredentialForm } from "./carrier-credential-form";

// tRPC proxy for stale dist types
// eslint-disable-next-line @typescript-eslint/no-explicit-any
const equipmentProxy = (trpc as any).equipment as {
  getCourierConfigs: { queryOptions: () => any };
};

export function DpdProviderSection() {
  const t = useTranslations("Equipment.carrier");
  const [configOpen, setConfigOpen] = useState(false);

  const configsQuery = useQuery(equipmentProxy.getCourierConfigs.queryOptions());
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const configs = (configsQuery.data ?? []) as Array<{ carrier: string }>;
  const isConfigured = configs.some((c) => c.carrier === "dpd");

  return (
    <div className="space-y-4">
      <ProviderConnectionCard
        provider="dpd"
        displayName="DPD"
        icon={<Truck className="size-8 text-red-600" />}
        description={t("dpdDescription")}
      />
      <Button variant="outline" size="sm" onClick={() => setConfigOpen(true)}>
        {t("configureDpd")}
      </Button>
      <Dialog open={configOpen} onOpenChange={setConfigOpen}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>DPD Configuration</DialogTitle>
          </DialogHeader>
          <CarrierCredentialForm carrier="dpd" carrierLabel="DPD" />
        </DialogContent>
      </Dialog>
    </div>
  );
}
```

NOTE: The ProviderConnectionCard internally queries `integration.getHealth` for the provider slug. Since DPD/UPS are NOT integration providers (they are courier configs stored in org settings), the card will show as "Not connected" by default. The `isConfigured` state from getCourierConfigs is for future enhancement. For now the card + configure button pattern is sufficient per D-06.

Alternatively, if ProviderConnectionCard shows a misleading "Connect" OAuth button for DPD, the executor should render a simple Card instead:
```typescript
<Card>
  <CardHeader className="flex flex-row items-center gap-3">
    <Truck className="size-8 text-red-600" />
    <div>
      <CardTitle className="text-base">DPD</CardTitle>
      <p className="text-sm text-muted-foreground">{t("dpdDescription")}</p>
    </div>
    <Badge variant={isConfigured ? "default" : "secondary"} className="ml-auto">
      {isConfigured ? "Connected" : "Not configured"}
    </Badge>
  </CardHeader>
</Card>
```

Use your judgment on which approach works better after reading ProviderConnectionCard behavior. The key requirement is: card with icon + status + "Configure" button that opens CarrierCredentialForm in a Dialog.
