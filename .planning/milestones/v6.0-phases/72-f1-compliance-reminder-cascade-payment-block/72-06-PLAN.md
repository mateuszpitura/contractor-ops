```ts
// Phase 72 D-17 — Snapshot builder for PaymentRunComplianceCheck.
// Captures FULL ContractorComplianceItem rows (frozen copy, structuredClone)
// regardless of status — PASS verdict still snapshots the full BLOCKING set
// so auditors can see what was CHECKED, not just what FAILED.

import type { Prisma } from '@contractor-ops/db';
import { getCurrentPolicyVersion } from '@contractor-ops/compliance-policy/version';

export interface ItemSnapshot {
  itemId: string;
  policyRuleId: string | null;
  severity: 'BLOCKING' | 'WARNING' | 'INFO';
  status: 'MISSING' | 'PENDING' | 'SATISFIED' | 'EXPIRED' | 'WAIVED';
  expiresAt: string | null;
  expiryJurisdictionTz: string | null;
  satisfiedByDocumentId: string | null;
  waivedReason: string | null;
  waivedAt: string | null;
  satisfiedAt: string | null;
  createdAt: string;
}

export interface FailureReason {
  itemId: string;
  reason: 'severity_blocking_expired' | 'severity_blocking_missing';
  expiredOnDate?: string;
}

export interface SnapshotResult {
  snapshotJson: {
    items: ItemSnapshot[];
    policyRuleSetVersion: string;
    jurisdictionDate: string;
    eligibilityVerdict: 'PASS' | 'FAIL';
    failureReasons: FailureReason[];
  };
  policyRuleSetVersion: string;
  eligibilityVerdict: 'PASS' | 'FAIL';
  failureReasons: FailureReason[];
}

export async function buildSnapshotForContractor(
  tx: Prisma.TransactionClient,
  contractorId: string,
  jurisdictionDate: string,
): Promise<SnapshotResult> {
  // Pull ALL BLOCKING-severity items regardless of status.
  const items = await tx.contractorComplianceItem.findMany({
    where: { contractorId, severity: 'BLOCKING' },
    select: {
      id: true,
      policyRuleId: true,
      severity: true,
      status: true,
      expiresAt: true,
      expiryJurisdictionTz: true,
      satisfiedByDocumentId: true,
      waivedReason: true,
      createdAt: true,
      // Note: waivedAt / satisfiedAt may not be on the existing schema today —
      // if they're not, drop them from the select and from the snapshot type
      // (and update the schema in Plan 72-02 if they're required).
    },
  });

  const itemSnapshots: ItemSnapshot[] = items.map(it => ({
    itemId: it.id,
    policyRuleId: it.policyRuleId,
    severity: it.severity!,
    status: it.status,
    expiresAt: it.expiresAt ? it.expiresAt.toISOString() : null,
    expiryJurisdictionTz: it.expiryJurisdictionTz,
    satisfiedByDocumentId: it.satisfiedByDocumentId,
    waivedReason: it.waivedReason ?? null,
    waivedAt: null,    // schema may not have this column — see note above
    satisfiedAt: null, // ditto
    createdAt: it.createdAt.toISOString(),
  }));

  const failureReasons: FailureReason[] = items
    .filter(it => it.severity === 'BLOCKING' && (it.status === 'EXPIRED' || it.status === 'MISSING'))
    .map(it => ({
      itemId: it.id,
      reason: it.status === 'EXPIRED' ? 'severity_blocking_expired' : 'severity_blocking_missing',
      expiredOnDate: it.expiresAt ? it.expiresAt.toISOString().slice(0, 10) : undefined,
    }));

  const eligibilityVerdict: 'PASS' | 'FAIL' = failureReasons.length === 0 ? 'PASS' : 'FAIL';
  const policyRuleSetVersion = getCurrentPolicyVersion();

  return {
    snapshotJson: {
      items: itemSnapshots,
      policyRuleSetVersion,
      jurisdictionDate,
      eligibilityVerdict,
      failureReasons,
    },
    policyRuleSetVersion,
    eligibilityVerdict,
    failureReasons,
  };
}
```
