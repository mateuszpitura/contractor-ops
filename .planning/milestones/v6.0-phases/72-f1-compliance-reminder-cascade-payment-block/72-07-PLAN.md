```tsx
'use client';

// Phase 72 D-10 — Payment-block modal renders the structured PRECONDITION_FAILED
// cause payload. One collapsible section per contractor; per-item deep links.

import { useTranslations } from 'next-intl';
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogFooter,
} from '@/components/ui/dialog';
import {
  Accordion,
  AccordionContent,
  AccordionItem,
  AccordionTrigger,
} from '@/components/ui/accordion';
import { Alert, AlertDescription, AlertTitle } from '@/components/ui/alert';
import { Button } from '@/components/ui/button';
import Link from 'next/link';
import { AlertTriangle, ExternalLink } from 'lucide-react';

export interface ItemReason {
  itemId: string;
  policyRuleId: string | null;
  documentTypeLabelKey: string;
  expiredOnDate: string;
  jurisdictionTz: string;
  deepLinkPath: string;
}

export interface ContractorReason {
  contractorId: string;
  contractorName: string;
  reasons: ItemReason[];
}

export interface PaymentBlockModalProps {
  open: boolean;
  onClose: () => void;
  contractorReasons: ContractorReason[];
}

export function PaymentBlockModal({ open, onClose, contractorReasons }: PaymentBlockModalProps) {
  const t = useTranslations('compliance');

  const totalContractors = contractorReasons.length;
  const totalItems = contractorReasons.reduce((acc, c) => acc + c.reasons.length, 0);

  return (
    <Dialog open={open} onOpenChange={(o) => { if (!o) onClose(); }}>
      <DialogContent className="max-w-2xl">
        <DialogHeader>
          <DialogTitle>{t('paymentBlockModal.title')}</DialogTitle>
          <DialogDescription>
            {t('paymentBlockModal.subtitle', { contractorCount: totalContractors, itemCount: totalItems })}
          </DialogDescription>
        </DialogHeader>

        {totalContractors === 0 ? (
          <Alert variant="destructive">
            <AlertTriangle className="h-4 w-4" />
            <AlertTitle>{t('paymentBlockModal.noReasonsTitle')}</AlertTitle>
            <AlertDescription>{t('paymentBlockModal.noReasonsDescription')}</AlertDescription>
          </Alert>
        ) : (
          <Accordion type="multiple" className="w-full">
            {contractorReasons.map((contractor) => (
              <AccordionItem key={contractor.contractorId} value={contractor.contractorId}>
                <AccordionTrigger className="text-left">
                  <span className="truncate" title={contractor.contractorName}>
                    {contractor.contractorName}
                  </span>
                  <span className="ml-2 text-sm text-muted-foreground">
                    {t('paymentBlockModal.itemCountSuffix', { count: contractor.reasons.length })}
                  </span>
                </AccordionTrigger>
                <AccordionContent>
                  <ul className="space-y-2">
                    {contractor.reasons.map((reason) => (
                      <li key={reason.itemId} className="flex items-start gap-2">
                        <Link
                          href={reason.deepLinkPath}
                          className="text-primary underline hover:no-underline inline-flex items-center gap-1"
                          target="_blank"
                          rel="noopener"
                          aria-label={t('paymentBlockModal.deepLinkAriaLabel', {
                            label: t(reason.documentTypeLabelKey),
                          })}
                        >
                          {t(reason.documentTypeLabelKey)}
                          <ExternalLink className="h-3 w-3" aria-hidden />
                        </Link>
                        <span className="text-sm text-muted-foreground">
                          {t('paymentBlockModal.expiredOn', {
                            date: reason.expiredOnDate,
                            tz: reason.jurisdictionTz,
                          })}
                        </span>
                      </li>
                    ))}
                  </ul>
                </AccordionContent>
              </AccordionItem>
            ))}
          </Accordion>
        )}

        <DialogFooter>
          <Button variant="default" onClick={onClose}>
            {t('paymentBlockModal.close')}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
```

In the wizard's submit handler:
```tsx
const createMutation = trpc.paymentRun.create.useMutation();
const [blockModalState, setBlockModalState] = useState<{ open: boolean; reasons: ContractorReason[] }>({ open: false, reasons: [] });

async function handleSubmit() {
  try {
    await createMutation.mutateAsync({ ... });
  } catch (err) {
    if (isPreconditionFailedWithReasons(err)) {
      setBlockModalState({ open: true, reasons: err.cause.contractorReasons });
      return;
    }
    throw err;
  }
}

function isPreconditionFailedWithReasons(err: unknown): err is { code: 'PRECONDITION_FAILED'; cause: { contractorReasons: ContractorReason[] } } {
  return (
    typeof err === 'object' && err !== null &&
    'data' in err && (err as any).data?.code === 'PRECONDITION_FAILED' &&
    'cause' in (err as any) && Array.isArray((err as any).cause?.contractorReasons)
  );
}

// In JSX:
<PaymentBlockModal
  open={blockModalState.open}
  onClose={() => setBlockModalState({ open: false, reasons: [] })}
  contractorReasons={blockModalState.reasons}
/>
```
