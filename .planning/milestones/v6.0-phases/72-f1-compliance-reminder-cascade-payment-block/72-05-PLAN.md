```ts
// Phase 72 D-13 â€” Plug-in operator registry for approval-engine conditions.
//
// Replaces the hardcoded switch statements in evaluateConditions with a
// Map-backed registry. Operators self-register at module-load via
// barrel-import side effects in operators/index.ts.

import type { Prisma, PrismaClient } from '@contractor-ops/db';

type TxClient = Parameters<Parameters<PrismaClient['$transaction']>[0]>[0];

export interface OperatorContext {
  tx: TxClient;
  contractorId: string;
  organizationId: string;
  invoice?: { totalMinor: number; contractorType?: string };  // legacy compat
}

export type ConditionEvaluator<TArgs = unknown> = (
  args: TArgs,
  ctx: OperatorContext,
) => Promise<boolean>;

const operators = new Map<string, ConditionEvaluator>();

export function registerOperator<TArgs>(name: string, evaluator: ConditionEvaluator<TArgs>): void {
  if (operators.has(name)) throw new Error(`Operator already registered: ${name}`);
  operators.set(name, evaluator as ConditionEvaluator);
}

export async function evaluateOperator(name: string, args: unknown, ctx: OperatorContext): Promise<boolean> {
  const evaluator = operators.get(name);
  if (!evaluator) throw new Error(`Unknown operator: ${name}`);
  return evaluator(args, ctx);
}

export function getRegisteredOperators(): string[] {
  return Array.from(operators.keys());
}

// Test-only: clear the registry between tests when needed.
// biome-ignore lint/style/useNamingConvention: test-internals
export const __testOnly = {
  reset(): void {
    operators.clear();
  },
};
```

```ts
async function checkComplianceHoldAtFinalStep(
  tx: TxClient,
  flow: { id: string; resourceType: string; resourceId: string; organizationId: string; chainConfigId: string | null },
): Promise<{ itemIds: string[]; heldAt: string; heldByOperator: string } | null> {
  // Resolve contractorId from the resource. For INVOICE, the contractor is on Invoice.contractorId.
  if (flow.resourceType !== 'INVOICE') return null;  // Phase 72 only gates invoice approvals
  const invoice = await tx.invoice.findUniqueOrThrow({
    where: { id: flow.resourceId },
    select: { contractorId: true },
  });
  if (!invoice.contractorId) return null;

  // Always run complianceCritical at final-step regardless of chain config; the
  // operator is the canonical gate per D-13. Chain-config conditions are
  // evaluated at routing time via evaluateConditions.
  const blocked = await evaluateOperator('complianceCritical', { status: 'EXPIRED' }, {
    tx,
    contractorId: invoice.contractorId,
    organizationId: flow.organizationId,
  });
  if (!blocked) return null;

  // Capture the holding itemIds for complianceHoldsJson.
  const items = await tx.contractorComplianceItem.findMany({
    where: { contractorId: invoice.contractorId, severity: 'BLOCKING', status: 'EXPIRED' },
    select: { id: true },
  });
  return {
    itemIds: items.map(i => i.id),
    heldAt: new Date().toISOString(),
    heldByOperator: 'complianceCritical',
  };
}
```

```ts
// Phase 72 D-15 â€” Recovery hooks for PENDING_COMPLIANCE flows.

import type { Prisma } from '@contractor-ops/db';
import { createLogger } from '@contractor-ops/logger';

import { assertContractorPaymentEligibility } from './compliance-payment-gate.js';
export { onComplianceItemExpiresAtChanged } from './compliance-reminder-scan.js';

const log = createLogger({ service: 'compliance-recovery' });

interface SatisfiedArgs {
  itemId: string;
  contractorId: string;
  organizationId: string;
}

/**
 * Listener invoked from classification.ts on item-satisfied events.
 * Releases held PENDING_COMPLIANCE flows whose complianceHoldsJson @>
 * { itemIds: [<itemId>] } AND whose re-asserted eligibility is PASS.
 */
export async function onComplianceItemSatisfied(
  tx: Prisma.TransactionClient,
  args: SatisfiedArgs,
): Promise<{ resumedFlowIds: string[] }> {
  const heldFlows: Array<{ id: string }> = await tx.$queryRaw`
    SELECT id FROM "ApprovalFlow"
    WHERE "status" = 'PENDING_COMPLIANCE'
      AND "organizationId" = ${args.organizationId}
      AND "complianceHoldsJson" @> ${JSON.stringify({ itemIds: [args.itemId] })}::jsonb
  `;
  const resumedFlowIds: string[] = [];
  for (const flow of heldFlows) {
    const eligibility = await assertContractorPaymentEligibility([args.contractorId], {
      tx,
      throwOnFail: false,
      organizationId: args.organizationId,
    });
    if (eligibility.blocked) {
      log.info({ flowId: flow.id, contractorId: args.contractorId, remainingReasons: eligibility.contractorReasons }, 'compliance-recovery: flow remains held â€” other items still block');
      continue;
    }
    await tx.approvalFlow.update({
      where: { id: flow.id },
      data: { status: 'PENDING', complianceHoldsJson: null },
    });
    await tx.auditLog.create({
      data: {
        organizationId: args.organizationId,
        actorType: 'SYSTEM',
        action: 'approval.compliance_resolved',
        resourceType: 'INVOICE',
        resourceId: flow.id,
        metadataJson: {
          releasedItemIds: [args.itemId],
          resolverEvent: 'item_satisfied',
          timestamp: new Date().toISOString(),
        },
      },
    });
    resumedFlowIds.push(flow.id);
  }
  if (resumedFlowIds.length > 0) {
    log.info({ resumedFlowIds, triggerItemId: args.itemId }, 'compliance-recovery: released held approvals');
  }
  return { resumedFlowIds };
}
```
