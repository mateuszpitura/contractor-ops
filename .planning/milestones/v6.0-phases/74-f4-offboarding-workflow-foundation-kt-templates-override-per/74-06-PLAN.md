         ```typescript
           accessToken: string,
           args: { calendarId?: string; timeMin: string; timeMax: string },
         ): Promise<{ busy: BusyRange[] }> {

             headers: { Authorization: `Bearer ${accessToken}`, 'Content-Type': 'application/json' },

               startTime: { dateTime: args.timeMin, timeZone: 'UTC' },
               endTime:   { dateTime: args.timeMax, timeZone: 'UTC' },
               availabilityViewInterval: 60,
             }),
           });
           if (!resp.ok) {
             const body = await resp.text();
             throw new Error(`Outlook getSchedule failed (${resp.status}): ${body}`);
           }
           const data = await resp.json() as { value?: Array<{ scheduleItems?: Array<{ status: string; subject?: string; start: { dateTime: string }; end: { dateTime: string }; isAllDay?: boolean }> }> };
           const items = data.value?.[0]?.scheduleItems ?? [];
           return {
         ```

       ```typescript
       import { TZDate } from '@date-fns/tz';
       import { startOfDay, endOfDay } from 'date-fns';
       import type { PrismaClient } from '@contractor-ops/db';
       import { PTO_KEYWORDS } from '@contractor-ops/offboarding-templates';
       import { createLogger } from '@contractor-ops/logger';
       import type { GoogleCalendarAdapter, OutlookCalendarAdapter, BusyRange } from '@contractor-ops/integrations';

       const logger = createLogger({ service: 'pto-detector' });

       export type PtoFallbackReason = 'manager_pto' | 'no_manager' | 'team_no_fallback';

       export interface ResolveAssigneeWithPtoArgs {
         prisma: PrismaClient;
         organizationId: string;
         managerUserId: string;
         teamId?: string | null;
         contractorJurisdictionTz: string;
         calendarAdapter: GoogleCalendarAdapter | OutlookCalendarAdapter | null;
         managerCalendarAccessToken?: string;
         managerCalendarId?: string;
       }

       export async function resolveAssigneeWithPto(
         args: ResolveAssigneeWithPtoArgs,
       ): Promise<{ assigneeUserId: string; fallbackReason?: PtoFallbackReason }> {
         const now = new Date();
         const todayLocalStart = startOfDay(new TZDate(now, args.contractorJurisdictionTz));
         const todayLocalEnd = endOfDay(new TZDate(now, args.contractorJurisdictionTz));

         // 1. Manual User.outOfOffice (always honoured, even with no calendar — D-07)
         const manager = await args.prisma.user.findUnique({
           where: { id: args.managerUserId },
           select: { outOfOffice: true },
         });
         if (manager?.outOfOffice) {
           const ooo = manager.outOfOffice as { from: string; until: string; fallbackUserId?: string };
           const oooStart = new TZDate(ooo.from, args.contractorJurisdictionTz);
           const oooEnd = new TZDate(ooo.until, args.contractorJurisdictionTz);
           if (todayLocalStart >= oooStart && todayLocalEnd <= oooEnd) {
             return resolveFallback(args, ooo.fallbackUserId, 'manager_pto');
           }
         }

         // 2. Calendar lookup (skip when no integration — D-07)
         if (args.calendarAdapter && args.managerCalendarAccessToken) {
           try {
             const isOnPto = await detectPtoFromCalendar({
               adapter: args.calendarAdapter,
               accessToken: args.managerCalendarAccessToken,
               calendarId: args.managerCalendarId,
               timeMin: todayLocalStart.toISOString(),
               timeMax: todayLocalEnd.toISOString(),
             });
             if (isOnPto) {
               return resolveFallback(args, undefined, 'manager_pto');
             }
           } catch (err) {
             logger.warn({ err, managerUserId: args.managerUserId }, 'Calendar PTO lookup failed; routing to manager');
             // Fail open — route to manager when calendar unreachable; ops user can re-route manually
           }
         }

         // 3. Default — route to manager
         return { assigneeUserId: args.managerUserId };
       }

       async function detectPtoFromCalendar(args: {
         adapter: GoogleCalendarAdapter | OutlookCalendarAdapter;
         accessToken: string;
         calendarId?: string;
         timeMin: string;
         timeMax: string;
       }): Promise<boolean> {
         const { busy } = await args.adapter.getFreeBusy(args.accessToken, {
           calendarId: args.calendarId,
           timeMin: args.timeMin,
           timeMax: args.timeMax,
         });
         const allKeywordsLower = [
           ...PTO_KEYWORDS.en,
           ...PTO_KEYWORDS.de,
           ...PTO_KEYWORDS.pl,
         ].map(k => k.toLowerCase());
         return busy.some((b: BusyRange) => {
           if (b.isAllDay) return true; // D-08 literal: any all-day busy → PTO
           const summaryLower = (b.summary ?? '').toLowerCase();
           return allKeywordsLower.some(k => summaryLower.includes(k));
         });
       }

       async function resolveFallback(
         args: ResolveAssigneeWithPtoArgs,
         explicitFallbackUserId: string | undefined,
         reason: PtoFallbackReason,
       ): Promise<{ assigneeUserId: string; fallbackReason: PtoFallbackReason }> {
         // 1. Per-user fallback (User.outOfOffice.fallbackUserId)
         if (explicitFallbackUserId) {
           return { assigneeUserId: explicitFallbackUserId, fallbackReason: reason };
         }
         // 2. Per-team fallback (Team.fallbackApproverId)
         if (args.teamId) {
           const team = await args.prisma.team.findFirst({
             where: { id: args.teamId, organizationId: args.organizationId },
             select: { fallbackApproverId: true },
           });
           if (team?.fallbackApproverId) {
             return { assigneeUserId: team.fallbackApproverId, fallbackReason: reason };
           }
         }
         // 3. Owner-role users (with admin-attention badge — Plan 74-08 renders)
         const ownerUser = await args.prisma.user.findFirst({
           where: {
             // Use whatever the canonical owner-role membership query is. Schema-dependent. Often via Member table:
             memberships: { some: { organizationId: args.organizationId, role: 'owner' } },
           },
           select: { id: true },
           orderBy: { createdAt: 'asc' }, // deterministic
         });
         if (ownerUser) {
           return { assigneeUserId: ownerUser.id, fallbackReason: 'team_no_fallback' };
         }
         // 4. Last-resort — return manager (caller logs admin-attention)
         logger.error({ organizationId: args.organizationId, managerUserId: args.managerUserId }, 'No fallback found; routing to manager despite PTO');
         return { assigneeUserId: args.managerUserId, fallbackReason: 'no_manager' };
       }
       ```
