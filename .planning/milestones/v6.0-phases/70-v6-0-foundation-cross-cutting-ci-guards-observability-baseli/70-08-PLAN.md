```ts
// Phase 70 D-15 — IdP deprovisioning audit logger.
//
// Returns a Pino child logger that:
//   - Inherits auth/secret redaction from PII_MASK_PATHS (passwords, tokens,
//     apiKeys, contractor PII)
//   - EXPLICITLY omits the '*.body' path that Plan 70-03 added at the root —
//     audit lines emit structured fields, not raw bodies, and the audit
//     `scopeDelta` field MUST survive in plaintext for compliance fidelity.
//
// Consumer: Phases 76–78 (IdP deprovisioning F2) — `getIdpAuditLogger().info(
//   { auditEvent: 'deprovision', externalUserId: 'usr_123', actionResult: 'OK',
//     provider: 'google', connectionId: 'c_x', scopeDelta: { revoked: ['user.deprovision'] } },
//   'IdP deprovision applied');`

import type { Logger } from 'pino';
import { logger as rootLogger } from './index.js';
import { PII_MASK_PATHS } from './pii-mask.js';

/**
 * Canonical schema of fields permitted on IdP audit log lines.
 *
 * Adding to this list is a deliberate audit-schema change — coordinate with
 * the compliance team before merging. CI does not currently enforce closure
 * (the type below is a partial map) but the documented contract is
 * "extras-discouraged".
 */
export const IDP_AUDIT_ALLOWED_FIELDS = [
  'auditEvent',
  'externalUserId',
  'actionResult',
  'provider',
  'connectionId',
  'scopeDelta',
  'organizationId',
  'userId',
  'timestamp',
] as const;

export type IdpAuditAllowedField = (typeof IDP_AUDIT_ALLOWED_FIELDS)[number];

/**
 * Strongly-typed audit event payload. `auditEvent` is mandatory; remaining
 * fields are optional but constrained to the allow-list.
 */
export type IdpAuditEvent = {
  [K in IdpAuditAllowedField]?: unknown;
} & {
  auditEvent: string;
};

/**
 * Returns the IdP audit child logger.
 *
 * The child inherits the root logger's PII redact paths, EXCEPT '*.body'
 * which is intentionally omitted so the audit `scopeDelta` and similar
 * fields survive in plaintext.
 *
 * Bindings `service: 'idp-audit'` are present on every line, enabling
 * downstream routing (e.g., separate Axiom dataset — deferred per CONTEXT.md).
 */
export function getIdpAuditLogger(): Logger {
  const auditPaths = PII_MASK_PATHS.filter(p => p !== '*.body');
  return rootLogger.child(
    { service: 'idp-audit' },
    { redact: { paths: auditPaths as string[], censor: '[REDACTED]' } },
  );
}
```

```ts
import { describe, expect, it, vi } from 'vitest';
import { getIdpAuditLogger, IDP_AUDIT_ALLOWED_FIELDS, type IdpAuditEvent } from '../index.js';

function captureLogs<T>(fn: () => T): { result: T; output: string } {
  const captured: string[] = [];
  const writeSpy = vi.spyOn(process.stdout, 'write').mockImplementation((s: any) => {
    captured.push(typeof s === 'string' ? s : s.toString());
    return true;
  });
  try {
    const result = fn();
    return { result, output: captured.join('') };
  } finally {
    writeSpy.mockRestore();
  }
}

describe('getIdpAuditLogger (FOUND6-06 — D-15)', () => {
  it('emits externalUserId in plaintext', () => {
    const { output } = captureLogs(() => {
      getIdpAuditLogger().info(
        { auditEvent: 'deprovision', externalUserId: 'usr_abc123' } satisfies IdpAuditEvent,
        'audit',
      );
    });
    expect(output).toContain('usr_abc123');
    expect(output).not.toContain('[REDACTED]');
  });

  it('emits scopeDelta in plaintext (D-15: audit field allow-list)', () => {
    const { output } = captureLogs(() => {
      getIdpAuditLogger().info(
        {
          auditEvent: 'reconnect',
          scopeDelta: { added: ['user.deprovision'], removed: [] },
        } satisfies IdpAuditEvent,
        'scope upgrade',
      );
    });
    expect(output).toContain('user.deprovision');
  });

  it('redacts password / token / apiKey even though emitted by audit logger', () => {
    const { output } = captureLogs(() => {
      // Cast to bypass IdpAuditEvent type-narrowing — we're testing redaction
      // behavior at the *.body / *.password etc. PII paths.
      const log = getIdpAuditLogger();
      log.info(
        { auditEvent: 'rotate', password: 'should-redact', token: 'tok-xyz', apiKey: 'ak-123' } as any,
        'rotate creds',
      );
    });
    expect(output).not.toContain('should-redact');
    expect(output).not.toContain('tok-xyz');
    expect(output).not.toContain('ak-123');
    expect(output).toContain('[REDACTED]');
  });

  it('does NOT redact body field (audit logger override of Plan 70-03 default-redact)', () => {
    const { output } = captureLogs(() => {
      const log = getIdpAuditLogger();
      // Body is a structural audit field for some events — must be plaintext.
      log.info({ auditEvent: 'webhook-received', body: { event_type: 'user.suspended' } } as any, 'webhook');
    });
    expect(output).toContain('user.suspended');
    expect(output).not.toContain('[REDACTED]');
  });

  it('binds service: idp-audit on every line', () => {
    const { output } = captureLogs(() => {
      getIdpAuditLogger().info({ auditEvent: 'noop' } satisfies IdpAuditEvent, 'noop');
    });
    expect(output).toContain('"service":"idp-audit"');
  });

  it('IDP_AUDIT_ALLOWED_FIELDS contains the canonical fields', () => {
    expect(IDP_AUDIT_ALLOWED_FIELDS).toEqual([
      'auditEvent',
      'externalUserId',
      'actionResult',
      'provider',
      'connectionId',
      'scopeDelta',
      'organizationId',
      'userId',
      'timestamp',
    ]);
  });
});
```
