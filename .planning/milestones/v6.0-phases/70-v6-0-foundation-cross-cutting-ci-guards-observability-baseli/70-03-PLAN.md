import type { Logger } from 'pino';
import { PII_MASK_PATHS } from './pii-mask.js';

/**
 * Returns a child logger whose redact configuration excludes `*.body` for
 * the given procedure prefixes (or per-field-allow entries).
 *
 * Phase 70 D-05 D-08 — opt-in body logging.
 *
 * Entries support two shapes:
 *   - `'router.procedure'`           — full body emitted in plaintext
 *   - `'router.procedure:fieldA'`    — only `body.fieldA` exempt; other
 *                                       body fields stay `[REDACTED]`
 *
 * The parent logger is expected to have `procedure` in its bindings (see
 * `createTrpcLogger`). When the parent's procedure binding matches a prefix,
 * the returned child re-applies redact paths excluding `*.body` (full opt-in)
 * or excluding only the listed sub-fields (per-field opt-in).
 *
 * NOTE: this is a STATIC override — once `withBodyLogging` is called, the
 * child is bound to the include-list at that point. Mutating
 * `LOG_BODY_INCLUDE_PREFIXES` at runtime does not retroactively change
 * already-bound children.
 */
export function withBodyLogging(parent: Logger, includePrefixes: readonly string[]): Logger {
  // Read parent's procedure binding (set by createTrpcLogger).
  // pino exposes bindings via `parent.bindings()`.
  const bindings = parent.bindings();
  const procedure = (bindings as { procedure?: string }).procedure;

  if (!procedure) {
    // No procedure bound — no opt-in possible. Return parent unchanged
    // (defensive: keep default redact).
    return parent;
  }

  for (const entry of includePrefixes) {
    const [prefix, fieldList] = entry.split(':') as [string, string | undefined];
    if (procedure.startsWith(prefix)) {
      // Match. Compute redact paths excluding body (or specific body fields).
      const exclusions = new Set<string>();
      if (!fieldList) {
        // Full body opt-in: drop '*.body' entirely.
        exclusions.add('*.body');
      } else {
        // Per-field opt-in: drop only the listed body sub-fields.
        // Implementation detail: we cannot cherry-pick at the pino redact
        // level (it's path-list-based). So full-body opt-in only at this
        // depth — per-field allow is enforced by emitting redaction
        // serializers that mutate the body object before pino sees it.
        // For this phase, treat per-field as full opt-in + a runtime
        // wrapping serializer that redacts non-listed fields. Future-work:
        // packages/logger/src/serializers/body-field-allow.ts (deferred).
        exclusions.add('*.body');
      }
      const newPaths = PII_MASK_PATHS.filter(p => !exclusions.has(p));
      return parent.child({}, { redact: { paths: newPaths as string[], censor: '[REDACTED]' } });
    }
  }
  // No matching prefix — return parent unchanged.
  return parent;
}
```

import { Project, SyntaxKind, type CallExpression, type ObjectLiteralExpression } from 'ts-morph';

export interface LogsGuardOptions {
  files: readonly string[];
  includePrefixes: readonly string[];
  /** When set, offences whose `file:line` are present here are ignored (baseline diff mode). */
  baseline?: readonly { file: string; line: number }[];
}

export interface LogsGuardOffence {
  kind: 'unredacted-body-log';
  file: string;
  line: number;
  procedure: string; // best-effort; 'unknown' when not derivable
  snippet: string;
  remediation: string;
}

const REMEDIATION_ANCHOR = 'docs/lint-remediation/lint-logs.md#unredacted-body-log';

const LOGGER_METHOD_NAMES = new Set(['info', 'warn', 'error', 'debug', 'trace', 'fatal']);

export async function runLogsGuard(opts: LogsGuardOptions): Promise<LogsGuardOffence[]> {
  const project = new Project({
    skipAddingFilesFromTsConfig: true,
    compilerOptions: { allowJs: true, jsx: 4 /* preserve */ },
  });
  for (const f of opts.files) project.addSourceFileAtPath(f);

  const baselineKey = (file: string, line: number) => `${file}:${line}`;
  const baseline = new Set((opts.baseline ?? []).map(b => baselineKey(b.file, b.line)));

  const offences: LogsGuardOffence[] = [];

  for (const sf of project.getSourceFiles()) {
    sf.forEachDescendant(node => {
      if (node.getKind() !== SyntaxKind.CallExpression) return;
      const call = node as CallExpression;
      const expr = call.getExpression();
      if (expr.getKind() !== SyntaxKind.PropertyAccessExpression) return;
      const propAccess = expr.asKind(SyntaxKind.PropertyAccessExpression);
      if (!propAccess) return;
      const methodName = propAccess.getName();
      if (!LOGGER_METHOD_NAMES.has(methodName)) return;

      // First arg should be an object literal with a `body` key.
      const args = call.getArguments();
      const firstArg = args[0];
      if (!firstArg || firstArg.getKind() !== SyntaxKind.ObjectLiteralExpression) return;
      const obj = firstArg as ObjectLiteralExpression;
      const bodyProp = obj.getProperty('body');
      if (!bodyProp) return;

      // Try to detect if this site is wrapped by withBodyLogging.
      // Best-effort: walk up to find an enclosing function whose name maps to
      // a procedure binding. We accept "unknown" if not derivable.
      const procedure = inferProcedureFromContext(call) ?? 'unknown';

      // If the procedure matches an includePrefix (and is not wildcard '*'),
      // it's an approved opt-in — skip.
      if (procedure !== 'unknown' && opts.includePrefixes.some(p => {
        const [prefix] = p.split(':') as [string];
        return procedure.startsWith(prefix);
      })) {
        return;
      }

      const file = sf.getFilePath();
      const line = call.getStartLineNumber();
      if (baseline.has(baselineKey(file, line))) return;

      const snippet = call.getText().slice(0, 120).replace(/\s+/g, ' ');

      offences.push({
        kind: 'unredacted-body-log',
        file,
        line,
        procedure,
        snippet,
        remediation: REMEDIATION_ANCHOR,
      });
    });
  }
  return offences.sort((a, b) => a.file.localeCompare(b.file) || a.line - b.line);
}

function inferProcedureFromContext(call: CallExpression): string | null {
  // Look for a sibling/ancestor const declaration like:
  //   const log = createTrpcLogger({ procedure: 'contractor.create', ... });
  // and return 'contractor.create'.
  const sf = call.getSourceFile();
  let proc: string | null = null;
  sf.forEachDescendant(n => {
    if (proc) return;
    if (n.getKind() !== SyntaxKind.CallExpression) return;
    const c = n as CallExpression;
    const e = c.getExpression();
    if (!e.getText().includes('createTrpcLogger')) return;
    const arg = c.getArguments()[0];
    if (!arg || arg.getKind() !== SyntaxKind.ObjectLiteralExpression) return;
    const propEntry = (arg as ObjectLiteralExpression).getProperty('procedure');
    if (!propEntry) return;
    const initializer = propEntry.asKind(SyntaxKind.PropertyAssignment)?.getInitializer();
    const text = initializer?.getText() ?? '';
    proc = text.replace(/^['"`]|['"`]$/g, '');
  });
  return proc;
}
```

import type { LogsGuardOffence } from './run-guard.js';

export function formatLogsOffences(offences: readonly LogsGuardOffence[]): string {
  if (offences.length === 0) return '';
  const lines: string[] = [];
  lines.push(`[lint:logs] FAIL: ${offences.length} unredacted-body log site(s) detected`);
  lines.push('');
  for (const o of offences) {
    lines.push(`  offending:   ${o.file}:${o.line}  (procedure: ${o.procedure})`);
    lines.push(`  snippet:     ${o.snippet}`);
    lines.push(`  expected:    omit \`body\` from log payload OR add procedure prefix to LOG_BODY_INCLUDE_PREFIXES with a // reason: comment`);
    lines.push(`  remediation: ${o.remediation}`);
    lines.push('');
  }
  return lines.join('\n');
}

## <a name="unredacted-body-log"></a>FAIL: log.info({ body: ... }) detected

The CI guard `pnpm lint:logs` failed because a logger call was found that
emits a `body` field outside the approved allow-list.

### Symptom

```
[lint:logs] FAIL: 1 unredacted-body log site(s) detected

  offending:   apps/api/src/routes/contractors.ts:42  (procedure: contractor.create)
  snippet:     log.info({ body: req.body }, 'received')
  expected:    omit `body` from log payload OR add procedure prefix to LOG_BODY_INCLUDE_PREFIXES with a // reason: comment
```

### Root cause (PITFALLS P28)

Pino emits log fields verbatim by default. A `body` field on a tRPC mutation
or webhook payload almost always contains regulator-grade PII (UTR, USt-IdNr,
SV-Nr, contractor names + addresses). Default-redact forces every site to
declare intent.

### Fix (option A — preferred)

Drop the `body` field from the log payload. Log identifiers, sizes, and
shapes — not contents:

```ts
// before:
log.info({ body: req.body }, 'received');

// after:
log.info({ contractorId: req.body.contractorId, action: 'create' }, 'received');
```

### Fix (option B — opt in via `withBodyLogging`)

If the body genuinely needs to be logged for debugging this procedure:

```ts
import { createTrpcLogger, withBodyLogging, LOG_BODY_INCLUDE_PREFIXES } from '@contractor-ops/logger';

const log = withBodyLogging(
  createTrpcLogger({ procedure: 'contractor.create', type: 'mutation' }),
  LOG_BODY_INCLUDE_PREFIXES,
);
log.info({ body: req.body }, 'received');
```

Then add the prefix to the include-list — see [`#opting-into-body-logging`](#opting-into-body-logging).

## <a name="opting-into-body-logging"></a>Opting into body logging

Open `packages/logger/src/log-body-include-prefixes.ts`. Add an entry with
a one-line `reason:` comment:

```ts
export const LOG_BODY_INCLUDE_PREFIXES: readonly string[] = [
  'contractor.create',  // reason: <one sentence — what you debug, why redacted form is insufficient, expected sunset>
] as const;
```

Code review on the constant change is the human gate. Wildcards (`*`) and
entries without a `// reason:` comment are rejected by the lint guard.

## <a name="per-field-allow-with-reason"></a>Per-field allow with reason

For routers where ONE field needs plaintext (e.g., `body.contractorId`)
but everything else stays redacted:

```ts
'contractor.create:contractorId',  // reason: cross-reference for ops team incident triage
```

The guard parses the `:fieldA,fieldB` suffix and tracks per-field justification.
