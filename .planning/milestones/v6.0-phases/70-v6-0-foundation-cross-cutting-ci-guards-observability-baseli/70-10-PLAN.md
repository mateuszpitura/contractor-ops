```tsx
'use client';

import { useTranslations } from 'next-intl';
import type { ScopeCapabilities } from '@contractor-ops/db';
import { Button } from '@/components/ui/button';
import { Card, CardHeader, CardTitle, CardDescription, CardContent, CardFooter } from '@/components/ui/card';
import { AlertCircle } from 'lucide-react';

export interface GoogleWorkspaceReconnectBannerProps {
  scopeCapabilities: ScopeCapabilities | null;
  /** OAuth start URL — defaults to the existing v3.0 endpoint. */
  reconnectHref?: string;
}

const REQUIRED_CAPABILITY = 'user.deprovision' as const;

/**
 * Phase 70 D-16 — Banner shown when an existing v3.0 Google Workspace
 * connection lacks the IdP-deprovisioning capability that Phase 76 will need.
 *
 * Visibility rules:
 *   - scopeCapabilities is null (legacy connection, never backfilled OR
 *     pre-D-14 grant) → show
 *   - scopeCapabilities.capabilities does NOT contain 'user.deprovision' → show
 *   - capabilities contains 'user.deprovision' → hide (already opted in)
 *
 * IMPORTANT: Phase 70 ships ZERO new OAuth scopes. The Reconnect button
 * routes to the EXISTING OAuth start URL with the EXISTING scope set.
 * Phase 76 (IdP deprovisioning) updates the scope set, at which point the
 * Reconnect flow grants the new capability and the banner self-removes.
 */
export function GoogleWorkspaceReconnectBanner({
  scopeCapabilities,
  reconnectHref = '/api/oauth/google_workspace/start',
}: GoogleWorkspaceReconnectBannerProps) {
  const t = useTranslations('Integrations.GoogleWorkspaceReconnect');

  const needsReconnect =
    scopeCapabilities === null ||
    !scopeCapabilities.capabilities.includes(REQUIRED_CAPABILITY);

  if (!needsReconnect) return null;

  return (
    <Card role="region" aria-label={t('bannerTitle')} className="border-amber-500/50 bg-amber-50 dark:bg-amber-950/30">
      <CardHeader className="flex flex-row items-center gap-3 ps-4 pe-4">
        <AlertCircle className="h-5 w-5 text-amber-600 dark:text-amber-400" aria-hidden />
        <div className="flex-1">
          <CardTitle className="text-base">{t('bannerTitle')}</CardTitle>
          <CardDescription>{t('bannerBody')}</CardDescription>
        </div>
      </CardHeader>
      <CardFooter className="ps-4 pe-4 pb-4">
        <Button asChild>
          <a href={reconnectHref}>{t('reconnectButton')}</a>
        </Button>
      </CardFooter>
    </Card>
  );
}
```
