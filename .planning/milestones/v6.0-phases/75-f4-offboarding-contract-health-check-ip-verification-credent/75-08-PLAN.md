```tsx
'use client';

import { useTranslations } from 'next-intl';
import { ipAssignmentResultsSchema, type IpAssignmentResults } from '@contractor-ops/validators/legal/ip-clauses-results-schema';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { trpc } from '@/lib/trpc';
import { Loader2 } from 'lucide-react';

interface HealthCheckPanelProps {
  contractId: string;
  resultsJson: unknown; // Contract.complianceFlagsJson or ContractHealthCheckRun.resultsJson
  /** When true, hides the Re-run button (used in audit-log drill-in modal where the view is immutable). */
  readOnly?: boolean;
}

export function HealthCheckPanel({ contractId, resultsJson, readOnly }: HealthCheckPanelProps): JSX.Element {
  const t = useTranslations('contracts.healthCheck');
  const utils = trpc.useUtils();
  const rerun = trpc.contract.rerunHealthCheck.useMutation({
    onSuccess: () => {
      void utils.contract.invalidate();
    },
  });

  const parseResult = ipAssignmentResultsSchema.safeParse(resultsJson);
  if (!parseResult.success) {
    return (
      <Card>
        <CardHeader>
          <CardTitle>{t('title')}</CardTitle>
        </CardHeader>
        <CardContent>
          <p className="text-muted-foreground">{t('schemaUnsupported')}</p>
        </CardContent>
      </Card>
    );
  }

  const results: IpAssignmentResults = parseResult.data;
  const { verdict, citedClauses, crossJurisdictionMismatch, pendingPhrasesCited } = results.ipAssignment;

  const verdictVariant = verdict === 'LIKELY_PRESENT'
    ? 'success'
    : verdict === 'LIKELY_MISSING'
    ? 'warning'
    : 'destructive';

  return (
    <Card>
      <CardHeader className="flex flex-row items-center justify-between">
        <CardTitle>{t('title')}</CardTitle>
        <div className="flex items-center gap-2">
          <Badge variant={verdictVariant}>{t(`verdict.${verdict}`)}</Badge>
          {!readOnly && (
            <Button
              size="sm"
              variant="outline"
              onClick={() => rerun.mutate({ contractIds: [contractId] })}
              disabled={rerun.isPending}
            >
              {rerun.isPending ? <Loader2 className="size-4 animate-spin" /> : t('rerunButton')}
            </Button>
          )}
        </div>
      </CardHeader>
      <CardContent className="space-y-4">
        {crossJurisdictionMismatch && (
          <div className="rounded border border-amber-300 bg-amber-50 p-3 text-amber-900">
            <p className="font-semibold">{t('crossJurisdictionMismatch.title')}</p>
            <p className="text-sm">
              {t('crossJurisdictionMismatch.description', {
                found: crossJurisdictionMismatch.foundJurisdiction,
                expected: crossJurisdictionMismatch.expectedJurisdiction,
              })}
            </p>
          </div>
        )}

        {citedClauses.length === 0 ? (
          <p className="text-sm text-muted-foreground">{t('noCitedClauses')}</p>
        ) : (
          <ul className="space-y-3">
            {citedClauses.map((clause, idx) => {
              const isPending = pendingPhrasesCited?.includes(clause.phraseId) ?? false;
              return (
                <li key={`${clause.phraseId}-${idx}`} className="border-l-2 border-l-blue-300 pl-3">
                  <div className="flex items-center gap-2 text-xs">
                    <Badge variant="outline">{clause.jurisdiction}</Badge>
                    <Badge variant={clause.regexMatched ? 'default' : 'secondary'}>
                      {clause.regexMatched ? t('source.regexAndLlm') : t('source.llmOnly')}
                    </Badge>
                    <span className="text-muted-foreground">
                      {t('confidence', { value: Math.round(clause.confidence * 100) })}
                    </span>
                    {isPending && (
                      <sup data-testid="pending-phrase-marker" className="text-amber-600">¹</sup>
                    )}
                  </div>
                  <blockquote className="mt-1 italic">{clause.citedText}</blockquote>
                </li>
              );
            })}
          </ul>
        )}

        {(pendingPhrasesCited?.length ?? 0) > 0 && (
          <p className="text-xs text-amber-700" data-testid="pending-phrase-footer">
            {t('pendingPhraseFooter')}
          </p>
        )}
      </CardContent>
    </Card>
  );
}
```

```ts
startIpRatificationSigning: tenantProcedure
  .use(requirePermission({ workflow: ['execute'] }))
  .input(z.object({ workflowTaskRunId: z.string().min(1) }))
  .mutation(async ({ ctx, input }) => {
    const task = await ctx.db.workflowTaskRun.findFirst({
      where: { id: input.workflowTaskRunId, organizationId: ctx.organizationId },
      include: {
        workflowRun: {
          include: {
            contract: { select: { id: true, jurisdiction: true } },
            contractor: { select: { country: true } },
          },
        },
      },
    });
    if (!task || task.taskType !== 'IP_VERIFICATION') {
      throw new TRPCError({ code: 'BAD_REQUEST', message: 'Task is not an IP_VERIFICATION task' });
    }

    // Resolve jurisdiction
    const isoToJurisdiction: Record<string, string> = {
      GBR: 'UK', UK: 'UK', DEU: 'DE', DE: 'DE', POL: 'PL', PL: 'PL',
      USA: 'US', US: 'US', SAU: 'KSA', KSA: 'KSA', ARE: 'UAE', UAE: 'UAE',
    };
    const rawIso = task.workflowRun.contract?.jurisdiction
      ?? task.workflowRun.contractor?.country
      ?? 'GBR';
    const jurisdiction = isoToJurisdiction[rawIso.toUpperCase()] ?? 'UK';

    // Load template
    const fs = await import('node:fs/promises');
    const templatePath = `packages/api/src/services/contract-health/templates/ip-ratification-${jurisdiction.toLowerCase()}.txt`;
    const templateText = await fs.readFile(templatePath, 'utf8');

    // Choose adapter — DocuSign for everyone except DE
    const adapterSlug = jurisdiction === 'DE' ? 'autenti' : 'docusign';
    const { getAdapter } = await import('@contractor-ops/integrations/registry');
    const adapter = getAdapter(adapterSlug);
    if (!adapter || !('createSigningEnvelope' in adapter)) {
      throw new TRPCError({ code: 'INTERNAL_SERVER_ERROR', message: `e-sign adapter ${adapterSlug} not registered` });
    }

    // Adapter contract — verify by reading docusign-adapter.ts createSigningEnvelope signature
    const envelope = await (adapter as { createSigningEnvelope: (input: unknown) => Promise<{ envelopeId: string; signingUrl: string }> }).createSigningEnvelope({
      organizationId: ctx.organizationId,
      documentType: 'IP_RATIFICATION',
      documentText: templateText,
      metadata: { workflowTaskRunId: input.workflowTaskRunId, workflowRunId: task.workflowRunId, jurisdiction },
      signerEmail: ctx.user.email,
      signerName: ctx.user.name ?? ctx.user.email,
    });

    return { provider: adapterSlug, envelopeId: envelope.envelopeId, signingUrl: envelope.signingUrl };
  }),
```
