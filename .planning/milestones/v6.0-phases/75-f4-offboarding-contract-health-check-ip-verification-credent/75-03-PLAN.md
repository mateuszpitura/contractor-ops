```ts
describe('Phase 75 — IP-assignment policy rules (D-07)', () => {
  // Sequencing assumption: registry tests already import all jurisdiction modules
  // via the package barrel (verified in Task 75-03-07).

  const phase75Rules = [
    { id: 'uk.ip_assignment@v1', jurisdiction: 'UK', tz: 'Europe/London' },
    { id: 'de.werkvertrag_ip@v1', jurisdiction: 'DE', tz: 'Europe/Berlin' },
    { id: 'pl.ip_assignment@v1', jurisdiction: 'PL', tz: 'Europe/Warsaw' },
    { id: 'us.ip_assignment@v1', jurisdiction: 'US', tz: 'America/New_York' },
    { id: 'ksa.ip_assignment@v1', jurisdiction: 'KSA', tz: 'Asia/Riyadh' },
    { id: 'uae.ip_assignment@v1', jurisdiction: 'UAE', tz: 'Asia/Dubai' },
  ];

  for (const rule of phase75Rules) {
    it(`registers ${rule.id} as severity WARNING with documentType IP_RATIFICATION`, () => {
      const resolved = resolvePolicyRules({
        jurisdiction: rule.jurisdiction as never,
        outcome: 'IR35-OUTSIDE', // any outcome works; appliesIf returns true unconditionally
        engagementType: 'B2B',
      });
      const target = resolved.find(r => r.policyRuleId === rule.id);
      expect(target).toBeDefined();
      expect(target?.severity).toBe('WARNING');
      expect(target?.documentType).toBe('IP_RATIFICATION');
      expect(target?.expiryJurisdictionTz).toBe(rule.tz);
    });
  }

  it('all 6 IP-assignment rules carry the PENDING legal review marker in draftLegalText', () => {
    const ids = phase75Rules.map(r => r.id);
    const resolved = ids.map(id =>
      resolvePolicyRules({ jurisdiction: id.split('.')[0]?.toUpperCase() as never, outcome: 'IR35-OUTSIDE', engagementType: 'B2B' })
        .find(r => r.policyRuleId === id),
    );
    for (const r of resolved) {
      expect(r?.draftLegalText).toMatch(/PENDING legal review/);
    }
  });

  it('the DE rule cites both §31 UrhG and §7 UrhG (Schöpferprinzip)', () => {
    const resolved = resolvePolicyRules({ jurisdiction: 'DE', outcome: 'IR35-OUTSIDE', engagementType: 'B2B' });
    const de = resolved.find(r => r.policyRuleId === 'de.werkvertrag_ip@v1');
    expect(de?.draftLegalText).toMatch(/§31/);
    expect(de?.draftLegalText).toMatch(/§7/);
    expect(de?.draftLegalText).toMatch(/Sch[öo]pferprinzip/);
    expect(de?.draftLegalText).toMatch(/INSUFFICIENT/);
  });
});
```
