async suspendAccount(externalUserId: string): Promise<DeprovisionResult> {
  const accessToken = await this.#getAccessToken();
  const requestPayload = { suspended: true };
  const requestSha256 = sha256Hex(canonicalizeRequest(requestPayload));

  const url = `https://admin.googleapis.com/admin/directory/v1/users/${encodeURIComponent(externalUserId)}`;
  let res: Response;
  try {
    res = await fetch(url, {
      method: 'PATCH',
      headers: {
        'Authorization': `Bearer ${accessToken}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(requestPayload),
    });
  } catch (err: any) {
    return {
      status: 'FAILED',
      failureKind: 'NETWORK',
      errorMessage: String(err?.message ?? err).slice(0, 1024),
      requestSha256,
      responseSha256: 'unhashed',
    };
  }

  const responseBody = await res.json().catch(() => ({}));
  const responseSha256 = sha256Hex(canonicalizeResponse(responseBody));

  if (res.status === 404) {
    return { status: 'SUCCEEDED', failureKind: 'USER_NOT_FOUND', errorMessage: 'User not found at provider — already absent', requestSha256, responseSha256 };
  }
  if (res.status === 429) {
    return { status: 'FAILED', failureKind: 'RATE_LIMITED', errorMessage: 'Rate limited', requestSha256, responseSha256 };
  }
  if (res.status === 401 || res.status === 403) {
    return { status: 'FAILED', failureKind: 'AUTH_REVOKED', errorMessage: 'Auth rejected by provider — re-OAuth required', requestSha256, responseSha256 };
  }
  if (res.status >= 500) {
    return { status: 'FAILED', failureKind: 'PROVIDER_ERROR', errorMessage: `Provider error ${res.status}`, requestSha256, responseSha256 };
  }
  if (res.status >= 200 && res.status < 300) {
    return { status: 'SUCCEEDED', requestSha256, responseSha256 };
  }
  return { status: 'FAILED', failureKind: 'UNKNOWN', errorMessage: `Unexpected status ${res.status}`, requestSha256, responseSha256 };
}

async revokeAllSessions(externalUserId: string): Promise<DeprovisionResult> {
  const accessToken = await this.#getAccessToken();
  const requestPayload = {};  // signOut endpoint takes no body
  const requestSha256 = sha256Hex(canonicalizeRequest({ method: 'POST', target: 'signOut' }));

  const url = `https://admin.googleapis.com/admin/directory/v1/users/${encodeURIComponent(externalUserId)}/signOut`;
  let res: Response;
  try {
    res = await fetch(url, {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${accessToken}` },
    });
  } catch (err: any) {
    return {
      status: 'FAILED',
      failureKind: 'NETWORK',
      errorMessage: String(err?.message ?? err).slice(0, 1024),
      requestSha256,
      responseSha256: 'unhashed',
    };
  }

  const responseBody = await res.text().catch(() => '');
  const responseSha256 = sha256Hex(canonicalizeResponse({ status: res.status, bodyLength: responseBody.length }));

  if (res.status === 404) return { status: 'SUCCEEDED', failureKind: 'USER_NOT_FOUND', requestSha256, responseSha256 };
  if (res.status === 429) return { status: 'FAILED', failureKind: 'RATE_LIMITED', requestSha256, responseSha256 };
  if (res.status === 401 || res.status === 403) return { status: 'FAILED', failureKind: 'AUTH_REVOKED', requestSha256, responseSha256 };
  if (res.status >= 500) return { status: 'FAILED', failureKind: 'PROVIDER_ERROR', requestSha256, responseSha256 };
  if (res.status >= 200 && res.status < 300) return { status: 'SUCCEEDED', requestSha256, responseSha256 };
  return { status: 'FAILED', failureKind: 'UNKNOWN', requestSha256, responseSha256 };
}

async verifyDeprovisioned(externalUserId: string): Promise<boolean> {
  const accessToken = await this.#getAccessToken();
  try {
    const res = await fetch(
      `https://admin.googleapis.com/admin/directory/v1/users/${encodeURIComponent(externalUserId)}`,
      { headers: { 'Authorization': `Bearer ${accessToken}` } },
    );
    if (res.status === 404) return true;  // user gone is also "deprovisioned"
    const data = await res.json().catch(() => ({}));
    return Boolean(data?.suspended);
  } catch {
    return false;
  }
}

override async handleWebhook(
  payload: unknown,
  organizationId: string,
  connectionId: string,
): Promise<unknown> {
  // Phase 76 D-09..D-12 — self-trigger filter.
  // Parse the payload; if it's a user-suspend event, check provenance.
  if (typeof payload === 'object' && payload && 'event' in payload) {
    const event = payload as { event: string; userId?: string };
    if (event.event === 'user.suspended' && event.userId) {
      const matched = await provenanceLookup(db, {
        organizationId,
        provider: 'GOOGLE_WORKSPACE',
        externalUserId: event.userId,
        actionKind: 'SUSPEND',
      });
      if (matched) {
        return { suppressed: true, provenanceId: matched.id };
      }
    }
  }
  // No match (or non-suspend event) → fall through to default v3.0 user-departed path
  return undefined;
}
