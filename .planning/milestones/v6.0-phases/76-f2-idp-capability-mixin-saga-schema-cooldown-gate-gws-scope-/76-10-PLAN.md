```tsx
import { describe, it, expect, vi } from 'vitest';
import { execSync } from 'node:child_process';
import { render, screen } from '@testing-library/react';
import { renderWithProviders } from '@/test/test-utils';

describe('No "Reactivate contractor" button anywhere (Phase 76 SC#7 / IDP-15)', () => {
  it('static grep across apps/web/src returns no matches for /reactivate.*contractor/i', () => {
    const out = execSync(
      'grep -rIE -i "reactivate.*contractor|re-?activate.{0,3}contractor" apps/web/src apps/web/messages || true',
      { encoding: 'utf8', cwd: process.cwd() },
    );
    expect(out.trim()).toBe('');
  });

  it('RTL: contractor profile page does not render any "Reactivate" labelled button', async () => {
    // Strategy: render the contractor profile page with a contractor whose status would
    // historically have triggered a "Reactivate" button (e.g., assignment.status = ENDED).
    // Assert no button matches the label.
    //
    // Mock all tRPC queries to return null/empty data — we only care about button absence.
    vi.mock('@/trpc/react', () => ({
      trpc: {
        contractor: { getById: { useQuery: () => ({ data: { id: 'c-1', status: 'INACTIVE' } }) } },
        contractorAssignment: {
          listByContractor: {
            useQuery: () => ({ data: [{ id: 'a-1', status: 'ENDED', endedAt: new Date() }] }),
          },
        },
      },
    }));

    // Dynamic import to ensure the mock applies before the component module loads
    const { default: ContractorProfilePage } = await import(
      '@/app/[locale]/contractors/[id]/page'
    ).catch(() => ({ default: () => null }));  // graceful fallback if path differs

    renderWithProviders(<ContractorProfilePage params={{ locale: 'en', id: 'c-1' }} />);
    const reactivateButtons = screen.queryAllByRole('button', { name: /reactivate/i });
    expect(reactivateButtons).toEqual([]);
  });

  it('RTL: ENDED contractors list does not render any "Reactivate" labelled button', async () => {
    // Similar strategy for the contractors-list page filtered to ENDED status.
    vi.mock('@/trpc/react', () => ({
      trpc: {
        contractor: { list: { useQuery: () => ({ data: [{ id: 'c-1', status: 'INACTIVE' }] }) } },
      },
    }));

    const { default: ContractorsListPage } = await import(
      '@/app/[locale]/contractors/page'
    ).catch(() => ({ default: () => null }));

    renderWithProviders(<ContractorsListPage params={{ locale: 'en' }} />);
    const reactivateButtons = screen.queryAllByRole('button', { name: /reactivate/i });
    expect(reactivateButtons).toEqual([]);
  });
});
```
