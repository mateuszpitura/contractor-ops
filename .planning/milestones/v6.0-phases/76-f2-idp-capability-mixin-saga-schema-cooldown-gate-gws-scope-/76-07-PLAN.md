```ts
// Phase 76 D-15 — lint:scopes guard.
//
// Asserts that every scope string in any IdP adapter's getOAuthConfig().scopes
// array is traceable to either:
//   (a) a typed-const declared in the same file, OR
//   (b) an imported typed-const from `packages/integrations/src/scopes/*.ts`.
//
// Adding a scope literal directly without putting it in a typed-const fails CI.

import { Project, type SourceFile, type ArrayLiteralExpression } from 'ts-morph';
import * as path from 'node:path';

export interface ScopesOffence {
  adapter: string;
  scope: string;
  remediation: string;
}

export interface ScopesGuardResult {
  ok: boolean;
  offences: ScopesOffence[];
}

const ADAPTERS_GLOB = 'packages/integrations/src/adapters/*.ts';
const SCOPES_DIR = 'packages/integrations/src/scopes';
const SKIP_PATTERNS = [/__tests__/, /__fixtures__/];
// Trigger heuristic — any scope matching these requires typed-const trace.
const WRITE_SCOPE_PATTERNS = [
  /admin\.directory\.user(?!\.readonly)/,
  /\.write/,
  /admin:org/,
  /admin\.users\.session:write/,
  /scim:write/,
  /User\.EnableDisableAccount\.All/,
  /User\.RevokeSessions\.All/,
];

export function runScopesGuard(opts: { tsConfigFilePath?: string; fixturesAllowed?: boolean } = {}): ScopesGuardResult {
  const project = new Project({ tsConfigFilePath: opts.tsConfigFilePath ?? 'tsconfig.base.json' });
  const adapterFiles = project.getSourceFiles(ADAPTERS_GLOB).filter(f =>
    !SKIP_PATTERNS.some(re => re.test(f.getFilePath())),
  );
  const offences: ScopesOffence[] = [];

  // Build the allow-set: all string literals in `*_DEPROVISION_SCOPES` const declarations across scopes/*.ts
  const scopeFiles = project.getSourceFiles(`${SCOPES_DIR}/*.ts`);
  const globalAllowSet = new Set<string>();
  for (const file of scopeFiles) {
    for (const decl of file.getVariableDeclarations()) {
      const init = decl.getInitializer();
      if (init && init.getKindName() === 'ArrayLiteralExpression') {
        const arr = init as ArrayLiteralExpression;
        for (const el of arr.getElements()) {
          if (el.getKindName() === 'StringLiteral') {
            globalAllowSet.add(el.getText().slice(1, -1));  // strip quotes
          }
        }
      }
    }
  }

  for (const adapter of adapterFiles) {
    const adapterName = path.basename(adapter.getFilePath());
    // Build the file-local allow-set: every string literal inside any const-as-const array
    const localAllowSet = new Set<string>(globalAllowSet);
    for (const decl of adapter.getVariableDeclarations()) {
      const init = decl.getInitializer();
      if (init && init.getKindName() === 'ArrayLiteralExpression') {
        const arr = init as ArrayLiteralExpression;
        for (const el of arr.getElements()) {
          if (el.getKindName() === 'StringLiteral') {
            localAllowSet.add(el.getText().slice(1, -1));
          }
        }
      }
    }

    // Find getOAuthConfig().scopes inline literals that match write patterns
    const inlineLiterals = findInlineScopeLiterals(adapter);
    for (const scope of inlineLiterals) {
      if (!WRITE_SCOPE_PATTERNS.some(re => re.test(scope))) continue;  // ignore read-only scopes
      if (localAllowSet.has(scope)) continue;  // traced to typed-const
      offences.push({
        adapter: adapterName,
        scope,
        remediation: `Move "${scope}" into a typed-const in ${SCOPES_DIR}/<provider>-deprovision-scopes.ts and import-spread into getOAuthConfig().scopes.`,
      });
    }
  }

  return { ok: offences.length === 0, offences };
}

function findInlineScopeLiterals(file: SourceFile): string[] {
  const literals: string[] = [];
  // Walk all string literals in the file. Only literals that appear inside
  // a `scopes: [...]` property are interesting. Heuristic: any string literal
  // matching the OAuth scope shape (URL or `admin:org`-style identifier).
  for (const node of file.forEachDescendantAsArray()) {
    if (node.getKindName() !== 'StringLiteral') continue;
    const value = node.getText().slice(1, -1);
    if (/^https:\/\/.+|^admin:|^User\./.test(value)) {
      literals.push(value);
    }
  }
  return literals;
}
```

Note the simple heuristic: the guard checks STRING LITERALS in the file that look like OAuth scopes (URL-shaped or `admin:org`-shaped). Anything matching the WRITE patterns must be traceable to a typed-const. False-positives on read-only literals are skipped because the WRITE_SCOPE_PATTERNS regex excludes `.readonly`.
