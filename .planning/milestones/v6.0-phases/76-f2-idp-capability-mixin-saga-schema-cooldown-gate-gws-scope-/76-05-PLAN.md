```ts
import { z } from 'zod';
import { TRPCError } from '@trpc/server';
import { canStartDeprovisioning } from '@contractor-ops/idp-saga';
import { getIdpAuditLogger } from '@contractor-ops/logger';
import { protectedProcedure, router } from '../trpc.js';

const log = getIdpAuditLogger();

export const deprovisioningRouter = router({
  /**
   * Phase 76 D-05/D-07 — Eligibility query.
   *
   * Single source of truth for the cooldown gate. Consumed by:
   *   - UI: button-disabled state + tooltip (Plan 76-08)
   *   - Server: same helper from `startDeprovisioningRun` mutation (Plan 76-06)
   *
   * Returns `{ allowed, earliestDate?, reason? }`. UI cannot lie about state —
   * the mutation re-runs the SAME helper before doing anything destructive.
   */
  getDeprovisioningEligibility: protectedProcedure
    .input(z.object({ assignmentId: z.string().min(1) }))
    .query(async ({ ctx, input }) => {
      const assignment = await ctx.db.contractorAssignment.findFirstOrThrow({
        where: { id: input.assignmentId, organizationId: ctx.organizationId },
        select: {
          id: true, status: true, endedAt: true,
          contractor: { select: { id: true, jurisdictionTz: true } },
        },
      });

      const tz = assignment.contractor.jurisdictionTz ?? 'Europe/Berlin';
      if (!assignment.contractor.jurisdictionTz) {
        log.warn(
          {
            auditEvent: 'deprovision_eligibility_default_tz_used',
            organizationId: ctx.organizationId,
            userId: ctx.userId,
          },
          'Contractor missing jurisdictionTz — defaulting to Europe/Berlin for cooldown computation',
        );
      }

      const decision = canStartDeprovisioning({
        endedAt: assignment.endedAt ?? null,
        jurisdictionTz: tz,
        status: assignment.status,
      });

      log.info(
        {
          auditEvent: 'deprovision_eligibility_checked',
          organizationId: ctx.organizationId,
          userId: ctx.userId,
          // assignmentId is NOT in IDP_AUDIT_ALLOWED_FIELDS — store as part of stepId-style derivation
          // Use organizationId + the actionResult to capture the decision
          actionResult: decision.allowed ? 'ALLOWED' : 'COOLDOWN_ACTIVE',
        },
        'Deprovisioning eligibility checked',
      );

      return decision;
    }),
});
```

import { describe, it, expect, vi } from 'vitest';
import { TRPCError } from '@trpc/server';
import { appRouter } from '../root.js';

// Reuse the existing test-context helper from sibling tests.
// The pattern: build a mock Prisma client + a mock ctx with organizationId + userId,
// then call the procedure via appRouter.createCaller(ctx).deprovisioning.getDeprovisioningEligibility(input).

const makeCtx = (overrides: Record<string, any> = {}) => {
  const findFirstOrThrow = vi.fn();
  const ctx = {
    db: {
      contractorAssignment: { findFirstOrThrow },
    },
    organizationId: 'org-1',
    userId: 'user-1',
    ...overrides,
  } as any;
  return { ctx, findFirstOrThrow };
};

describe('getDeprovisioningEligibility query (Phase 76 D-05/D-07)', () => {
  it('returns CooldownDecision for an ENDED assignment in cooldown', async () => {
    const { ctx, findFirstOrThrow } = makeCtx();
    const tenDaysAgo = new Date(Date.now() - 10 * 24 * 60 * 60 * 1000);
    findFirstOrThrow.mockResolvedValue({
      id: 'a-1', status: 'ENDED', endedAt: tenDaysAgo,
      contractor: { id: 'c-1', jurisdictionTz: 'Europe/Berlin' },
    });

    const caller = appRouter.createCaller(ctx);
    const result = await caller.deprovisioning.getDeprovisioningEligibility({ assignmentId: 'a-1' });
    expect(result.allowed).toBe(false);
    expect(result.reason).toMatch(/14-day cooldown active/);
    expect(result.earliestDate).toBeInstanceOf(Date);
  });

  it('returns allowed: true when 14d cooldown elapsed', async () => {
    const { ctx, findFirstOrThrow } = makeCtx();
    const fifteenDaysAgo = new Date(Date.now() - 15 * 24 * 60 * 60 * 1000);
    findFirstOrThrow.mockResolvedValue({
      id: 'a-1', status: 'ENDED', endedAt: fifteenDaysAgo,
      contractor: { id: 'c-1', jurisdictionTz: 'Europe/Berlin' },
    });

    const caller = appRouter.createCaller(ctx);
    const result = await caller.deprovisioning.getDeprovisioningEligibility({ assignmentId: 'a-1' });
    expect(result.allowed).toBe(true);
  });

  it('rejects when assignment not found in caller organization (cross-tenant leak guard)', async () => {
    const { ctx, findFirstOrThrow } = makeCtx();
    findFirstOrThrow.mockRejectedValue(new Error('Record not found'));

    const caller = appRouter.createCaller(ctx);
    await expect(
      caller.deprovisioning.getDeprovisioningEligibility({ assignmentId: 'a-other-tenant' }),
    ).rejects.toThrow();
  });

  it('defaults to Europe/Berlin TZ + warns when contractor.jurisdictionTz is null', async () => {
    const { ctx, findFirstOrThrow } = makeCtx();
    findFirstOrThrow.mockResolvedValue({
      id: 'a-1', status: 'ENDED', endedAt: new Date(),
      contractor: { id: 'c-1', jurisdictionTz: null },
    });

    const caller = appRouter.createCaller(ctx);
    const result = await caller.deprovisioning.getDeprovisioningEligibility({ assignmentId: 'a-1' });
    expect(result.allowed).toBe(false);  // 0d ago: still in cooldown
    // (warn emission verified by structured-log integration tests; here we just verify the call succeeds)
  });
});
