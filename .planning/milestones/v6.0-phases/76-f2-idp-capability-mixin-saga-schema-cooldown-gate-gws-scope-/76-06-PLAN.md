```ts
// Phase 76 D-15 / SC#2 — Canonicalisation helpers for SOC2 evidence-grade hashes.
//
// The hash is computed on a deterministic JSON serialisation with PII removed.
// This means: the same logical request from two different orchestration paths
// hashes to the same value, AND the hash never embeds raw user data.
//
// PII denylist: 'token', 'accessToken', 'refreshToken', 'apiKey', 'password',
// 'authorization', 'Authorization', 'externalUserId' (we use the externalUserId
// for the provenance lookup, but the hash that goes into DeprovisioningStep
// must NOT embed it — the step.externalUserId column is the audit channel).

import { createHash } from 'node:crypto';

const PII_DENYLIST = new Set([
  'token', 'accessToken', 'refreshToken', 'apiKey', 'password',
  'authorization', 'Authorization', 'externalUserId',
]);

function redact(value: unknown): unknown {
  if (value === null || typeof value !== 'object') return value;
  if (Array.isArray(value)) return value.map(redact);
  const obj: Record<string, unknown> = {};
  for (const [k, v] of Object.entries(value as Record<string, unknown>)) {
    if (PII_DENYLIST.has(k)) continue;
    obj[k] = redact(v);
  }
  return obj;
}

function canonicalJson(value: unknown): string {
  if (value === null || typeof value !== 'object') return JSON.stringify(value);
  if (Array.isArray(value)) return `[${value.map(canonicalJson).join(',')}]`;
  const entries = Object.entries(value as Record<string, unknown>).sort(([a], [b]) => a.localeCompare(b));
  return `{${entries.map(([k, v]) => `${JSON.stringify(k)}:${canonicalJson(v)}`).join(',')}}`;
}

export function canonicalizeRequest(req: unknown): string {
  return canonicalJson(redact(req));
}

export function canonicalizeResponse(res: unknown): string {
  return canonicalJson(redact(res));
}

export function sha256Hex(input: string): string {
  return createHash('sha256').update(input, 'utf8').digest('hex');
}
```

```ts
      idempotencyKey: z.string().min(8).max(128),

        where: { id: input.assignmentId, organizationId: ctx.organizationId },
        select: {
          id: true, status: true, endedAt: true, contractorId: true,

          userId: ctx.userId,
          runId: step.runId,
          stepId: step.id,
          stepKind: step.stepKind,
          provider: step.provider,
        },
        'Deprovisioning step manually retried',
      );

      return { ok: true };
    }),
});
```

```ts
// Phase 76 D-03 — QStash step runner.
//
// One independent QStash job per (provider, stepKind) tuple. NO Promise.allSettled
// aggregation per Pitfall 10. Aggregate run status is computed by recomputeRunStatus
// after every step transition (CONTEXT.md D-02).

import { NextResponse } from 'next/server';
import { verifySignatureAppRouter } from '@upstash/qstash/nextjs';
import { z } from 'zod';
import { db } from '@contractor-ops/db';
import { recomputeRunStatus } from '@contractor-ops/idp-saga';
import { insertProvenance } from '@contractor-ops/idp-saga';
import { MAX_ATTEMPTS } from '@contractor-ops/idp-saga';
import { getDeprovisionableAdapter } from '@contractor-ops/integrations';
import { getIdpAuditLogger } from '@contractor-ops/logger';

const log = getIdpAuditLogger();

const BodySchema = z.object({
  runId: z.string(),
  stepId: z.string(),
  organizationId: z.string(),
  provider: z.enum(['GOOGLE_WORKSPACE', 'SLACK', 'ENTRA', 'OKTA', 'GITHUB']),
  stepKind: z.enum(['SUSPEND_ACCOUNT', 'REVOKE_ALL_SESSIONS']),
  externalUserId: z.string(),
});

async function handler(req: Request) {
  const raw = await req.json();
  const body = BodySchema.parse(raw);

  // 1. Read step + check attempts cap
  const step = await db.deprovisioningStep.findUniqueOrThrow({
    where: { id: body.stepId },
    select: { id: true, runId: true, status: true, attempts: true },
  });

  if (step.attempts >= MAX_ATTEMPTS) {
    await db.deprovisioningStep.update({
      where: { id: step.id },
      data: { status: 'FAILED', finishedAt: new Date(), lastErrorMessage: 'MAX_ATTEMPTS exceeded' },
    });
    await recomputeRunStatus(db, step.runId);
    return NextResponse.json({ ok: false, reason: 'max-attempts' }, { status: 200 });
  }

  // 2. Mark step IN_PROGRESS + increment attempts atomically
  await db.deprovisioningStep.update({
    where: { id: step.id },
    data: {
      status: 'IN_PROGRESS',
      attempts: { increment: 1 },
      startedAt: step.attempts === 0 ? new Date() : undefined,
    },
  });

  // 3. Insert provenance row BEFORE adapter call (D-09 ordering)
  await insertProvenance(db, {
    organizationId: body.organizationId,
    provider: body.provider,
    externalUserId: body.externalUserId,
    actionKind: body.stepKind === 'SUSPEND_ACCOUNT' ? 'SUSPEND' : 'REVOKE_SESSION',
    deprovisioningStepId: step.id,
  });

  // 4. Resolve adapter + invoke method
  let result;
  try {
    const adapter = getDeprovisionableAdapter(body.provider);
    result = body.stepKind === 'SUSPEND_ACCOUNT'
      ? await adapter.suspendAccount(body.externalUserId)
      : await adapter.revokeAllSessions(body.externalUserId);
  } catch (err: any) {
    result = {
      status: 'FAILED' as const,
      failureKind: 'UNKNOWN' as const,
      errorMessage: String(err?.message ?? err).slice(0, 1024),
      requestSha256: 'unhashed',
      responseSha256: 'unhashed',
    };
  }

  // 5. USER_NOT_FOUND → success-equivalent rule (RESEARCH "DeprovisionResult type shape")
  const finalStatus = result.status === 'SUCCEEDED' || result.failureKind === 'USER_NOT_FOUND'
    ? 'SUCCEEDED'
    : 'FAILED';
  const sanitisedError = result.errorMessage ? result.errorMessage.slice(0, 1024) : null;

  await db.deprovisioningStep.update({
    where: { id: step.id },
    data: {
      status: finalStatus,
      finishedAt: new Date(),
      requestSha256: result.requestSha256,
      responseSha256: result.responseSha256,
      lastErrorMessage: sanitisedError,
    },
  });

  // 6. Emit audit entry
  log.info(
    {
      auditEvent: 'deprovision_step_completed',
      organizationId: body.organizationId,
      runId: body.runId,
      stepId: body.stepId,
      stepKind: body.stepKind,
      provider: body.provider,
      externalUserId: body.externalUserId,
      actionResult: finalStatus,
      attempts: step.attempts + 1,
      requestSha256: result.requestSha256,
      responseSha256: result.responseSha256,
      failureKind: result.failureKind,
    },
    'Deprovisioning step completed',
  );

  // 7. Recompute aggregate run status (D-02 single source of truth)
  await recomputeRunStatus(db, step.runId);

  return NextResponse.json({ ok: finalStatus === 'SUCCEEDED' }, { status: 200 });
}

export const POST = verifySignatureAppRouter(handler);
export const dynamic = 'force-dynamic';
```
