```tsx
'use client';

import { useState } from 'react';
import { useTranslations } from 'next-intl';
import { toast } from 'sonner';  // adjust per discovery 71-06-01
import { trpc } from '~/trpc/react';  // adjust per discovery 71-06-01

// Match the closed enum from packages/api/src/routers/classification.ts
type RecomputeReason = 'policy_version_bump' | 'classification_outcome_change' | 'admin_correction';

interface Props {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  contractorIds: string[];
  /** Optional: callback after successful mutation, e.g., to refresh the parent. */
  onSuccess?: () => void;
}

export function RecomputeComplianceDialog({ open, onOpenChange, contractorIds, onSuccess }: Props) {
  const t = useTranslations('Contractors.Compliance.Recompute');
  const [reason, setReason] = useState<RecomputeReason | null>(null);

  const mutation = trpc.classification.recreateComplianceAssessment.useMutation({
    onSuccess: data => {
      const updated = data.results
        .filter(r => !('noop' in r) || r.noop === false)
        .filter(r => !('error' in r))
        .reduce((acc, r) => acc + (('waivedCount' in r) ? r.waivedCount : 0) + (('insertedCount' in r) ? r.insertedCount : 0), 0);
      const skipped = data.results.filter(r => 'noop' in r && r.noop === true).length;
      const errored = data.results.filter(r => 'error' in r).length;

      if (errored > 0) {
        toast.warning(t('toast.partial', { updated, skipped, errored }));
      } else if (skipped > 0) {
        toast.success(t('toast.success_with_skipped', { updated, skipped }));
      } else {
        toast.success(t('toast.success', { updated }));
      }
      onOpenChange(false);
      setReason(null);
      onSuccess?.();
    },
    onError: err => {
      toast.error(t('toast.error', { message: err.message }));
    },
  });

  const handleConfirm = () => {
    if (!reason) return;
    mutation.mutate({ contractorIds, reason });
  };

  // Use the project's existing Dialog/Select primitives — shadcn-ui or custom
  // The Dialog/Select imports come from existing components; use the same path
  // pattern as revalidate-vat-button.tsx or contractor-wizard/wizard-dialog.tsx.

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>
            {contractorIds.length === 1
              ? t('dialog.title.single')
              : t('dialog.title.bulk', { count: contractorIds.length })}
          </DialogTitle>
          <DialogDescription>
            {t('dialog.description')}
          </DialogDescription>
        </DialogHeader>

        <div className="py-4">
          <label htmlFor="recompute-reason" className="block text-sm font-medium mb-2">
            {t('dialog.reason_label')}
          </label>
          <Select value={reason ?? undefined} onValueChange={v => setReason(v as RecomputeReason)}>
            <SelectTrigger id="recompute-reason" data-testid="recompute-reason-select">
              <SelectValue placeholder={t('dialog.reason_placeholder')} />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="policy_version_bump">{t('reasons.policy_version_bump')}</SelectItem>
              <SelectItem value="classification_outcome_change">{t('reasons.classification_outcome_change')}</SelectItem>
              <SelectItem value="admin_correction">{t('reasons.admin_correction')}</SelectItem>
            </SelectContent>
          </Select>
        </div>

        <DialogFooter>
          <Button variant="outline" onClick={() => onOpenChange(false)}>
            {t('dialog.cancel')}
          </Button>
          <Button
            onClick={handleConfirm}
            disabled={!reason || mutation.isPending}
            data-testid="recompute-confirm"
          >
            {mutation.isPending ? t('dialog.confirming') : t('dialog.confirm')}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
```

'use client';

import { useState } from 'react';
import { useTranslations } from 'next-intl';
import { Button } from '~/components/ui/button';  // adjust per project
import { RecomputeComplianceDialog } from './recompute-compliance-dialog';

interface Props {
  contractorId: string;
  /** Optional: callback after successful recompute, e.g., to refresh the parent. */
  onSuccess?: () => void;
  /** Optional: render in a smaller variant (admin tab footer). */
  size?: 'sm' | 'default';
}

export function RecomputeComplianceButton({ contractorId, onSuccess, size = 'default' }: Props) {
  const t = useTranslations('Contractors.Compliance.Recompute');
  const [dialogOpen, setDialogOpen] = useState(false);

  return (
    <>
      <Button
        variant="outline"
        size={size}
        onClick={() => setDialogOpen(true)}
        data-testid="recompute-compliance-button"
      >
        {t('button.label')}
      </Button>
      <RecomputeComplianceDialog
        open={dialogOpen}
        onOpenChange={setDialogOpen}
        contractorIds={[contractorId]}
        onSuccess={onSuccess}
      />
    </>
  );
}
