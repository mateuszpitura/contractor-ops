```tsx
'use client';

// Phase 73 · Plan 06 — useComplDocName hook (D-15, D-16).
//
// Resolves a Phase 71 PolicyRuleId to the matching locked-phrase from the
// per-jurisdiction registry (Plan 73-04). Renders a `¹` superscript when the
// corresponding signoff-registry entry is PENDING — visible "this phrasing
// is unverified" signal until per-jurisdiction legal review flips to APPROVED.

import { useTranslations, useLocale } from 'next-intl';
import rawSignoffRegistry from '@contractor-ops/validators/legal/signoff-registry.json' with { type: 'json' };

const SIGNOFF_REGISTRY = rawSignoffRegistry as Record<string, { status: 'PENDING' | 'APPROVED' }>;

function flatKey(policyRuleId: string): string {
  return `COMPL_DOCNAME_${policyRuleId.replace(/\./g, '_').replace(/@v/g, '_v')}`;
}

interface ComplDocName {
  /** The translated string in the active locale. */
  text: string;
  /** True if signoff entry is PENDING — UI should append a `¹` footnote. */
  isPending: boolean;
}

export function useComplDocName(policyRuleId: string | null | undefined): ComplDocName {
  const t = useTranslations('compliance.docName');
  const locale = useLocale();

  if (!policyRuleId) {
    return { text: '', isPending: false };
  }

  // Strip @vN — the i18n key is per stable namespace, not per version.
  const m = /^([a-z]+)\.([a-z][a-z_0-9]*)@v\d+$/.exec(policyRuleId);
  if (!m) {
    return { text: policyRuleId, isPending: false };
  }
  const [, jurisdiction, namespace] = m;

  // Resolve via i18n message-key. If missing, fall back to the policyRuleId.
  let text: string;
  try {
    text = t(`${jurisdiction}.${namespace}` as Parameters<typeof t>[0]);
  } catch {
    text = policyRuleId;
  }

  const signoffStatus = SIGNOFF_REGISTRY[flatKey(policyRuleId)]?.status;
  const isPending = signoffStatus === 'PENDING';

  // Suppress unused-locale warning; locale may drive future per-locale fallback
  void locale;
  return { text, isPending };
}
```

```tsx
'use client';

// Phase 73 · Plan 06 — Upcoming renewals tab (D-03, D-05).

import { useQuery } from '@tanstack/react-query';
import { useTranslations } from 'next-intl';
import { Skeleton } from '@/components/ui/skeleton';
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from '@/components/ui/table';
import { Link } from '@/i18n/navigation';
import { trpc } from '@/trpc/init';
import { useComplDocName } from './use-compl-doc-name';

export function UpcomingRenewalsTable() {
  const t = useTranslations('compliance.dashboard.tables');
  const { data, isLoading } = useQuery(trpc.compliance.dashboardUpcomingRenewals.queryOptions());

  if (isLoading) return <Skeleton className="h-[300px] w-full" />;
  if (!data || data.length === 0) {
    return <p className="py-12 text-center text-sm text-muted-foreground">{t('emptyUpcomingRenewals')}</p>;
  }

  return (
    <Table>
      <TableHeader>
        <TableRow>
          <TableHead>{t('contractorName')}</TableHead>
          <TableHead>{t('document')}</TableHead>
          <TableHead>{t('expiresAt')}</TableHead>
          <TableHead className="hidden md:table-cell">{t('daysToExpiry')}</TableHead>
        </TableRow>
      </TableHeader>
      <TableBody>
        {data.map((row) => {
          const { text: docName, isPending } = useComplDocName(row.policyRuleId);
          const contractorName = row.contractor?.displayName ?? row.contractor?.businessName ?? '—';
          const daysToExpiry = row.expiresAt
            ? Math.ceil((new Date(row.expiresAt).getTime() - Date.now()) / (1000 * 60 * 60 * 24))
            : null;
          return (
            <TableRow key={row.id}>
              <TableCell>
                <Link href={`/contractors/${row.contractorId}/compliance#item-${row.id}`} className="hover:underline">
                  {contractorName}
                </Link>
              </TableCell>
              <TableCell>
                {docName}
                {isPending && <sup className="ml-0.5 text-xs text-muted-foreground">1</sup>}
              </TableCell>
              <TableCell>{row.expiresAt ? new Date(row.expiresAt).toISOString().slice(0, 10) : '—'}</TableCell>
              <TableCell className="hidden md:table-cell text-sm text-muted-foreground">
                {daysToExpiry !== null ? t('daysCount', { count: daysToExpiry }) : '—'}
              </TableCell>
            </TableRow>
          );
        })}
      </TableBody>
    </Table>
  );
}
```
