```ts
overrideItem: tenantProcedure
  .use(requirePermission({ compliance: ['override'] }))
  .input(overrideItemInput)
  .mutation(async ({ ctx, input }) => {
    return await ctx.db.$transaction(async (tx) => {
      const before = await tx.contractorComplianceItem.findFirstOrThrow({
        where: { id: input.itemId, organizationId: ctx.organizationId },
      });
      if (before.status === 'WAIVED') {
        throw new TRPCError({
          code: 'PRECONDITION_FAILED',
          message: 'Item is already WAIVED',
        });
      }
      const updated = await tx.contractorComplianceItem.update({
        where: { id: input.itemId },
        data: {
          status: 'WAIVED',
          waivedReason: 'admin_manual_waive',
          waivedReasonCategory: input.reasonCategory,
          waivedReasonNote: input.reasonNote,
        },
      });
      const auditWriterTx = tx as unknown as Parameters<typeof writeAuditLog>[0]['tx'];
      await writeAuditLog({
        organizationId: ctx.organizationId,
        actorType: 'USER',
        actorId: ctx.user?.id ?? null,
        actorName: ctx.user?.name ?? null,
        action: 'compliance.item.overridden',
        resourceType: 'CONTRACTOR',
        resourceId: before.contractorId,
        metadata: {
          itemId: input.itemId,
          reasonCategory: input.reasonCategory,
          reasonNote: input.reasonNote,
          previousStatus: before.status,
          actorRoleSnapshot: (ctx as unknown as { session?: { role?: string } }).session?.role ?? null,
        },
        tx: auditWriterTx,
      });
```

```ts
itemAuditTrail: tenantProcedure
  .use(requirePermission({ compliance: ['read'] }))
  .input(z.object({ itemId: z.string().min(1), limit: z.number().int().min(1).max(200).default(50) }))
  .query(async ({ ctx, input }) => {
    // First confirm the item belongs to this org (defence in depth)
    const item = await ctx.db.contractorComplianceItem.findFirst({
      where: { id: input.itemId, organizationId: ctx.organizationId },
      select: { id: true, contractorId: true },
    });
    if (!item) {
      throw new TRPCError({ code: 'NOT_FOUND', message: 'Compliance item not found' });
    }
    // Use the partial GIN index from Plan 73-02 — `metadataJson @> {itemId}` is indexed.
    const trail = await ctx.db.auditLog.findMany({
      where: {
        organizationId: ctx.organizationId,
        resourceType: 'CONTRACTOR',
        resourceId: item.contractorId,
        metadataJson: { path: ['itemId'], equals: input.itemId } as Prisma.JsonFilter,
      },
      orderBy: { createdAt: 'desc' },
      take: input.limit,
    });
    return trail;
  }),
```

Note: Prisma's typed JSON filter `{ path: ['itemId'], equals: ... }` translates to `metadataJson #>> '{itemId}' = ?` in Postgres. This does NOT use the GIN index directly. If the typed filter does not use the index, switch to `$queryRaw` with `metadataJson @> '{"itemId":"..."}'::jsonb` — Plan 73-08 verifies the EXPLAIN plan; if the typed filter is sequential-scan, refactor to raw query at that point. For Phase 73 we ship the typed filter (correct semantics, simpler code) and accept slow-path until verified.
