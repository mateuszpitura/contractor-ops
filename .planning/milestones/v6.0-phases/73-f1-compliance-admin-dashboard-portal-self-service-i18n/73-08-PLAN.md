```tsx
'use client';

// Phase 73 · Plan 08 — Manual override modal (D-12).

import { useMutation, useQueryClient } from '@tanstack/react-query';
import { useTranslations } from 'next-intl';
import { useState } from 'react';
import { toast } from 'sonner';
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
} from '@/components/ui/alert-dialog';
import { Label } from '@/components/ui/label';
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from '@/components/ui/select';
import { Textarea } from '@/components/ui/textarea';
import { trpc } from '@/trpc/init';

type OverrideReason =
  | 'contractor_offboarded'
  | 'engagement_changed'
  | 'regulatory_exemption'
  | 'temporary_grace_period'
  | 'admin_correction'
  | 'other';

const REASON_OPTIONS: OverrideReason[] = [
  'contractor_offboarded',
  'engagement_changed',
  'regulatory_exemption',
  'temporary_grace_period',
  'admin_correction',
  'other',
];

const MIN_NOTE_CHARS = 20;
const MAX_NOTE_CHARS = 1000;

interface Props {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  itemId: string;
  onSuccess?: () => void;
}

export function OverrideComplianceItemDialog({ open, onOpenChange, itemId, onSuccess }: Props) {
  const t = useTranslations('Contractors.Compliance.Override');
  const queryClient = useQueryClient();
  const [reasonCategory, setReasonCategory] = useState<OverrideReason | ''>('');
  const [reasonNote, setReasonNote] = useState('');

  const mutation = useMutation(
    trpc.compliance.overrideItem.mutationOptions({
      onSuccess: () => {
        toast.success(t('toast.success'));
        queryClient.invalidateQueries({ queryKey: ['compliance'] });
        setReasonCategory('');
        setReasonNote('');
        onOpenChange(false);
        onSuccess?.();
      },
      onError: (err: { message?: string }) => toast.error(err.message ?? t('toast.error')),
    }),
  );

  const isValid =
    reasonCategory !== '' && reasonNote.length >= MIN_NOTE_CHARS && reasonNote.length <= MAX_NOTE_CHARS;

  const handleConfirm = () => {
    if (!isValid) return;
    mutation.mutate({ itemId, reasonCategory, reasonNote });
  };

  const handleOpenChange = (next: boolean) => {
    if (!next) {
      setReasonCategory('');
      setReasonNote('');
    }
    onOpenChange(next);
  };

  return (
    <AlertDialog open={open} onOpenChange={handleOpenChange}>
      <AlertDialogContent>
        <AlertDialogHeader>
          <AlertDialogTitle>{t('title')}</AlertDialogTitle>
          <AlertDialogDescription>{t('description')}</AlertDialogDescription>
        </AlertDialogHeader>
        <div className="space-y-4 py-2">
          <div className="space-y-2">
            <Label htmlFor="override-reason-category">{t('categoryLabel')}</Label>
            <Select value={reasonCategory} onValueChange={(v) => setReasonCategory(v as OverrideReason)}>
              <SelectTrigger id="override-reason-category">
                <SelectValue placeholder={t('categoryPlaceholder')} />
              </SelectTrigger>
              <SelectContent>
                {REASON_OPTIONS.map((opt) => (
                  <SelectItem key={opt} value={opt}>
                    {t(`category.${opt}` as Parameters<typeof t>[0])}
                  </SelectItem>
                ))}
              </SelectContent>
            </Select>
          </div>
          <div className="space-y-2">
            <Label htmlFor="override-reason-note">{t('noteLabel')}</Label>
            <Textarea
              id="override-reason-note"
              value={reasonNote}
              onChange={(e) => setReasonNote(e.target.value)}
              placeholder={t('notePlaceholder')}
              rows={4}
              minLength={MIN_NOTE_CHARS}
              maxLength={MAX_NOTE_CHARS}
            />
            <p className="text-xs text-muted-foreground">
              {t('noteHint', { min: MIN_NOTE_CHARS, current: reasonNote.length })}
            </p>
          </div>
        </div>
        <AlertDialogFooter>
          <AlertDialogCancel>{t('cancel')}</AlertDialogCancel>
          <AlertDialogAction onClick={handleConfirm} disabled={!isValid || mutation.isPending}>
            {mutation.isPending ? t('submitting') : t('submit')}
          </AlertDialogAction>
        </AlertDialogFooter>
      </AlertDialogContent>
    </AlertDialog>
  );
}
```

```tsx
'use client';

// Phase 73 · Plan 08 — Override-button per-row trigger (D-12).

import { useState } from 'react';
import { useTranslations } from 'next-intl';
import { Button } from '@/components/ui/button';
import { OverrideComplianceItemDialog } from './override-compliance-item-dialog';
import { useHasPermission } from '@/hooks/use-permission';

interface Props {
  itemId: string;
  itemStatus: string;
  itemSeverity: string | null;
  onSuccess?: () => void;
}

/**
 * Renders an "Override" button when:
 *   - User has compliance:override permission
 *   - Item has BLOCKING severity
 *   - Item is in MISSING or EXPIRED status (overriding SATISFIED is not the use case)
 *
 * Returns null otherwise — keeps the row chrome clean for healthy items.
 */
export function OverrideComplianceItemButton({ itemId, itemStatus, itemSeverity, onSuccess }: Props) {
  const t = useTranslations('Contractors.Compliance.Override');
  const canOverride = useHasPermission({ compliance: ['override'] });
  const [open, setOpen] = useState(false);

  if (!canOverride) return null;
  if (itemSeverity !== 'BLOCKING') return null;
  if (itemStatus !== 'MISSING' && itemStatus !== 'EXPIRED') return null;

  return (
    <>
      <Button variant="ghost" size="sm" onClick={() => setOpen(true)}>
        {t('button')}
      </Button>
      <OverrideComplianceItemDialog
        open={open}
        onOpenChange={setOpen}
        itemId={itemId}
        onSuccess={onSuccess}
      />
    </>
  );
}
```

```tsx
// Phase 73 · Plan 08 — useHasPermission hook.
// Reads role permissions from the auth session; checks subset against required.

import { authClient } from '@/lib/auth-client';
import type { Permission } from '@contractor-ops/auth';

export function useHasPermission(required: Permission): boolean {
  const session = authClient.useSession();
  const role = (session.data as unknown as { activeRole?: string })?.activeRole;
  if (!role) return false;
  // Defensive: rely on server-side permission check; hide UI optimistically.
  // Owner/admin always have compliance:override. Others fall through to server check.
  if (role === 'owner' || role === 'admin') return true;
  return false;
}
```

```tsx
'use client';

// Phase 73 · Plan 08 — Compliance item History timeline (D-13).

import { useQuery } from '@tanstack/react-query';
import { Activity, ChevronDown, ChevronRight } from 'lucide-react';
import { useTranslations } from 'next-intl';
import { useState } from 'react';
import { Button } from '@/components/ui/button';
import { Skeleton } from '@/components/ui/skeleton';
import { trpc } from '@/trpc/init';

interface Props {
  itemId: string;
}

/**
 * Lazy timeline disclosure — query only fires when expanded.
 * Renders ONLY whitelisted audit-log fields (no raw metadataJson dump).
 */
export function ComplianceItemHistory({ itemId }: Props) {
  const t = useTranslations('Contractors.Compliance.History');
  const [open, setOpen] = useState(false);

  return (
    <div className="space-y-2">
      <Button
        variant="ghost"
        size="sm"
        onClick={() => setOpen((v) => !v)}
        className="text-xs text-muted-foreground"
      >
        {open ? <ChevronDown className="h-4 w-4" aria-hidden /> : <ChevronRight className="h-4 w-4" aria-hidden />}
        <Activity className="h-4 w-4" aria-hidden />
        {t('toggle')}
      </Button>
      {open && <HistoryList itemId={itemId} />}
    </div>
  );
}

function HistoryList({ itemId }: { itemId: string }) {
  const t = useTranslations('Contractors.Compliance.History');
  const { data, isLoading } = useQuery(
    trpc.compliance.itemAuditTrail.queryOptions({ itemId, limit: 50 }),
  );

  if (isLoading) return <Skeleton className="h-[120px] w-full" />;
  if (!data || data.length === 0) {
    return <p className="py-2 text-xs text-muted-foreground">{t('empty')}</p>;
  }

  return (
    <ol className="space-y-2 border-l border-muted pl-4">
      {data.map((entry) => {
        const meta = (entry.metadataJson ?? {}) as Record<string, unknown>;
        const reasonCategory = typeof meta.reasonCategory === 'string' ? meta.reasonCategory : null;
        const previousStatus = typeof meta.previousStatus === 'string' ? meta.previousStatus : null;
        return (
          <li key={entry.id} className="text-xs">
            <p className="font-medium">{t(`action.${entry.action}` as Parameters<typeof t>[0])}</p>
            <p className="text-muted-foreground">
              {t('byOn', {
                actor: entry.actorName ?? t('systemActor'),
                date: new Date(entry.createdAt).toISOString().slice(0, 16).replace('T', ' '),
              })}
            </p>
            {reasonCategory && (
              <p className="mt-1 text-muted-foreground">
                {t('reasonCategoryLabel')}: {t(`reasonCategory.${reasonCategory}` as Parameters<typeof t>[0])}
              </p>
            )}
            {previousStatus && (
              <p className="text-muted-foreground">{t('previousStatusLabel')}: {previousStatus}</p>
            )}
          </li>
        );
      })}
    </ol>
  );
}
```

```tsx
// Phase 73 · Plan 08 — Admin upload-review modal (D-07, D-08).

import { useMutation, useQueryClient } from '@tanstack/react-query';

import { Textarea } from '@/components/ui/textarea';
import { trpc } from '@/trpc/init';

type RejectReason = 'wrong_document_type' | 'illegible' | 'already_expired' | 'forged_or_altered' | 'other';

const REJECT_REASONS: RejectReason[] = ['wrong_document_type', 'illegible', 'already_expired', 'forged_or_altered', 'other'];

interface Props {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  itemId: string;
  documentId: string;
  defaultExpiresAt: string; // ISO yyyy-mm-dd

export function UploadReviewDialog({ open, onOpenChange, itemId, documentId, defaultExpiresAt }: Props) {
  const t = useTranslations('Contractors.Compliance.Upload');
  const queryClient = useQueryClient();
  const [mode, setMode] = useState<'approve' | 'reject'>('approve');
  const [expiresAt, setExpiresAt] = useState(defaultExpiresAt);
  const [rejectReason, setRejectReason] = useState<RejectReason | ''>('');
  const [rejectFreeText, setRejectFreeText] = useState('');

  const approveMutation = useMutation(
    trpc.compliance.approveUploadReplacement.mutationOptions({
      onSuccess: () => {
        toast.success(t('toast.approveSuccess'));
        queryClient.invalidateQueries({ queryKey: ['compliance'] });
        onOpenChange(false);
      },
      onError: (err: { message?: string }) => toast.error(err.message ?? t('toast.error')),
    }),
  );

  const rejectMutation = useMutation(
    trpc.compliance.rejectUploadReplacement.mutationOptions({
      onSuccess: () => {
        toast.success(t('toast.rejectSuccess'));
        queryClient.invalidateQueries({ queryKey: ['compliance'] });
        onOpenChange(false);
      },
      onError: (err: { message?: string }) => toast.error(err.message ?? t('toast.error')),
    }),
  );

  const handleSubmit = () => {
    if (mode === 'approve') {
      approveMutation.mutate({ itemId, documentId, expiresAt: new Date(expiresAt) });
    } else {
      if (!rejectReason) return;
      rejectMutation.mutate({ itemId, documentId, reasonCategory: rejectReason, freeText: rejectFreeText || undefined });
    }
  };

  const isPending = approveMutation.isPending || rejectMutation.isPending;
  const isApproveValid = mode === 'approve' && expiresAt !== '';
  const isRejectValid = mode === 'reject' && rejectReason !== '';

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>{t('title')}</DialogTitle>
          <DialogDescription>{t('description')}</DialogDescription>
        </DialogHeader>
        <div className="flex gap-2 py-2">
          <Button variant={mode === 'approve' ? 'default' : 'outline'} size="sm" onClick={() => setMode('approve')}>
            {t('approve')}
          </Button>
          <Button variant={mode === 'reject' ? 'destructive' : 'outline'} size="sm" onClick={() => setMode('reject')}>
            {t('reject')}
          </Button>
        </div>
        {mode === 'approve' ? (
          <div className="space-y-2">
            <Label htmlFor="approve-expires-at">{t('expiresAtLabel')}</Label>
            <Input
              id="approve-expires-at"
              type="date"
              value={expiresAt}
              onChange={(e) => setExpiresAt(e.target.value)}
            />
            <p className="text-xs text-muted-foreground">{t('expiresAtHint')}</p>
          </div>
        ) : (
          <div className="space-y-3">
            <div className="space-y-2">
              <Label htmlFor="reject-reason">{t('rejectReasonLabel')}</Label>
              <Select value={rejectReason} onValueChange={(v) => setRejectReason(v as RejectReason)}>
                <SelectTrigger id="reject-reason">
                  <SelectValue placeholder={t('rejectReasonPlaceholder')} />
                </SelectTrigger>
                <SelectContent>
                  {REJECT_REASONS.map((r) => (
                    <SelectItem key={r} value={r}>
                      {t(`rejectReason.${r}` as Parameters<typeof t>[0])}
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>
            <div className="space-y-2">
              <Label htmlFor="reject-freetext">{t('rejectFreeTextLabel')}</Label>
              <Textarea
                id="reject-freetext"
                value={rejectFreeText}
                onChange={(e) => setRejectFreeText(e.target.value)}
                rows={3}
                maxLength={500}
              />
            </div>
          </div>
        )}
        <DialogFooter>
          <Button variant="outline" onClick={() => onOpenChange(false)}>
            {t('cancel')}
          </Button>
          <Button
            variant={mode === 'reject' ? 'destructive' : 'default'}

          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
```
