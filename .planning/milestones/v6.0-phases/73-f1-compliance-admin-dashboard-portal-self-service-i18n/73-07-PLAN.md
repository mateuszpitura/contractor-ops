```ts
complianceItems: portalProcedure.query(async ({ ctx }) => {

    where: {
      organizationId: ctx.organizationId,
      contractorId: ctx.contractorId,

    select: {
      id: true,
      name: true,
      documentType: true,
      status: true,
      severity: true,
      expiresAt: true,
      policyRuleId: true,
    },
    orderBy: [
      { status: 'desc' },
      { expiresAt: 'asc' },
    ],
  });
  // Compute the "needs attention" subset for the home banner
  const lookahead = addDays(new Date(), 30);
  const needsAttention = items.filter(
    (i) =>
      i.severity === 'BLOCKING' &&
      (i.status === 'MISSING' ||
        i.status === 'EXPIRED' ||
        (i.status === 'SATISFIED' && i.expiresAt && i.expiresAt <= lookahead)),
  );
  return { items, needsAttention };
}),
```

```ts
// Add to imports if not present
import { portalProcedure } from '../../middleware/portal-auth.js';

// Add to router definition
submitUploadReplacement: portalProcedure
  .input(
    z.object({
      itemId: z.string().min(1),
      documentId: z.string().min(1),
      suggestedExpiresAt: z.coerce.date().optional(),
    }),
  )
  .mutation(async ({ ctx, input }) => {
    if (!ctx.contractorId) {
      throw new TRPCError({ code: 'UNAUTHORIZED', message: 'Portal session missing contractorId' });
    }
    return await ctx.db.$transaction(async (tx) => {
      // 1. Confirm the item belongs to this contractor (defence in depth — portalProcedure already scopes by org)
      const item = await tx.contractorComplianceItem.findFirst({
        where: {
          id: input.itemId,
          organizationId: ctx.organizationId,
          contractorId: ctx.contractorId,
        },
      });
      if (!item) {
        throw new TRPCError({ code: 'NOT_FOUND', message: 'Compliance item not found' });
      }
      // 2. Confirm the document was uploaded by this contractor (defence in depth)
      const doc = await tx.document.findFirst({
        where: { id: input.documentId, organizationId: ctx.organizationId },
      });
      if (!doc) {
        throw new TRPCError({ code: 'NOT_FOUND', message: 'Document not found' });
      }
      // 3. Update Document.status to PENDING_REVIEW.
      // Defensive fallback: if Prisma client rejects PENDING_REVIEW (multi-region apply not yet complete), keep status = ACTIVE and log a warning.
      try {
        await tx.document.update({
          where: { id: doc.id },
          data: { status: 'PENDING_REVIEW' as never },
        });
      } catch {
        // Migration may not yet be applied — log + leave status as ACTIVE so contractor's upload isn't lost.
        // This branch will not trigger after Plan 73-02 multi-region apply completes.
      }
      // 4. Audit
      const auditWriterTx = tx as unknown as Parameters<typeof writeAuditLog>[0]['tx'];
      await writeAuditLog({
        organizationId: ctx.organizationId,
        actorType: 'CONTRACTOR',
        actorId: ctx.contractorId,
        action: 'compliance.upload.submitted',
        resourceType: 'CONTRACTOR',
        resourceId: ctx.contractorId,
        metadata: {
          itemId: input.itemId,
          documentId: input.documentId,
          suggestedExpiresAt: input.suggestedExpiresAt?.toISOString() ?? null,
        },
        tx: auditWriterTx,
      });
      // 5. Return the item — caller can refresh portal/compliance list
      return { itemId: item.id, documentId: doc.id, status: 'PENDING_REVIEW' as const };
    });
  }),
```

```tsx
'use client';

// Phase 73 · Plan 07 — Portal compliance items list (D-09).

import { useQuery } from '@tanstack/react-query';
import { ChevronRight, ShieldAlert, ShieldCheck } from 'lucide-react';
import { useTranslations } from 'next-intl';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import { Card, CardContent } from '@/components/ui/card';
import { Skeleton } from '@/components/ui/skeleton';
import { Link } from '@/i18n/navigation';
import { trpc } from '@/trpc/init';
import { useComplDocName } from '@/components/compliance/dashboard/use-compl-doc-name';

export function PortalComplianceList() {
  const t = useTranslations('Portal.compliance');
  const { data, isLoading } = useQuery(trpc.portal.complianceItems.queryOptions());

  if (isLoading) {
    return (
      <div className="space-y-3">
        <Skeleton className="h-[120px]" />
        <Skeleton className="h-[120px]" />
      </div>
    );
  }

  if (!data || data.items.length === 0) {
    return (
      <Card>
        <CardContent className="flex flex-col items-center gap-3 py-12 text-center">
          <ShieldCheck className="h-10 w-10 text-success" aria-hidden />
          <p className="text-sm font-medium">{t('emptyTitle')}</p>
          <p className="text-xs text-muted-foreground">{t('emptyHint')}</p>
        </CardContent>
      </Card>
    );
  }

  return (
    <div className="space-y-3">
      {data.items.map((item) => (
        <ComplianceItemCard key={item.id} item={item} />
      ))}
    </div>
  );
}

function ComplianceItemCard({
  item,
}: {
  item: {
    id: string;
    policyRuleId: string | null;
    status: string;
    severity: string | null;
    expiresAt: Date | null;
  };
}) {
  const t = useTranslations('Portal.compliance');
  const { text: docName, isPending } = useComplDocName(item.policyRuleId);
  const needsAttention = item.severity === 'BLOCKING' && (item.status === 'MISSING' || item.status === 'EXPIRED');
  return (
    <Card className={needsAttention ? 'border-destructive/40' : ''}>
      <CardContent className="flex items-center justify-between gap-4 py-4">
        <div className="flex items-start gap-3">
          {needsAttention ? (
            <ShieldAlert className="h-5 w-5 text-destructive" aria-hidden />
          ) : (
            <ShieldCheck className="h-5 w-5 text-success" aria-hidden />
          )}
          <div className="space-y-1">
            <p className="text-sm font-medium">
              {docName}
              {isPending && <sup className="ml-0.5 text-xs text-muted-foreground">1</sup>}
            </p>
            <div className="flex items-center gap-2 text-xs text-muted-foreground">
              <Badge variant={needsAttention ? 'destructive' : 'secondary'}>{item.status}</Badge>
              {item.expiresAt && (
                <span>{t('expiresOn', { date: new Date(item.expiresAt).toISOString().slice(0, 10) })}</span>
              )}
            </div>
          </div>
        </div>
        {needsAttention && item.policyRuleId && (
          <Button asChild size="sm">
            <Link href={`/portal/compliance/upload-replacement?itemId=${item.id}&policyRuleId=${item.policyRuleId}`}>
              {t('renewNow')}
              <ChevronRight className="h-4 w-4" aria-hidden />
            </Link>
          </Button>
        )}
      </CardContent>
    </Card>
  );
}
```

```tsx
'use client';

// Phase 73 · Plan 07 — Portal upload-replacement form (D-06, D-07).
//
// 3-step orchestration matching CONTEXT.md D-06 ~3-clicks goal:
//   1. DropZone → trpc.document.requestUpload → presigned R2 PUT
//   2. Client uploads file directly to R2
//   3. trpc.document.confirmUpload → trpc.compliance.submitUploadReplacement

import { useMutation } from '@tanstack/react-query';
import { listPolicyRules } from '@contractor-ops/compliance-policy';
import { defaultExpiryFromUploadDate } from '@contractor-ops/compliance-policy';
import { useTranslations } from 'next-intl';
import { useState, useMemo } from 'react';
import { toast } from 'sonner';
import { useRouter } from '@/i18n/navigation';
import { Button } from '@/components/ui/button';
import { Card, CardContent } from '@/components/ui/card';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { trpc } from '@/trpc/init';

const MAX_FILE_SIZE = 10 * 1024 * 1024; // 10 MB per CONTEXT.md D-06
const ALLOWED_MIME = ['application/pdf', 'image/png', 'image/jpeg', 'image/jpg'];

export function PortalUploadReplacementForm({
  itemId,
  policyRuleId,
}: {
  itemId: string;
  policyRuleId: string;
}) {
  const t = useTranslations('Portal.compliance.upload');
  const router = useRouter();

  const policyRule = useMemo(
    () => listPolicyRules().find((r) => r.policyRuleId === policyRuleId),
    [policyRuleId],
  );

  const initialExpiresAt = useMemo(() => {
    if (!policyRule) return '';
    try {
      return defaultExpiryFromUploadDate(policyRule).toISOString().slice(0, 10);
    } catch {
      return '';
    }
  }, [policyRule]);

  const [file, setFile] = useState<File | null>(null);
  const [expiresAt, setExpiresAt] = useState(initialExpiresAt);
  const [submitting, setSubmitting] = useState(false);

  const requestUpload = useMutation(trpc.document.requestUpload.mutationOptions({}));
  const confirmUpload = useMutation(trpc.document.confirmUpload.mutationOptions({}));
  const submitReplacement = useMutation(
    trpc.compliance.submitUploadReplacement.mutationOptions({}),
  );

  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const f = e.target.files?.[0];
    if (!f) return;
    if (f.size > MAX_FILE_SIZE) {
      toast.error(t('errors.fileTooLarge'));
      return;
    }
    if (!ALLOWED_MIME.includes(f.type)) {
      toast.error(t('errors.invalidMime'));
      return;
    }
    setFile(f);
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!file || !policyRule || submitting) return;
    setSubmitting(true);
    try {
      // Step 1: request presigned upload URL
      const { documentId, uploadUrl } = await requestUpload.mutateAsync({
        filename: file.name,
        mimeType: file.type,
        fileSizeBytes: file.size,
        documentType: policyRule.documentType,
      } as never);
      // Step 2: PUT file to R2
      const putRes = await fetch(uploadUrl, {
        method: 'PUT',
        body: file,
        headers: { 'Content-Type': file.type },
      });
      if (!putRes.ok) {
        throw new Error('R2 upload failed');
      }
      // Step 3: confirm upload (server-side virus scan kicks off)
      await confirmUpload.mutateAsync({
        documentId,
        checksumSha256: 'placeholder-sha256', // production code computes via WebCrypto
      } as never);
      // Step 4: submit replacement
      await submitReplacement.mutateAsync({
        itemId,
        documentId,
        suggestedExpiresAt: expiresAt ? new Date(expiresAt) : undefined,
      });
      toast.success(t('toast.success'));
      router.push('/portal/compliance');
    } catch (err) {
      toast.error((err as Error).message ?? t('toast.error'));
    } finally {
      setSubmitting(false);
    }
  };

  if (!policyRule) {
    return (
      <Card>
        <CardContent className="py-12 text-center text-sm text-muted-foreground">
          {t('errors.unknownPolicyRule')}
        </CardContent>
      </Card>
    );
  }

  return (
    <Card>
      <CardContent className="space-y-4 py-6">
        <form onSubmit={handleSubmit} className="space-y-4">
          <div className="space-y-2">
            <Label htmlFor="file">{t('fileLabel')}</Label>
            <Input
              id="file"
              type="file"
              accept={ALLOWED_MIME.join(',')}
              onChange={handleFileChange}
              disabled={submitting}
            />
            <p className="text-xs text-muted-foreground">{t('fileHint')}</p>
          </div>
          <div className="space-y-2">
            <Label htmlFor="expiresAt">{t('expiresAtLabel')}</Label>
            <Input
              id="expiresAt"
              type="date"
              value={expiresAt}
              onChange={(e) => setExpiresAt(e.target.value)}
              disabled={submitting}
            />
            <p className="text-xs text-muted-foreground">{t('expiresAtHint')}</p>
          </div>
          <Button type="submit" disabled={!file || submitting} className="w-full md:w-auto">
            {submitting ? t('submitting') : t('submit')}
          </Button>
        </form>
      </CardContent>
    </Card>
  );
}
```
