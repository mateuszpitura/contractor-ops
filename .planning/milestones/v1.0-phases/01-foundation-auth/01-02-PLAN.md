    **packages/api/src/middleware/sensitive.ts:** Sensitive action re-authentication middleware per locked decision ("Re-authentication required for sensitive actions — payment runs, role changes, settings changes").
    ```typescript
    import { TRPCError } from "@trpc/server";
    import { t } from "../init";

    const SENSITIVE_ACTION_MAX_AGE_MS = 5 * 60 * 1000; // 5 minutes

    /**
     * Middleware that checks if the current session was created recently enough
     * to perform sensitive actions. If the session's createdAt timestamp is older
     * than SENSITIVE_ACTION_MAX_AGE_MS, the user must re-authenticate.
     *
     * Throws FORBIDDEN with code "REAUTH_REQUIRED" so the client can prompt
     * the user to re-enter their password.
     */
    export const sensitiveActionMiddleware = t.middleware(async ({ ctx, next }) => {
      const sessionCreatedAt = new Date(ctx.session.session.createdAt).getTime();
      const now = Date.now();

      if (now - sessionCreatedAt > SENSITIVE_ACTION_MAX_AGE_MS) {
        throw new TRPCError({
          code: "FORBIDDEN",
          message: "This action requires re-authentication. Please sign in again to continue.",
          cause: "REAUTH_REQUIRED",
        });
      }

      return next({ ctx });
    });

    /**
     * Procedure that requires both tenant scope and fresh session.
     * Use for: role changes, user deactivation/reactivation, settings updates.
     * Chain: auth -> tenant -> sensitive -> handler
     */
    export const sensitiveActionProcedure = tenantProcedure.use(sensitiveActionMiddleware);
    ```
    Import `tenantProcedure` from "./tenant". Export both `sensitiveActionMiddleware` and `sensitiveActionProcedure`.

    NOTE: The `ctx.session.session.createdAt` path comes from Better Auth's session object shape. If Better Auth stores it differently, adapt accordingly — the key requirement is checking when the session was established, not when it was last refreshed.
